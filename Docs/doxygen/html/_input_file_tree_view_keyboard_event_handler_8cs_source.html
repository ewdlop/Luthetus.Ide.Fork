54 public override Task OnKeyDownAsync(TreeViewCommandArgs commandArgs)
55 {
56 base.OnKeyDownAsync(commandArgs);
57
58 if (commandArgs.KeyboardEventArgs is null)
59 return Task.CompletedTask;
60
61 switch (commandArgs.KeyboardEventArgs.Code)
62 {
63 case KeyboardKeyFacts.WhitespaceCodes.ENTER_CODE:
64 SetInputFileContentTreeViewRoot(commandArgs);
65 return Task.CompletedTask;
66 case KeyboardKeyFacts.WhitespaceCodes.SPACE_CODE:
67 SetSelectedTreeViewModel(commandArgs);
68 return Task.CompletedTask;
69 }
70
71 switch (commandArgs.KeyboardEventArgs.Key)
72 {
73 // Tried to have { "Ctrl" + "f" } => MoveFocusToSearchBar however, the webview was ending up taking over
74 // and displaying its search bar with focus being set to it.
75 //
76 // Doing preventDefault just for this one case would be a can of worms as JSInterop is needed, as well a custom Blazor event.
77 case "/":
78 case "?":
79 MoveFocusToSearchBar(commandArgs);
80 return Task.CompletedTask;
81 // TODO: Add move to next match and move to previous match
82 //
83 // case "*":
84 // treeViewCommand = new TreeViewCommand(SetNextMatchAsActiveTreeViewNode);
85 // return Task.CompletedTask true;
86 // case "#":
87 // treeViewCommand = new TreeViewCommand(SetPreviousMatchAsActiveTreeViewNode);
88 // return Task.CompletedTask true;
89 }
90
91 if (commandArgs.KeyboardEventArgs.AltKey)
92 {
93 AltModifiedKeymap(commandArgs);
94 return Task.CompletedTask;
95 }
96
97 return Task.CompletedTask;
98 }

100 private void AltModifiedKeymap(TreeViewCommandArgs commandArgs)
101 {
102 if (commandArgs.KeyboardEventArgs is null)
103 return;
104
105 switch (commandArgs.KeyboardEventArgs.Key)
106 {
107 case KeyboardKeyFacts.MovementKeys.ARROW_LEFT:
108 HandleBackButtonOnClick(commandArgs);
109 break;
110 case KeyboardKeyFacts.MovementKeys.ARROW_UP:
111 HandleUpwardButtonOnClick(commandArgs);
112 break;
113 case KeyboardKeyFacts.MovementKeys.ARROW_RIGHT:
114 HandleForwardButtonOnClick(commandArgs);
115 break;
116 case "r":
117 HandleRefreshButtonOnClick(commandArgs);
118 break;
119 }
120 }
