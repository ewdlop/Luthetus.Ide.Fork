89 private async Task StartNewDotNetSolutionCommandOnClick()
90 {
91 var localFormattedCommand = FormattedCommand;
92 var localSolutionName = _solutionName;
93 var localParentDirectoryName = _parentDirectoryName;
94
95 if (string.IsNullOrWhiteSpace(localSolutionName) ||
96 string.IsNullOrWhiteSpace(localParentDirectoryName))
97 {
98 return;
99 }
100
101 if (LuthetusHostingInformation.LuthetusHostingKind != LuthetusHostingKind.Photino)
102 {
103 await HackForWebsite_StartNewDotNetSolutionCommandOnClick(
104 localSolutionName,
105 localParentDirectoryName)
106 .ConfigureAwait(false);
107 }
108 else
109 {
110 var terminalCommandRequest = new TerminalCommandRequest(
111 localFormattedCommand.Value,
112 _parentDirectoryName,
113 new Key<TerminalCommandRequest>(NewDotNetSolutionTerminalCommandRequestKey.Guid))
114 {
115 ContinueWithFunc = parsedCommand =>
116 {
117 // Close Dialog
118 Dispatcher.Dispatch(new DialogState.DisposeAction(DialogRecord.DynamicViewModelKey));
119
120 // Open the created .NET Solution
121 var parentDirectoryAbsolutePath = EnvironmentProvider.AbsolutePathFactory(
122 localParentDirectoryName,
123 true);
124
125 var solutionAbsolutePathString =
126 parentDirectoryAbsolutePath.Value +
127 localSolutionName +
128 EnvironmentProvider.DirectorySeparatorChar +
129 localSolutionName +
130 '.' +
131 ExtensionNoPeriodFacts.DOT_NET_SOLUTION;
132
133 var solutionAbsolutePath = EnvironmentProvider.AbsolutePathFactory(
134 solutionAbsolutePathString,
135 false);
136
137 CompilerServicesBackgroundTaskApi.DotNetSolution.SetDotNetSolution(solutionAbsolutePath);
138 return Task.CompletedTask;
139 }
140 };
141
142 TerminalStateWrap.Value.TerminalMap[TerminalFacts.GENERAL_KEY].EnqueueCommand(terminalCommandRequest);
143 }
144 }

146 private async Task HackForWebsite_StartNewDotNetSolutionCommandOnClick(
147 string localSolutionName,
148 string localParentDirectoryName)
149 {
150 var directoryContainingSolution =
151 EnvironmentProvider.JoinPaths(localParentDirectoryName, localSolutionName) +
152 EnvironmentProvider.DirectorySeparatorChar;
153
154 await FileSystemProvider.Directory
155 .CreateDirectoryAsync(directoryContainingSolution)
156 .ConfigureAwait(false);
157
158 var localSolutionFilenameWithExtension =
159 localSolutionName +
160 '.' +
161 ExtensionNoPeriodFacts.DOT_NET_SOLUTION;
162
163 var solutionAbsolutePathString = EnvironmentProvider.JoinPaths(
164 directoryContainingSolution,
165 localSolutionFilenameWithExtension);
166
167 await FileSystemProvider.File.WriteAllTextAsync(
168 solutionAbsolutePathString,
169 HackForWebsite_NEW_SOLUTION_TEMPLATE)
170 .ConfigureAwait(false);
171
172 // Close Dialog
173 Dispatcher.Dispatch(new DialogState.DisposeAction(DialogRecord.DynamicViewModelKey));
174
175 NotificationHelper.DispatchInformative("Website .sln template was used", "No terminal available", CommonComponentRenderers, Dispatcher, TimeSpan.FromSeconds(7));
176
177 var solutionAbsolutePath = EnvironmentProvider.AbsolutePathFactory(
178 solutionAbsolutePathString,
179 false);
180
181 CompilerServicesBackgroundTaskApi.DotNetSolution.SetDotNetSolution(solutionAbsolutePath);
182 }
