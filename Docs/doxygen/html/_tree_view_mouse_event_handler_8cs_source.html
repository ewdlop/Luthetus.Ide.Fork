101 protected virtual void OnMouseDown(TreeViewCommandArgs commandArgs)
102 {
103 if (commandArgs.NodeThatReceivedMouseEvent is null || commandArgs.MouseEventArgs is null)
104 return;
105
106 if ((commandArgs.MouseEventArgs.Buttons & 1) == 1) // Left Click
107 {
108 // This boolean asks: Should I ADD this TargetNode to the list of selected nodes,
109 // OR
110 // Should I CLEAR the list of selected nodes and make this TargetNode the only entry.
111 var addSelectedNodes = commandArgs.MouseEventArgs.CtrlKey || commandArgs.MouseEventArgs.ShiftKey;
112
113 // This boolean asks: Should I ALSO SELECT the nodes between the currentNode and the targetNode.
114 var selectNodesBetweenCurrentAndNextActiveNode = commandArgs.MouseEventArgs.ShiftKey;
115
116 TreeViewService.SetActiveNode(
117 commandArgs.TreeViewContainer.Key,
118 commandArgs.NodeThatReceivedMouseEvent,
119 addSelectedNodes,
120 selectNodesBetweenCurrentAndNextActiveNode);
121 }
122 else // Presume Right Click or Context Menu
123 {
124 if (commandArgs.MouseEventArgs.CtrlKey)
125 {
126 // Open context menu, but do not move the active node, regardless who the TargetNode is
127 }
128 else
129 {
130 var targetNodeAlreadySelected = commandArgs.TreeViewContainer.SelectedNodeList.Any(x => x.Key == commandArgs.NodeThatReceivedMouseEvent.Key);
131
132 if (targetNodeAlreadySelected)
133 {
134 // Open context menu, but do not move the active node, regardless who the TargetNode is
135 }
136 else
137 {
138 // Move the active node, and open context menu
139 TreeViewService.SetActiveNode(
140 commandArgs.TreeViewContainer.Key,
141 commandArgs.NodeThatReceivedMouseEvent,
142 false,
143 false);
144 }
145 }
146 }
147 }
