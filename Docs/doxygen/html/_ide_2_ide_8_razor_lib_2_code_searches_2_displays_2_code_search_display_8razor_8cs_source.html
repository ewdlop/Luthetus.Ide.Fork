138 private async Task UpdateContent()
139 {
140 Console.WriteLine(nameof(UpdateContent));
141
142 if (!TreeViewService.TryGetTreeViewContainer(
143 CodeSearchState.TreeViewCodeSearchContainerKey,
144 out var treeViewContainer))
145 {
146 Console.WriteLine("TryGetTreeViewContainer");
147 return;
148 }
149
150 if (treeViewContainer.SelectedNodeList.Count > 1)
151 {
152 Console.WriteLine("treeViewContainer.SelectedNodeList.Count > 1");
153 return;
154 }
155
156 var activeNode = treeViewContainer.ActiveNode;
157
158 if (activeNode is not TreeViewCodeSearchTextSpan treeViewCodeSearchTextSpan)
159 {
160 Console.WriteLine("activeNode is not TreeViewCodeSearchTextSpan treeViewCodeSearchTextSpan");
161 return;
162 }
163
164 var inPreviewViewModelKey = CodeSearchStateWrap.Value.PreviewViewModelKey;
165 var outPreviewViewModelKey = Key<TextEditorViewModel>.NewKey();
166
167 var filePath = treeViewCodeSearchTextSpan.Item.ResourceUri.Value;
168 var resourceUri = treeViewCodeSearchTextSpan.Item.ResourceUri;
169
170 if (TextEditorConfig.RegisterModelFunc is null)
171 return;
172
173 await TextEditorConfig.RegisterModelFunc.Invoke(
174 new RegisterModelArgs(resourceUri, ServiceProvider))
175 .ConfigureAwait(false);
176
177 if (TextEditorConfig.TryRegisterViewModelFunc is not null)
178 {
179 var viewModelKey = await TextEditorConfig.TryRegisterViewModelFunc.Invoke(new TryRegisterViewModelArgs(
180 outPreviewViewModelKey,
181 resourceUri,
182 new Category(nameof(CodeSearchDisplay)),
183 false,
184 ServiceProvider))
185 .ConfigureAwait(false);
186
187 if (viewModelKey != Key<TextEditorViewModel>.Empty &&
188 TextEditorConfig.TryShowViewModelFunc is not null)
189 {
190 Dispatcher.Dispatch(new CodeSearchState.WithAction(inState => inState with
191 {
192 PreviewFilePath = filePath,
193 PreviewViewModelKey = viewModelKey,
194 }));
195
196 if (inPreviewViewModelKey != Key<TextEditorViewModel>.Empty &&
197 inPreviewViewModelKey != viewModelKey)
198 {
199 TextEditorService.ViewModelApi.Dispose(inPreviewViewModelKey);
200 }
201 }
202 }
203 }
