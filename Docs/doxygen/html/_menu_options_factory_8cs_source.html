188 private void PerformNewFile(
189 string fileName,
190 IFileTemplate? exactMatchFileTemplate,
191 ImmutableArray<IFileTemplate> relatedMatchFileTemplatesList,
192 NamespacePath namespacePath,
193 Func<Task> onAfterCompletion)
194 {
195 _backgroundTaskService.Enqueue(
196 Key<IBackgroundTask>.NewKey(),
197 ContinuousBackgroundTaskWorker.GetQueueKey(),
198 "New File Action",
199 async () =>
200 {
201 if (exactMatchFileTemplate is null)
202 {
203 var emptyFileAbsolutePathString = namespacePath.AbsolutePath.Value + fileName;
204
205 var emptyFileAbsolutePath = _environmentProvider.AbsolutePathFactory(
206 emptyFileAbsolutePathString,
207 false);
208
209 await _fileSystemProvider.File.WriteAllTextAsync(
210 emptyFileAbsolutePath.Value,
211 string.Empty,
212 CancellationToken.None)
213 .ConfigureAwait(false);
214 }
215 else
216 {
217 var allTemplates = new[] { exactMatchFileTemplate }
218 .Union(relatedMatchFileTemplatesList)
219 .ToArray();
220
221 foreach (var fileTemplate in allTemplates)
222 {
223 var templateResult = fileTemplate.ConstructFileContents.Invoke(
224 new FileTemplateParameter(fileName, namespacePath, _environmentProvider));
225
226 await _fileSystemProvider.File.WriteAllTextAsync(
227 templateResult.FileNamespacePath.AbsolutePath.Value,
228 templateResult.Contents,
229 CancellationToken.None)
230 .ConfigureAwait(false);
231 }
232 }
233
234 await onAfterCompletion.Invoke().ConfigureAwait(false);
235 });
236 }

321 private void PerformPasteFile(IAbsolutePath receivingDirectory, Func<Task> onAfterCompletion)
322 {
323 _backgroundTaskService.Enqueue(
324 Key<IBackgroundTask>.NewKey(),
325 ContinuousBackgroundTaskWorker.GetQueueKey(),
326 "Paste File Action",
327 async () =>
328 {
329 var clipboardContents = await _clipboardService.ReadClipboard().ConfigureAwait(false);
330
331 if (ClipboardFacts.TryParseString(clipboardContents, out var clipboardPhrase))
332 {
333 if (clipboardPhrase is not null &&
334 clipboardPhrase.DataType == ClipboardFacts.AbsolutePathDataType)
335 {
336 if (clipboardPhrase.Command == ClipboardFacts.CopyCommand ||
337 clipboardPhrase.Command == ClipboardFacts.CutCommand)
338 {
339 IAbsolutePath? clipboardAbsolutePath = null;
340
341 // Should the if and else if be kept as inline awaits?
342 // If kept as inline awaits then the else if won't execute if the first one succeeds.
343 if (await _fileSystemProvider.Directory.ExistsAsync(clipboardPhrase.Value).ConfigureAwait(false))
344 {
345 clipboardAbsolutePath = _environmentProvider.AbsolutePathFactory(
346 clipboardPhrase.Value,
347 true);
348 }
349 else if (await _fileSystemProvider.File.ExistsAsync(clipboardPhrase.Value).ConfigureAwait(false))
350 {
351 clipboardAbsolutePath = _environmentProvider.AbsolutePathFactory(
352 clipboardPhrase.Value,
353 false);
354 }
355
356 if (clipboardAbsolutePath is not null)
357 {
358 var successfullyPasted = true;
359
360 try
361 {
362 if (clipboardAbsolutePath.IsDirectory)
363 {
364 var clipboardDirectoryInfo = new DirectoryInfo(clipboardAbsolutePath.Value);
365 var receivingDirectoryInfo = new DirectoryInfo(receivingDirectory.Value);
366
367 CopyFilesRecursively(clipboardDirectoryInfo, receivingDirectoryInfo);
368 }
369 else
370 {
371 var destinationAbsolutePathString = receivingDirectory.Value +
372 clipboardAbsolutePath.NameWithExtension;
373
374 var sourceAbsolutePathString = clipboardAbsolutePath.Value;
375
376 await _fileSystemProvider.File.CopyAsync(
377 sourceAbsolutePathString,
378 destinationAbsolutePathString)
379 .ConfigureAwait(false);
380 }
381 }
382 catch (Exception)
383 {
384 successfullyPasted = false;
385 }
386
387 if (successfullyPasted && clipboardPhrase.Command == ClipboardFacts.CutCommand)
388 {
389 // TODO: Rerender the parent of the deleted due to cut file
390 PerformDeleteFile(clipboardAbsolutePath, onAfterCompletion);
391 }
392 else
393 {
394 await onAfterCompletion.Invoke().ConfigureAwait(false);
395 }
396 }
397 }
398 }
399 }
400 });
401 }

403 private IAbsolutePath? PerformRename(IAbsolutePath sourceAbsolutePath, string nextName, IDispatcher dispatcher, Func<Task> onAfterCompletion)
404 {
405 // Check if the current and next name match when compared with case insensitivity
406 if (0 == string.Compare(sourceAbsolutePath.NameWithExtension, nextName, StringComparison.OrdinalIgnoreCase))
407 {
408 var temporaryNextName = _environmentProvider.GetRandomFileName();
409
410 var temporaryRenameResult = PerformRename(
411 sourceAbsolutePath,
412 temporaryNextName,
413 dispatcher,
414 () => Task.CompletedTask);
415
416 if (temporaryRenameResult is null)
417 {
418 onAfterCompletion.Invoke();
419 return null;
420 }
421 else
422 {
423 sourceAbsolutePath = temporaryRenameResult;
424 }
425 }
426
427 var sourceAbsolutePathString = sourceAbsolutePath.Value;
428 var parentOfSource = sourceAbsolutePath.AncestorDirectoryList.Last().Value;
429 var destinationAbsolutePathString = parentOfSource + nextName;
430
431 try
432 {
433 if (sourceAbsolutePath.IsDirectory)
434 _fileSystemProvider.Directory.MoveAsync(sourceAbsolutePathString, destinationAbsolutePathString);
435 else
436 _fileSystemProvider.File.MoveAsync(sourceAbsolutePathString, destinationAbsolutePathString);
437 }
438 catch (Exception e)
439 {
440 NotificationHelper.DispatchError("Rename Action", e.Message, _commonComponentRenderers, dispatcher, TimeSpan.FromSeconds(14));
441 onAfterCompletion.Invoke();
442 return null;
443 }
444
445 onAfterCompletion.Invoke();
446
447 return _environmentProvider.AbsolutePathFactory(destinationAbsolutePathString, sourceAbsolutePath.IsDirectory);
448 }

454 public static DirectoryInfo CopyFilesRecursively(DirectoryInfo source, DirectoryInfo target)
455 {
456 var newDirectoryInfo = target.CreateSubdirectory(source.Name);
457 foreach (var fileInfo in source.GetFiles())
458 fileInfo.CopyTo(Path.Combine(newDirectoryInfo.FullName, fileInfo.Name));
459
460 foreach (var childDirectoryInfo in source.GetDirectories())
461 CopyFilesRecursively(childDirectoryInfo, newDirectoryInfo);
462
463 return newDirectoryInfo;
464 }
