65 [ReducerMethod]
66 public static TextEditorState ReduceRegisterViewModelAction(
67 TextEditorState inState,
68 RegisterViewModelAction registerViewModelAction)
69 {
70 // The category and ViewModelKey do NOT need to be a compound unique identifier
71 // Only check for the 'ViewModelKey' already existing.
72 //
73 // Category is just a way to filter a list of view models.
74 //
75 // TODO: What is the difference between Category and Group? I'm asking this to myself. If their redundant then get rid of one. Otherwise...
76 // ...write down the justification for both existing before you forget again.
77 //
78 // I think I made both Category and Group because:
79 // Category describes relationships between view models
80 //
81 // Group is solely meant to provide tab UI.
82 // - One can put a view model of any category into a group.
83 // - Or one could add dropzone logic that validates the category of a 'being dragged view model'
84 // to ensure it belongs in that group.
85
86 var inViewModel = inState.ViewModelGetOrDefault(registerViewModelAction.ViewModelKey);
87
88 if (inViewModel is not null)
89 return inState;
90
91 if (registerViewModelAction.ViewModelKey == Key<TextEditorViewModel>.Empty)
92 throw new InvalidOperationException($"Provided {nameof(Key<TextEditorViewModel>)} cannot be {nameof(Key<TextEditorViewModel>)}.{Key<TextEditorViewModel>.Empty}");
93
94 var viewModel = new TextEditorViewModel(
95 registerViewModelAction.ViewModelKey,
96 registerViewModelAction.ResourceUri,
97 registerViewModelAction.TextEditorService,
98 registerViewModelAction.Dispatcher,
99 registerViewModelAction.DialogService,
100 registerViewModelAction.JsRuntime,
101 VirtualizationResult<RichCharacter[]>.GetEmptyRichCharacters(),
102 new TextEditorDimensions(0, 0, 0, 0),
103 new ScrollbarDimensions(0, 0, 0, 0, 0),
104 new CharAndLineMeasurements(0, 0),
105 false,
106 registerViewModelAction.Category);
107
108 inState._viewModelMap.Add(viewModel.ViewModelKey, viewModel);
109
110 return inState with {};
111 }

166 [ReducerMethod]
167 public static TextEditorState ReduceSetModelAndViewModelRangeAction(
168 TextEditorState inState,
169 SetModelAndViewModelRangeAction setModelAndViewModelRangeAction)
170 {
171 // Models
172 foreach (var modelModifier in setModelAndViewModelRangeAction.ModelModifierList)
173 {
174 // Enumeration was modified shouldn't occur here because only the reducer
175 // should be adding or removing, and the reducer is thread safe.
176 var exists = inState._modelMap.TryGetValue(
177 modelModifier.ResourceUri, out var inModel);
178
179 if (!exists)
180 continue;
181
182 inState._modelMap[modelModifier.ResourceUri] = modelModifier.ToModel();
183 }
184
185 // ViewModels
186 foreach (var viewModelModifier in setModelAndViewModelRangeAction.ViewModelModifierList)
187 {
188 // Enumeration was modified shouldn't occur here because only the reducer
189 // should be adding or removing, and the reducer is thread safe.
190 var exists = inState._viewModelMap.TryGetValue(
191 viewModelModifier.ViewModel.ViewModelKey, out var inViewModel);
192
193 if (!exists)
194 continue;
195
196 inState._viewModelMap[viewModelModifier.ViewModel.ViewModelKey] = viewModelModifier.ViewModel;
