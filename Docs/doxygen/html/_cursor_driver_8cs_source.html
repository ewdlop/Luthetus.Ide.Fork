66 public string GetCursorStyleCss()
67 {
68 var renderBatchLocal = _renderBatch;
69 if (renderBatchLocal is null)
70 return string.Empty;
71
72 try
73 {
74 var measurements = renderBatchLocal.ViewModel.CharAndLineMeasurements;
75
76 var leftInPixels = 0d;
77
78 // Tab key column offset
79 {
80 var tabsOnSameRowBeforeCursor = renderBatchLocal.Model.GetTabCountOnSameLineBeforeCursor(
81 renderBatchLocal.ViewModel.PrimaryCursor.LineIndex,
82 renderBatchLocal.ViewModel.PrimaryCursor.ColumnIndex);
83
84 // 1 of the character width is already accounted for
85
86 var extraWidthPerTabKey = TextEditorModel.TAB_WIDTH - 1;
87
88 leftInPixels += extraWidthPerTabKey *
89 tabsOnSameRowBeforeCursor *
90 measurements.CharacterWidth;
91 }
92
93 leftInPixels += measurements.CharacterWidth * renderBatchLocal.ViewModel.PrimaryCursor.ColumnIndex;
94
95 var leftInPixelsInvariantCulture = leftInPixels.ToCssValue();
96 var left = $"left: {leftInPixelsInvariantCulture}px;";
97
98 var topInPixelsInvariantCulture = (measurements.LineHeight * renderBatchLocal.ViewModel.PrimaryCursor.LineIndex)
99 .ToCssValue();
100
101 var top = $"top: {topInPixelsInvariantCulture}px;";
102
103 var heightInPixelsInvariantCulture = measurements.LineHeight.ToCssValue();
104 var height = $"height: {heightInPixelsInvariantCulture}px;";
105
106 var widthInPixelsInvariantCulture = renderBatchLocal.Options.CursorWidthInPixels.ToCssValue();
107 var width = $"width: {widthInPixelsInvariantCulture}px;";
108
109 var keymapStyling = ((ITextEditorKeymap)renderBatchLocal.Options.Keymap).GetCursorCssStyleString(
110 renderBatchLocal.Model,
111 renderBatchLocal.ViewModel,
112 renderBatchLocal.Options);
113
114 // This feels a bit hacky, exceptions are happening because the UI isn't accessing
115 // the text editor in a thread safe way.
116 //
117 // When an exception does occur though, the cursor should receive a 'text editor changed'
118 // event and re-render anyhow however.
119 //
120 // So store the result of this method incase an exception occurs in future invocations,
121 // to keep the cursor on screen while the state works itself out.
122 return _previousGetCursorStyleCss = $"{left} {top} {height} {width} {keymapStyling}";
123 }
124 catch (LuthetusTextEditorException)
125 {
126 return _previousGetCursorStyleCss;
127 }
128 }
