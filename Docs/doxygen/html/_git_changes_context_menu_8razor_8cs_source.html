60 private async Task<MenuRecord> GetMenuRecord(TreeViewCommandArgs commandArgs, bool isRecursiveCall = false)
61 {
62 if (!isRecursiveCall && _previousGetMenuRecordInvocation.treeViewCommandArgs == commandArgs)
63 return _previousGetMenuRecordInvocation.menuRecord;
64
65 if (!isRecursiveCall && commandArgs.TreeViewContainer.SelectedNodeList.Count > 1)
66 {
