119 // KeywordToken, OpenParenthesisToken, InitializationSyntaxList.Length, InitializationStatementDelimiterToken,
120 // ConditionExpressionNode, ConditionStatementDelimiterToken, UpdationExpressionNode, CloseParenthesisToken,
121 var childCount =
122 1 + // KeywordToken,
123 1 + // OpenParenthesisToken,
124 InitializationSyntaxList.Length + // InitializationSyntaxList.Length
125 1 + // InitializationStatementDelimiterToken,
126 1 + // ConditionExpressionNode,
127 1 + // ConditionStatementDelimiterToken,
128 1 + // UpdationExpressionNode,
129 1; // CloseParenthesisToken,
130
131 if (OpenBraceToken.ConstructorWasInvoked)
132 childCount++;
133 if (CodeBlockNode is not null)
134 childCount++;
135
136 var childList = new ISyntax[childCount];
137 var i = 0;
138
139 childList[i++] = KeywordToken;
140 childList[i++] = OpenParenthesisToken;
141 foreach (var item in InitializationSyntaxList)
142 {
143 childList[i++] = item;
144 }
145 childList[i++] = InitializationStatementDelimiterToken;
146 childList[i++] = ConditionExpressionNode;
147 childList[i++] = ConditionStatementDelimiterToken;
148 childList[i++] = UpdationExpressionNode;
149 childList[i++] = CloseParenthesisToken;
150 if (OpenBraceToken.ConstructorWasInvoked)
151 childList[i++] = OpenBraceToken;
152 if (CodeBlockNode is not null)
153 childList[i++] = CodeBlockNode;
154
155 _childList = childList;
156
157 _childListIsDirty = false;
