111 private async Task HandleTreeViewOnContextMenu(
112 MouseEventArgs? mouseEventArgs,
113 Key<TreeViewContainer> treeViewContainerKey,
114 TreeViewNoType? treeViewMouseWasOver)
115 {
116 if (treeViewContainerKey == Key<TreeViewContainer>.Empty || mouseEventArgs is null)
117 return;
118
119 var treeViewContainer = TreeViewStateSelection.Value;
120 // Validate that the treeViewContainer did not change out from under us
121 if (treeViewContainer is null || treeViewContainer.Key != treeViewContainerKey)
122 return;
123
124 ContextMenuFixedPosition contextMenuFixedPosition;
125 TreeViewNoType contextMenuTargetTreeViewNoType;
126
127 if (mouseEventArgs.Button == -1) // -1 here means ContextMenu event was from keyboard
128 {
129 if (treeViewContainer.ActiveNode is null)
130 return;
131
132 // If dedicated context menu button or shift + F10 was pressed as opposed to
133 // a mouse RightClick then use JavaScript to determine the ContextMenu position.
134 contextMenuFixedPosition = await JsRuntime.GetLuthetusCommonApi()
135 .GetTreeViewContextMenuFixedPosition(TreeViewService.GetNodeElementId(treeViewContainer.ActiveNode))
136 .ConfigureAwait(false);
137
138 contextMenuTargetTreeViewNoType = treeViewContainer.ActiveNode;
139 }
140 else
141 {
142 // If a mouse RightClick caused the event then
143 // use the MouseEventArgs to determine the ContextMenu position
144 if (treeViewMouseWasOver is null)
145 {
146 // 'whitespace' of the TreeView was right clicked as opposed to
147 // a TreeView node and the event should be ignored.
148 return;
149 }
150
151 contextMenuFixedPosition = new ContextMenuFixedPosition(
152 true,
153 mouseEventArgs.ClientX,
154 mouseEventArgs.ClientY);
155
156 contextMenuTargetTreeViewNoType = treeViewMouseWasOver;
157 }
158
159 _treeViewContextMenuCommandArgs = new TreeViewCommandArgs(
160 TreeViewService,
161 treeViewContainer,
162 contextMenuTargetTreeViewNoType,
163 async () =>
164 {
165 _treeViewContextMenuCommandArgs = null;
166 await InvokeAsync(StateHasChanged);
167
168 var localTreeViewStateDisplayElementReference = _treeViewStateDisplayElementReference;
169
170 try

184 // on an ElementReference.
185 }
186 },
187 contextMenuFixedPosition,
188 mouseEventArgs,
189 null);
190
191 if (OnContextMenuFunc is not null)
192 {
193 BackgroundTaskService.Enqueue(
194 Key<IBackgroundTask>.NewKey(),
195 ContinuousBackgroundTaskWorker.GetQueueKey(),
196 "TreeView.HandleTreeViewOnContextMenu",
197 async () => await OnContextMenuFunc
198 .Invoke(_treeViewContextMenuCommandArgs)
199 .ConfigureAwait(false));
200 }
201
202 await InvokeAsync(StateHasChanged);
203 }
