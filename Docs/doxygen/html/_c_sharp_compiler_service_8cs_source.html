203 private Task PropSnippet(string word, TextEditorTextSpan textSpan, string textToInsert)
204 {
205 _textEditorService.PostUnique(
206 nameof(PropSnippet),
207 editContext =>
208 {
209 var modelModifier = editContext.GetModelModifier(textSpan.ResourceUri);
210
211 if (modelModifier is null)
212 return Task.CompletedTask;
213
214 var viewModelList = _textEditorService.ModelApi.GetViewModelsOrEmpty(textSpan.ResourceUri);
215
216 var viewModel = viewModelList.FirstOrDefault(x => x.Category.Value == "main")
217 ?? viewModelList.FirstOrDefault();
218
219 if (viewModel is null)
220 return Task.CompletedTask;
221
222 var viewModelModifier = editContext.GetViewModelModifier(viewModel.ViewModelKey);
223
224 var primaryCursorModifier = editContext.GetPrimaryCursorModifier(
225 editContext.GetCursorModifierBag(viewModelModifier?.ViewModel));
226
227 if (viewModelModifier is null || primaryCursorModifier is null)
228 return Task.CompletedTask;
229
230 var cursorModifierBag = new CursorModifierBagTextEditor(
231 Key<TextEditorViewModel>.Empty,
232 new List<TextEditorCursorModifier> { primaryCursorModifier });
233
234 var cursorPositionIndex = modelModifier.GetPositionIndex(primaryCursorModifier);
235 var behindPositionIndex = cursorPositionIndex - 1;
236
237 var behindCursor = new TextEditorCursor(
238 primaryCursorModifier.LineIndex,
239 primaryCursorModifier.ColumnIndex - 1,
240 primaryCursorModifier.IsPrimaryCursor);
241
242 modelModifier.Insert(
243 textToInsert,
244 cursorModifierBag,
245 cancellationToken: CancellationToken.None);
246
247 if (behindPositionIndex > 0 && modelModifier.GetCharacter(behindPositionIndex) == 'p')
248 {
249 modelModifier.Delete(
250 new CursorModifierBagTextEditor(
251 Key<TextEditorViewModel>.Empty,
