155 private async Task ParseSolutionAsync()
156 {
157 var allFiles = new List<string>();
158
159 await RecursiveStep(
160 new List<string> { "/" },
161 allFiles)
162 .ConfigureAwait(false);
163
164 async Task RecursiveStep(IEnumerable<string> directories, List<string> allFiles)
165 {
166 foreach (var directory in directories)
167 {
168 var childDirectories = await FileSystemProvider.Directory
169 .GetDirectoriesAsync(directory)
170 .ConfigureAwait(false);
171
172 allFiles.AddRange(
173 await FileSystemProvider.Directory.GetFilesAsync(directory).ConfigureAwait(false));
174
175 await RecursiveStep(childDirectories, allFiles).ConfigureAwait(false);
176 }
177 }
178
179 foreach (var file in allFiles)
180 {
181 var absolutePath = EnvironmentProvider.AbsolutePathFactory(file, false);
182 var resourceUri = new ResourceUri(file);
183 var fileLastWriteTime = await FileSystemProvider.File.GetLastWriteTimeAsync(file).ConfigureAwait(false);
184 var content = await FileSystemProvider.File.ReadAllTextAsync(file).ConfigureAwait(false);
185
186 var decorationMapper = DecorationMapperRegistry.GetDecorationMapper(absolutePath.ExtensionNoPeriod);
187 var compilerService = CompilerServiceRegistry.GetCompilerService(absolutePath.ExtensionNoPeriod);
188
189 var textEditorModel = new TextEditorModel(
190 resourceUri,
191 fileLastWriteTime,
192 absolutePath.ExtensionNoPeriod,
193 content,
194 decorationMapper,
195 compilerService);
196
197 TextEditorService.ModelApi.RegisterCustom(textEditorModel);
198
199 TextEditorService.PostUnique(
200 nameof(TextEditorService.ModelApi.AddPresentationModel),
201 editContext =>
202 {
203 var modelModifier = editContext.GetModelModifier(textEditorModel.ResourceUri);
204
205 if (modelModifier is null)
206 return Task.CompletedTask;
207
208 TextEditorService.ModelApi.AddPresentationModel(
209 editContext,
210 modelModifier,
211 CompilerServiceDiagnosticPresentationFacts.EmptyPresentationModel);
212
213 TextEditorService.ModelApi.AddPresentationModel(
214 editContext,
215 modelModifier,
216 FindOverlayPresentationFacts.EmptyPresentationModel);
217
218 TextEditorService.ModelApi.AddPresentationModel(
219 editContext,
220 modelModifier,
221 DiffPresentationFacts.EmptyInPresentationModel);
222
223 TextEditorService.ModelApi.AddPresentationModel(
224 editContext,
225 modelModifier,
226 DiffPresentationFacts.EmptyOutPresentationModel);
227
228 textEditorModel.CompilerService.RegisterResource(
229 textEditorModel.ResourceUri,
230 shouldTriggerResourceWasModified: true);
231 return Task.CompletedTask;
232 });
233 }
234 }
