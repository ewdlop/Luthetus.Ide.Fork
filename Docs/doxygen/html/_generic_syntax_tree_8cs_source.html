21 public virtual GenericSyntaxUnit ParseText(ResourceUri resourceUri, string content)
22 {
23 var documentChildList = new List<IGenericSyntax>();
24 var diagnosticBag = new DiagnosticBag();
25
26 var stringWalker = new StringWalker(resourceUri, content);
27
28 while (!stringWalker.IsEof)
29 {
30 if (stringWalker.PeekForSubstring(GenericLanguageDefinition.StringStart))
31 {
32 var genericStringSyntax = ParseString(stringWalker, diagnosticBag);
33 documentChildList.Add(genericStringSyntax);
34 }
35 else if (stringWalker.PeekForSubstring(GenericLanguageDefinition.CommentSingleLineStart))
36 {
37 var genericCommentSingleLineSyntax = ParseCommentSingleLine(stringWalker, diagnosticBag);
38 documentChildList.Add(genericCommentSingleLineSyntax);
39 }
40 else if (stringWalker.PeekForSubstring(GenericLanguageDefinition.CommentMultiLineStart))
41 {
42 var genericCommentMultiLineSyntax = ParseCommentMultiLine(stringWalker, diagnosticBag);
43 documentChildList.Add(genericCommentMultiLineSyntax);
44 }
45 else if (stringWalker.PeekForSubstring(GenericLanguageDefinition.FunctionInvocationStart))
46 {
47 if (TryParseFunctionIdentifier(stringWalker, diagnosticBag, out var genericFunctionSyntax) &&
48 genericFunctionSyntax is not null)
49 {
50 documentChildList.Add(genericFunctionSyntax);
51 }
52 }
53 else if (!WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter) &&
54 !KeyboardKeyFacts.PunctuationCharacters.AllList.Contains(stringWalker.CurrentCharacter))
55 {
56 if (TryParseKeyword(stringWalker, diagnosticBag, out var genericKeywordSyntax) &&
57 genericKeywordSyntax is not null)
58 {
59 documentChildList.Add(genericKeywordSyntax);
60 }
61 }
62 else if (stringWalker.PeekForSubstring(GenericLanguageDefinition.PreprocessorDefinition.TransitionSubstring))
63 {
64 var genericCommentMultiLineSyntax = ParsePreprocessorDirective(stringWalker, diagnosticBag);
65 documentChildList.Add(genericCommentMultiLineSyntax);
66 }
67
68 _ = stringWalker.ReadCharacter();
69 }
70
71 var genericDocumentSyntax = new GenericDocumentSyntax(new TextEditorTextSpan(
72 0,
73 stringWalker.PositionIndex,
74 (byte)GenericDecorationKind.None,
75 stringWalker.ResourceUri,
76 stringWalker.SourceText),
77 documentChildList.ToImmutableArray());
78
79 return new GenericSyntaxUnit(genericDocumentSyntax, diagnosticBag);
80 }

194 private bool TryParseKeyword(
195 StringWalker stringWalker,
196 DiagnosticBag diagnosticBag,
197 out GenericKeywordSyntax? genericKeywordSyntax)
198 {
199 var startingPositionIndex = stringWalker.PositionIndex;
200
201 while (!stringWalker.IsEof)
202 {
203 var backtrackedToCharacter = stringWalker.BacktrackCharacter();
204
205 if (backtrackedToCharacter == ParserFacts.END_OF_FILE)
206 break;
207
208 if (WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter) ||
209 KeyboardKeyFacts.PunctuationCharacters.AllList.Contains(stringWalker.CurrentCharacter))
210 {
211 _ = stringWalker.ReadCharacter();
212 break;
213 }
214 }
215
216 var wordTuple = stringWalker.ReadWordTuple(KeyboardKeyFacts.PunctuationCharacters.AllList);
217
218 var foundKeyword = GenericLanguageDefinition.KeywordsList.FirstOrDefault(
219 keyword => keyword == wordTuple.value);
220
221 if (foundKeyword is not null)
222 {
223 stringWalker.BacktrackCharacter();
224
225 genericKeywordSyntax = new GenericKeywordSyntax(wordTuple.textSpan with
226 {
227 DecorationByte = (byte)GenericDecorationKind.Keyword
228 });
229
230 return true;
231 }
232
233 if (wordTuple.textSpan.StartingIndexInclusive != -1)
234 {
235 // backtrack to the original starting position
236 stringWalker.BacktrackRange(stringWalker.PositionIndex - startingPositionIndex);
237 }
238
239 genericKeywordSyntax = null;
240 return false;
241 }

243 private bool TryParseFunctionIdentifier(
244 StringWalker stringWalker,
245 DiagnosticBag diagnosticBag,
246 out GenericFunctionSyntax? genericFunctionSyntax)
247 {
248 var rememberPositionIndex = stringWalker.PositionIndex;
249
250 bool startedReadingWord = false;
251
252 var wordBuilder = new StringBuilder();
253
254 // Enter here at '('
255 while (!stringWalker.IsEof)
256 {
257 var backtrackedToCharacter = stringWalker.BacktrackCharacter();
258
259 if (backtrackedToCharacter == ParserFacts.END_OF_FILE)
260 break;
261
262 if (WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter))
263 {
264 if (startedReadingWord)
265 break;
266 }
267 else if (KeyboardKeyFacts.IsPunctuationCharacter(stringWalker.CurrentCharacter) ||
268 stringWalker.PeekForSubstring(GenericLanguageDefinition.MemberAccessToken) ||
269 stringWalker.PeekForSubstring(GenericLanguageDefinition.FunctionInvocationEnd))
270 {
271 break;
272 }
273 else
274 {
275 startedReadingWord = true;
276 wordBuilder.Insert(0, stringWalker.CurrentCharacter);
277 }
278 }
279
280 var word = wordBuilder.ToString();
281
282 if (word.Length == 0 || char.IsDigit(word[0]))
283 {
284 genericFunctionSyntax = null;
285
286 _ = stringWalker.ReadRange(rememberPositionIndex - stringWalker.PositionIndex);
287 return false;
288 }
289
290 genericFunctionSyntax = new GenericFunctionSyntax(
291 new TextEditorTextSpan(
292 stringWalker.PositionIndex + 1,
293 rememberPositionIndex,
294 (byte)GenericDecorationKind.Function,
295 stringWalker.ResourceUri,
296 stringWalker.SourceText));
297
298 _ = stringWalker.ReadRange(rememberPositionIndex - stringWalker.PositionIndex);
299 return true;
300 }

302 private GenericPreprocessorDirectiveSyntax ParsePreprocessorDirective(
303 StringWalker stringWalker,
304 DiagnosticBag diagnosticBag)
305 {
306 var startingPositionIndex = stringWalker.PositionIndex;
307
308 _ = stringWalker.ReadRange(
309 GenericLanguageDefinition.PreprocessorDefinition.TransitionSubstring.Length);
310
311 while (!stringWalker.IsEof)
312 {
313 if (WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter))
314 {
315 _ = stringWalker.ReadCharacter();
316 continue;
317 }
318
319 break;
320 }
321
322 var identifierBuilder = new StringBuilder();
323
324 while (!stringWalker.IsEof)
325 {
326 if (WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter))
327 break;
328
329 identifierBuilder.Append(stringWalker.CurrentCharacter);
330
331 _ = stringWalker.ReadCharacter();
332 }
333
334 var textSpan = new TextEditorTextSpan(
335 startingPositionIndex,
336 stringWalker.PositionIndex,
337 (byte)GenericDecorationKind.PreprocessorDirective,
338 stringWalker.ResourceUri,
339 stringWalker.SourceText);
340
341 var success = TryParsePreprocessorDirectiveDeliminationExtendedSyntaxes(
342 stringWalker,
343 diagnosticBag,
344 out var genericSyntax);
345
346 var children = success && genericSyntax is not null
347 ? new[] { genericSyntax }.ToImmutableArray()
348 : ImmutableArray<IGenericSyntax>.Empty;
349
350 return new GenericPreprocessorDirectiveSyntax(textSpan, children);
351 }

353 private bool TryParsePreprocessorDirectiveDeliminationExtendedSyntaxes(
354 StringWalker stringWalker,
355 DiagnosticBag diagnosticBag,
356 out IGenericSyntax? genericSyntax)
357 {
358 var entryPositionIndex = stringWalker.PositionIndex;
359
360 while (!stringWalker.IsEof)
361 {
362 if (WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter))
363 {
364 _ = stringWalker.ReadCharacter();
365 continue;
366 }
367
368 break;
369 }
370
371 DeliminationExtendedSyntaxDefinition? matchedDeliminationExtendedSyntax = null;
372
373 foreach (var deliminationExtendedSyntax in GenericLanguageDefinition.PreprocessorDefinition.DeliminationExtendedSyntaxList)
374 {
375 if (stringWalker.PeekForSubstring(deliminationExtendedSyntax.SyntaxStart))
376 {
377 matchedDeliminationExtendedSyntax = deliminationExtendedSyntax;
378 break;
379 }
380 }
381
382 if (matchedDeliminationExtendedSyntax is not null)
383 {
384 var deliminationExtendedSyntaxStartingInclusiveIndex = stringWalker.PositionIndex;
385 var deliminationExtendedSyntaxBuilder = new StringBuilder();
386
387 while (!stringWalker.IsEof)
388 {
389 if (stringWalker.PeekForSubstring(matchedDeliminationExtendedSyntax.SyntaxEnd))
390 {
391 _ = stringWalker.ReadCharacter();
392 break;
393 }
394
395 deliminationExtendedSyntaxBuilder.Append(stringWalker.CurrentCharacter);
396
397 _ = stringWalker.ReadCharacter();
398 }
399
400 var textSpan = new TextEditorTextSpan(
401 deliminationExtendedSyntaxStartingInclusiveIndex,
402 stringWalker.PositionIndex,
403 (byte)matchedDeliminationExtendedSyntax.GenericDecorationKind,
404 stringWalker.ResourceUri,
405 stringWalker.SourceText);
406
407 genericSyntax = new GenericDeliminationExtendedSyntax(textSpan);
408
409 return true;
410 }
411
412 stringWalker.BacktrackRange(stringWalker.PositionIndex - entryPositionIndex);
413
414 genericSyntax = null;
415 return false;
416 }
