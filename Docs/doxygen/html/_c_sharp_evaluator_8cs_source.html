82 private EvaluatorResult EvaluateBinaryExpressionNode(BinaryExpressionNode boundBinaryExpressionNode)
83 {
84 if (boundBinaryExpressionNode.ResultTypeClauseNode.ValueType == typeof(int))
85 {
86 var leftValue = EvaluateExpression(

90 boundBinaryExpressionNode.RightExpressionNode);
91
92 switch (boundBinaryExpressionNode.BinaryOperatorNode.OperatorToken.SyntaxKind)

94 case SyntaxKind.PlusToken:
95 {
96 var resultingValue = (int)leftValue.Result + (int)rightValue.Result;
97
98 return new EvaluatorResult(
99 boundBinaryExpressionNode.ResultTypeClauseNode.ValueType,
100 resultingValue);
101 }
102 case SyntaxKind.MinusToken:
103 {
104 var resultingValue = (int)leftValue.Result - (int)rightValue.Result;
105
106 return new EvaluatorResult(
107 boundBinaryExpressionNode.ResultTypeClauseNode.ValueType,
108 resultingValue);
109 }
110 case SyntaxKind.StarToken:
111 {
112 var resultingValue = (int)leftValue.Result * (int)rightValue.Result;
113
114 return new EvaluatorResult(
115 boundBinaryExpressionNode.ResultTypeClauseNode.ValueType,
116 resultingValue);
117 }

119 {
120 var resultingValue = (int)leftValue.Result / (int)rightValue.Result;
