43 public SolutionVisualizationModel MakeDrawing(
44 DotNetSolutionCompilerService dotNetSolutionCompilerService,
45 CSharpProjectCompilerService cSharpProjectCompilerService,
46 CSharpCompilerService cSharpCompilerService)
47 {
48 var localSolutionVisualizationModel = ShallowClone();
49 localSolutionVisualizationModel.SolutionVisualizationDrawingList.Clear();
50 localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList.Clear();
51
52 // Do not clear the 'ParentMap', it will stop the connections from project to class.
53 // localSolutionVisualizationModel.ParentMap.Clear();
54
55 if (SolutionAbsolutePath is null)
56 return localSolutionVisualizationModel;
57
58 var dotNetSolutionResource = dotNetSolutionCompilerService.CompilerServiceResources.FirstOrDefault(
59 x => x.ResourceUri.Value == SolutionAbsolutePath.Value);
60
61 if (dotNetSolutionResource is null)
62 return localSolutionVisualizationModel;
63
64 var cSharpProjectResourceList = cSharpProjectCompilerService.CompilerServiceResources;
65 var cSharpResourceList = cSharpCompilerService.CompilerServiceResources;
66
67 var radius = 12;
68 var centerX = 12;
69 var centerY = 12;
70 var rowIndex = 0;
71 var columnIndex = 0;
72 var renderCycleIndex = 0;
73
74 // Regarding drawing the connections between the solution node, and its child project nodes.
75 //
76 // We need to draw the connections, before the nodes themselves, as to permit the connections be drawn
77 // from the center of each node, rather than calculating the arc on the edge of the circle.
78 //
79 // i.e.: if the connections are drawn first, the node will perfectly cover the connection, curvature and all,
80 // even though the connection is simply a line from one point to another.
81 //
82 // That being said, we don't know the position of the nodes until their respective 'Draw...(...)' methods are invoked.
83 // But, these draw methods don't actually render anything on the UI. It is just populating a list,
84 // in which from index 0 to the end of that list, things will be rendered in that order.
85 //
86 // So, if at the end we insert the connections at index 0 of the "to be rendered list", it should work out.
87 // An issue however, there are properties on the 'ISolutionVisualizationDrawing.cs' relating to the order that things were rendered.
88 //
89 // If we do not set these properties correctly, the C# code won't be able to properly determine what UI element was
90 // clicked, (when two UI elements overlap, the most recently rendered element receives the click event).
91 //
92 // Therefore, an awkard incrementation of 'renderCycleIndex' is being done here. This leaves renderCycleIndex of index 0
93 // free for use at a later time. renderCycleIndex of index 0 is saying, "this was part of the UI element group that was first rendered".
94 //
95 // Thus, if one clicks a node, the "covered connection" that extends through the perimiter of a node, to its center point.
96 // This element won't get the click event.
97 {
98 // Reserve renderCycleIndex of 0 for the connections drawn between nodes.
99 localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList.Add(new List<ISolutionVisualizationDrawing>());
100 renderCycleIndex++;
101 }
102
103 DrawSolution(dotNetSolutionResource, localSolutionVisualizationModel, radius, centerX, centerY, rowIndex, columnIndex, renderCycleIndex);
104 renderCycleIndex++;
105 rowIndex++;
106
107 DrawProjects(cSharpProjectResourceList, localSolutionVisualizationModel, radius, centerX, centerY, rowIndex, columnIndex, renderCycleIndex);
108 renderCycleIndex++;
109 rowIndex++;
110
111 DrawClasses(cSharpResourceList, localSolutionVisualizationModel, radius, centerX, centerY, rowIndex, columnIndex, renderCycleIndex);
112 renderCycleIndex++;
113 rowIndex++;
114
115 DrawConnections(localSolutionVisualizationModel, renderCycleIndex: 0);
116
117 return localSolutionVisualizationModel;
118 }

120 private void DrawSolution(
121 ICompilerServiceResource dotNetSolutionResource,

151 private void DrawProjects(
152 ImmutableArray<ICompilerServiceResource> cSharpProjectResourceList,
153 SolutionVisualizationModel localSolutionVisualizationModel,
154 int radius,
155 int centerX,
156 int centerY,
157 int rowIndex,
158 int columnIndex,
159 int renderCycleIndex)
160 {
161 var solutionCenterX = localSolutionVisualizationModel.Dimensions.SvgWidth / 2 - radius / 2;
162 var leftNodeCounter = 0;
163 var rightNodeCounter = 0;
164
165 localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList.Add(new List<ISolutionVisualizationDrawing>());
166
167 for (int i = 0; i < cSharpProjectResourceList.Length; i++)
168 {
169 var cSharpProjectResource = cSharpProjectResourceList[i];
170
171 if (i % 2 == 0)
172 {
173 leftNodeCounter++;
174 var solutionRadius = radius;
175 var myRadius = leftNodeCounter * radius;
176 var previousCirclesRadius = (leftNodeCounter - 1) * radius;
177 var cumulativeHorizontalPadding = (leftNodeCounter - 1) * localSolutionVisualizationModel.Dimensions.HorizontalPadding;
178 centerX = solutionCenterX - solutionRadius - myRadius - previousCirclesRadius - cumulativeHorizontalPadding;
179 }
180 else
181 {
182 rightNodeCounter++;
183 var solutionRadius = radius;
184 var myRadius = rightNodeCounter * radius;
185 var previousCirclesRadius = (rightNodeCounter - 1) * radius;
186 var cumulativeHorizontalPadding = (rightNodeCounter - 1) * localSolutionVisualizationModel.Dimensions.HorizontalPadding;
187 centerX = solutionCenterX + solutionRadius + myRadius + previousCirclesRadius + cumulativeHorizontalPadding;
188 }
189
190 var cSharpProjectDrawing = new SolutionVisualizationDrawingCircle<CSharpProjectResource>
191 {
192 Item = (CSharpProjectResource)cSharpProjectResource,
193 SolutionVisualizationItemKind = SolutionVisualizationItemKind.Project,
194 CenterX = centerX,
195 CenterY = (1 + rowIndex) * centerY + rowIndex * radius + rowIndex * localSolutionVisualizationModel.Dimensions.VerticalPadding,
196 Radius = radius,
197 Fill = "var(--luth_icon-project-font-color)",
198 RenderCycle = renderCycleIndex,
199 };
200
201 columnIndex++;
202
203 localSolutionVisualizationModel.SolutionVisualizationDrawingList.Add(cSharpProjectDrawing);
204 localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList[renderCycleIndex].Add(cSharpProjectDrawing);
205 }
206 }

240 private void DrawConnections(
241 SolutionVisualizationModel localSolutionVisualizationModel,
242 int renderCycleIndex)
243 {
244 var solutionDrawing = localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList[RENDER_CYCLE_LIST_SOLUTION_INDEX].Single();
245 var projectDrawingList = localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList[RENDER_CYCLE_LIST_PROJECTS_INDEX];
246
247 // From solution to project connections
248 foreach (var projectDrawing in projectDrawingList)
249 {
250 var solutionDrawingCircle = (ISolutionVisualizationDrawingCircle)solutionDrawing;
251 var projectDrawingCircle = (ISolutionVisualizationDrawingCircle)projectDrawing;
252
253 var connectionDrawing = new SolutionVisualizationDrawingLine<SolutionVisualizationConnection>
254 {
255 Item = new SolutionVisualizationConnection(solutionDrawing, projectDrawing),
256 SolutionVisualizationItemKind = SolutionVisualizationItemKind.Connection,
257 StartPoint = (solutionDrawingCircle.CenterX, solutionDrawingCircle.CenterY),
258 EndPoint = (projectDrawingCircle.CenterX, projectDrawingCircle.CenterY),
259 Stroke = "var(--luth_primary-foreground-color)",
260 RenderCycle = renderCycleIndex,
261 };
262
263 localSolutionVisualizationModel.SolutionVisualizationDrawingList.Insert(0, connectionDrawing);
264 localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList[renderCycleIndex].Add(connectionDrawing);
265 }
266
267 var classDrawingList = localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList[RENDER_CYCLE_LIST_CLASSES_INDEX];
268
269 // From project to class connections
270 foreach (var classDrawing in classDrawingList)
271 {
272 var cSharpResource = (CSharpResource)classDrawing.Item;
273
274 Console.WriteLine($"localSolutionVisualizationModel.ParentMap::{localSolutionVisualizationModel.ParentMap.Count}");
275
276 if (!localSolutionVisualizationModel.ParentMap.TryGetValue(cSharpResource.ResourceUri.Value, out var projectDrawing))
277 continue;
278
279 var projectDrawingCircle = (ISolutionVisualizationDrawingCircle)projectDrawing;
280 var classDrawingCircle = (ISolutionVisualizationDrawingCircle)classDrawing;
281
282 var connectionDrawing = new SolutionVisualizationDrawingLine<SolutionVisualizationConnection>
283 {
284 Item = new SolutionVisualizationConnection(projectDrawing, classDrawingCircle),
285 SolutionVisualizationItemKind = SolutionVisualizationItemKind.Connection,
286 StartPoint = (projectDrawingCircle.CenterX, projectDrawingCircle.CenterY),
287 EndPoint = (classDrawingCircle.CenterX, classDrawingCircle.CenterY),
288 Stroke = "var(--luth_primary-foreground-color)",
289 RenderCycle = renderCycleIndex,
290 };
291
292 localSolutionVisualizationModel.SolutionVisualizationDrawingList.Insert(0, connectionDrawing);
293 localSolutionVisualizationModel.SolutionVisualizationDrawingRenderCycleList[renderCycleIndex].Add(connectionDrawing);
294 }
295 }
