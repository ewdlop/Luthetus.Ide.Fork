182 private async Task SetInputFileContentTreeViewRootFunc(IAbsolutePath absolutePath)
183 {
184 var pseudoRootNode = new TreeViewAbsolutePath(
185 absolutePath,
186 IdeComponentRenderers,
187 CommonComponentRenderers,
188 FileSystemProvider,
189 EnvironmentProvider,
190 true,
191 false);
192
193 await pseudoRootNode.LoadChildListAsync().ConfigureAwait(false);
194
195 var adhocRootNode = TreeViewAdhoc.ConstructTreeViewAdhoc(pseudoRootNode.ChildList.ToArray());
196
197 foreach (var child in adhocRootNode.ChildList)
198 {
199 child.IsExpandable = false;
200 }
201
202 var activeNode = adhocRootNode.ChildList.FirstOrDefault();
203
204 if (!TreeViewService.TryGetTreeViewContainer(InputFileContent.TreeViewContainerKey, out var treeViewContainer))
205 {
206 TreeViewService.RegisterTreeViewContainer(new TreeViewContainer(
207 InputFileContent.TreeViewContainerKey,
208 adhocRootNode,
209 activeNode is null
210 ? ImmutableList<TreeViewNoType>.Empty
211 : new TreeViewNoType[] { activeNode }.ToImmutableList()));
212 }
213 else
214 {
215 TreeViewService.SetRoot(InputFileContent.TreeViewContainerKey, adhocRootNode);
216
217 TreeViewService.SetActiveNode(
218 InputFileContent.TreeViewContainerKey,
219 activeNode,
220 true,
221 false);
222 }
223
224 await pseudoRootNode.LoadChildListAsync().ConfigureAwait(false);
225
226 var setOpenedTreeViewModelAction = new InputFileState.SetOpenedTreeViewModelAction(
227 pseudoRootNode,
228 IdeComponentRenderers,
229 CommonComponentRenderers,
230 FileSystemProvider,
231 EnvironmentProvider);
232
233 Dispatcher.Dispatch(setOpenedTreeViewModelAction);
234 }
