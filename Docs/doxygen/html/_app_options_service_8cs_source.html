200 public async Task SetFromLocalStorageAsync()
201 {
202 var optionsJsonString = await _storageService.GetValue(StorageKey).ConfigureAwait(false) as string;
203
204 if (string.IsNullOrWhiteSpace(optionsJsonString))
205 return;
206
207 var optionsJson = JsonSerializer.Deserialize<CommonOptionsJsonDto>(optionsJsonString);
208
209 if (optionsJson is null)
210 return;
211
212 if (optionsJson.ThemeKey is not null)
213 {
214 var matchedTheme = ThemeStateWrap.Value.ThemeList.FirstOrDefault(
215 x => x.Key == optionsJson.ThemeKey);
216
217 SetTheme(matchedTheme ?? ThemeFacts.VisualStudioDarkThemeClone, false);
218 }
219
220 if (optionsJson.FontFamily is not null)
221 SetFontFamily(optionsJson.FontFamily, false);
222
223 if (optionsJson.FontSizeInPixels is not null)
224 SetFontSize(optionsJson.FontSizeInPixels.Value, false);
225
226 if (optionsJson.ResizeHandleWidthInPixels is not null)
227 SetResizeHandleWidth(optionsJson.ResizeHandleWidthInPixels.Value, false);
228
229 if (optionsJson.ResizeHandleHeightInPixels is not null)
230 SetResizeHandleHeight(optionsJson.ResizeHandleHeightInPixels.Value, false);
231
232 if (optionsJson.IconSizeInPixels is not null)
233 SetIconSize(optionsJson.IconSizeInPixels.Value, false);
234 }
