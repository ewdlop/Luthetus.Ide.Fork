138 public virtual Task ParseAsync(ITextEditorEditContext editContext, TextEditorModelModifier modelModifier)
139 {
140 _textEditorService.ModelApi.StartPendingCalculatePresentationModel(
141 editContext,
142 modelModifier,
143 CompilerServiceDiagnosticPresentationFacts.PresentationKey,
144 CompilerServiceDiagnosticPresentationFacts.EmptyPresentationModel);
145
146 var presentationModel = modelModifier.PresentationModelList.First(
147 x => x.TextEditorPresentationKey == CompilerServiceDiagnosticPresentationFacts.PresentationKey);
148
149 if (presentationModel.PendingCalculation is null)
150 throw new LuthetusTextEditorException($"{nameof(presentationModel)}.{nameof(presentationModel.PendingCalculation)} was not expected to be null here.");
151
152 if (_compilerServiceOptions.GetLexerFunc is null)
153 return Task.CompletedTask;
154
155 var resourceUri = modelModifier.ResourceUri;
156
157 ILexer lexer;
158 lock (_resourceMapLock)
159 {
160 if (!_resourceMap.ContainsKey(resourceUri))
161 return Task.CompletedTask;
162
163 var resource = _resourceMap[resourceUri];
164 lexer = _compilerServiceOptions.GetLexerFunc.Invoke(resource, presentationModel.PendingCalculation.ContentAtRequest);
165 }
166
167 lexer.Lex();
168 lock (_resourceMapLock)
169 {
170 if (!_resourceMap.ContainsKey(resourceUri))
171 return Task.CompletedTask;
172
173 var resource = _resourceMap[resourceUri];
174 _compilerServiceOptions.OnAfterLexAction?.Invoke(resource, lexer);
175 }
176
177 CompilationUnit? compilationUnit = null;
178 // Even if the parser throws an exception, be sure to
179 // make use of the Lexer to do whatever syntax highlighting is possible.
180 try
181 {
182 if (_compilerServiceOptions.GetParserFunc is null || Binder is null)
183 return Task.CompletedTask;
184
185 IParser parser;
186 lock (_resourceMapLock)
187 {
188 if (!_resourceMap.ContainsKey(resourceUri))
189 return Task.CompletedTask;
190
191 var resource = _resourceMap[resourceUri];
192 parser = _compilerServiceOptions.GetParserFunc.Invoke(resource, lexer);
193 }
194
195 compilationUnit = parser.Parse(Binder, resourceUri);
196 }
197 finally
198 {
199 lock (_resourceMapLock)
200 {
201 if (_resourceMap.ContainsKey(resourceUri))
202 {
203 var resource = _resourceMap[resourceUri];
204
205 resource.SyntaxTokenList = lexer.SyntaxTokenList;
206
207 if (compilationUnit is not null)
208 resource.CompilationUnit = compilationUnit;
209
210 _compilerServiceOptions.OnAfterParseAction?.Invoke(resource, compilationUnit);
211 }
212 }
213
214 var diagnosticTextSpans = GetDiagnosticsFor(modelModifier.ResourceUri)
215 .Select(x => x.TextSpan)
216 .ToImmutableArray();
217
218 modelModifier.CompletePendingCalculatePresentationModel(
219 CompilerServiceDiagnosticPresentationFacts.PresentationKey,
220 CompilerServiceDiagnosticPresentationFacts.EmptyPresentationModel,
221 diagnosticTextSpans);
222
223 editContext.TextEditorService.ModelApi.ApplySyntaxHighlighting(
224 editContext,
225 modelModifier);
226
227 OnResourceParsed();
228 }
229
230 return Task.CompletedTask;
231 }
