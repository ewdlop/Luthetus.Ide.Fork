92 public void WriteOutput(TerminalCommandParsed terminalCommandParsed, CommandEvent commandEvent)
93 {
94 var output = (string?)null;
95
96 switch (commandEvent)
97 {
98 case StartedCommandEvent started:
99
100 // Delete any output of the previous invocation.
101 lock (_listLock)
102 {
103 var indexPreviousOutput = _parsedCommandList.FindIndex(x =>
104 x.SourceTerminalCommandRequest.Key ==
105 terminalCommandParsed.SourceTerminalCommandRequest.Key);
106
107 if (indexPreviousOutput != -1)
108 _parsedCommandList.RemoveAt(indexPreviousOutput);
109
110 _parsedCommandList.Add(terminalCommandParsed);
111 }
112
113 break;
114 case StandardOutputCommandEvent stdOut:
115 terminalCommandParsed.OutputCache.AppendTwo(stdOut.Text, "\n");
116 break;
117 case StandardErrorCommandEvent stdErr:
118 terminalCommandParsed.OutputCache.AppendTwo(stdErr.Text, "\n");
119 break;
120 case ExitedCommandEvent exited:
121 break;
122 }
123
124 OnWriteOutput?.Invoke();
125 }

155 public void ClearHistoryWhenExistingOutputTooLong()
156 {
157 lock (_listLock)
158 {
159 var sumOutputLength = _parsedCommandList.Sum(x => x.OutputCache.GetLength());
160
161 if (sumOutputLength > TerminalOutputFacts.MAX_OUTPUT_LENGTH ||
162 _parsedCommandList.Count > TerminalOutputFacts.MAX_COMMAND_COUNT)
163 {
164 var rememberLastCommand = _parsedCommandList.LastOrDefault();
165
166 _parsedCommandList.Clear();
167
168 if (rememberLastCommand is not null &&
169 rememberLastCommand.OutputCache.GetLength() < TerminalOutputFacts.OUTPUT_LENGTH_PADDING)
170 {
171 // It feels odd to clear the entire terminal when there is too much text output
172 // that has accumulated.
173 //
174 // So, keep the most recent command's output,
175 // unless its output length is greater than or equal to the TerminalOutputFacts.OUTPUT_LENGTH_PADDING.
176 _parsedCommandList.Add(rememberLastCommand);
177 }
178 }
179 }
180
181 OnWriteOutput?.Invoke();
182 }
