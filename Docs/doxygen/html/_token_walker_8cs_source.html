49 public ISyntaxToken Peek(int offset)
50 {
51 var index = _index + offset;
52
53 if (index < 0)
54 return GetBadToken();
55 else if (index >= _tokenList.Length)
56 return EOF; // Return the end of file token (the last token)
57

59 }

61 public ISyntaxToken Consume()
62 {
63 if (_index >= _tokenList.Length)
64 return EOF; // Return the end of file token (the last token)
65
66 if (_deferredParsingTuple is not null)
67 {
68 if (_index == _deferredParsingTuple.Value.closeTokenIndex)
69 {
70 var closeChildScopeToken = _tokenList[_index];
71 _index = _deferredParsingTuple.Value.tokenIndexToRestore;
72 ConsumeCounter++;
73 _deferredParsingTuple = null;
74 return closeChildScopeToken;
75 }
76 }
77
78 var consumedToken = _tokenList[_index++];
79 ConsumeCounter++;
80
81 #if DEBUG
82 if (!SuppressProtectedSyntaxKindConsumption)
83 {
84 if (ProtectedTokenSyntaxKindList.Contains(consumedToken.SyntaxKind))
85 Console.WriteLine($"The protected syntax kind: '{consumedToken.SyntaxKind}' was unexpectedly consumed.");
86 // throw new Exception($"The protected syntax kind: '{consumedToken.SyntaxKind}' was unexpectedly consumed.");
87 }
88 #endif
89
90 return consumedToken;
91 }

105 public ISyntaxToken Match(SyntaxKind expectedSyntaxKind)
106 {
107 var currentToken = Peek(0);
108
109 // TODO: Checking for the text 'args' is likely not a good solution. When parsing a main method, it might have the method arguments: 'string[] args'. The issue here is that 'args' comes up as a keyword while being the identifier for that method argument.
110 if (currentToken.TextSpan.GetText() == "args" && expectedSyntaxKind == SyntaxKind.IdentifierToken)
111 {
112 _ = Consume();
113 return new IdentifierToken(currentToken.TextSpan);
114 }
115
116 if (currentToken.SyntaxKind == expectedSyntaxKind)
117 return Consume();
118
119 var fabricatedToken = this.FabricateToken(expectedSyntaxKind);
120
121 _diagnosticBag.ReportUnexpectedToken(
122 fabricatedToken.TextSpan,
123 currentToken.SyntaxKind.ToString(),
124 expectedSyntaxKind.ToString());
125
126 return fabricatedToken;
127 }
