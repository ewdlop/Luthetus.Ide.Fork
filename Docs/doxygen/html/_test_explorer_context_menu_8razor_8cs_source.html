81 private async Task<MenuRecord> GetMenuRecord(TreeViewCommandArgs commandArgs, bool isRecursiveCall = false)

84 {
85 return await GetMultiSelectionMenuRecord(commandArgs).ConfigureAwait(false);

87
88 if (commandArgs.NodeThatReceivedMouseEvent is null)

91 var menuRecordsList = new List<MenuOptionRecord>();
92
93 if (commandArgs.NodeThatReceivedMouseEvent is TreeViewStringFragment treeViewStringFragment)
94 {
95 var target = treeViewStringFragment;

100 fullyQualifiedNameBuilder.Insert(0, $"{parentNode.Item.Value}.");
101 target = parentNode;
102 }
103
104 if (target.Parent is TreeViewProjectTestModel treeViewProjectTestModel &&
105 treeViewStringFragment.Item.IsEndpoint)
106 {
107 var fullyQualifiedName = fullyQualifiedNameBuilder.ToString();
108
109 var menuOptionRecord = GetEndPointMenuOption(
110 treeViewStringFragment,
111 treeViewProjectTestModel,
112 fullyQualifiedName);
113
114 menuRecordsList.Add(menuOptionRecord);
115
116 if (commandArgs.TreeViewContainer.SelectedNodeList.Count == 1)
117 {
118 menuRecordsList.Add(new MenuOptionRecord(
119 $"Send to Output panel",
120 MenuOptionKind.Other,
121 OnClickFunc: () =>

123 return SendToOutputPanelAsync(treeViewStringFragment.Item.TerminalCommandParsed?.OutputCache.ToString() ?? string.Empty);

125 }
126 }
127 else
128 {
129 menuRecordsList.AddRange(await GetNamespaceMenuOption(
130 treeViewStringFragment,
131 commandArgs,
132 isRecursiveCall));
133 }
134 }
135 else if (commandArgs.NodeThatReceivedMouseEvent is TreeViewProjectTestModel treeViewProjectTestModel)
136 {
137 menuRecordsList.Add(new MenuOptionRecord(
138 $"Refresh: {treeViewProjectTestModel.Item.AbsolutePath.NameWithExtension}",
139 MenuOptionKind.Other,
140 OnClickFunc: async () =>
141 {
142 // TODO: This code is not concurrency safe with 'TestExplorerScheduler.Task_DiscoverTests()'

150 var mutableFailedTestHashSet = inState.FailedTestHashSet.ToHashSet();
151

154 mutablePassedTestHashSet.Remove(fullyQualifiedTestName);
155 mutableNotRanTestHashSet.Remove(fullyQualifiedTestName);
156 mutableFailedTestHashSet.Remove(fullyQualifiedTestName);
157 }
158
159 return inState with
160 {
161 PassedTestHashSet = mutablePassedTestHashSet.ToImmutableHashSet(),
162 NotRanTestHashSet = mutableNotRanTestHashSet.ToImmutableHashSet(),
163 FailedTestHashSet = mutableFailedTestHashSet.ToImmutableHashSet(),

165 }));

167 treeViewProjectTestModel.Item.TestNameFullyQualifiedList = null;

169
170 await treeViewProjectTestModel.LoadChildListAsync();
171 TreeViewService.ReRenderNode(TestExplorerState.TreeViewTestExplorerKey, treeViewProjectTestModel);
172
173 DotNetBackgroundTaskApi.TestExplorer.MoveNodeToCorrectBranch(treeViewProjectTestModel);
174
175 Dispatcher.Dispatch(new TestExplorerState.WithAction(inState =>
176 {
177 if (treeViewProjectTestModel.Item.TestNameFullyQualifiedList is null)
178 return inState;
179
180 var mutableNotRanTestHashSet = inState.NotRanTestHashSet.ToHashSet();
181
182 foreach (var fullyQualifiedTestName in treeViewProjectTestModel.Item.TestNameFullyQualifiedList)
183 {
184 mutableNotRanTestHashSet.Add(fullyQualifiedTestName);
185 }
186
187 return inState with
188 {
189 NotRanTestHashSet = mutableNotRanTestHashSet.ToImmutableHashSet(),
190 };
191 }));
192 }));
193
194 if (treeViewProjectTestModel.Parent is TreeViewGroup tvg &&
195 tvg.Item == "Projects that threw an exception during discovery")
196 {
197 menuRecordsList.Add(new MenuOptionRecord(
198 $"Send to Output panel",
199 MenuOptionKind.Other,
200 OnClickFunc: () =>
201 {
202 return SendToOutputPanelAsync(treeViewProjectTestModel.Item.TerminalCommandParsed?.OutputCache.ToString() ?? string.Empty);
203 }));
204 }
205 }
206
207 if (!menuRecordsList.Any())
208 return MenuRecord.Empty;
209
210 return new MenuRecord(menuRecordsList.ToImmutableArray());

246 private async Task<List<MenuOptionRecord>> GetNamespaceMenuOption(
247 TreeViewStringFragment treeViewStringFragment,
248 TreeViewCommandArgs commandArgs,
249 bool isRecursiveCall = false)
250 {
251 void RecursiveStep(TreeViewStringFragment treeViewStringFragmentNamespace, List<TreeViewNoType> fabricateSelectedNodeList)
252 {

254 {
255 if (childNode is TreeViewStringFragment childTreeViewStringFragment)
256 {
257 if (childTreeViewStringFragment.Item.IsEndpoint)
258 {
259 fabricateSelectedNodeList.Add(childTreeViewStringFragment);
260 }
261 else
262 {
263 RecursiveStep(childTreeViewStringFragment, fabricateSelectedNodeList);
264 }
265 }
266 }
267 }

269 var fabricateSelectedNodeList = new List<TreeViewNoType>();
270
271 RecursiveStep(treeViewStringFragment, fabricateSelectedNodeList);
272
273 var fabricateTreeViewContainer = commandArgs.TreeViewContainer with
274 {
275 SelectedNodeList = fabricateSelectedNodeList.ToImmutableList()
276 };
277
278 var fabricateCommandArgs = new TreeViewCommandArgs(
279 commandArgs.TreeViewService,
280 fabricateTreeViewContainer,
281 commandArgs.NodeThatReceivedMouseEvent,
282 commandArgs.RestoreFocusToTreeView,
283 commandArgs.ContextMenuFixedPosition,
284 commandArgs.MouseEventArgs,
285 commandArgs.KeyboardEventArgs);
286
287 var multiSelectionMenuRecord = await GetMultiSelectionMenuRecord(fabricateCommandArgs);
288
289 var menuOptionRecord = new MenuOptionRecord(
290 $"Namespace: {treeViewStringFragment.Item.Value} | {fabricateSelectedNodeList.Count}",
291 MenuOptionKind.Other,
292 SubMenu: multiSelectionMenuRecord);
293
294 return new() { menuOptionRecord };

297 private async Task<MenuRecord> GetMultiSelectionMenuRecord(TreeViewCommandArgs commandArgs)
298 {
299 var menuOptionRecordList = new List<MenuOptionRecord>();
300 Func<Task> runAllOnClicksWithinSelection = () => Task.CompletedTask;
301 bool runAllOnClicksWithinSelectionHasEffect = false;
302
303 foreach (var node in commandArgs.TreeViewContainer.SelectedNodeList)
304 {
305 MenuOptionRecord menuOption;
306
307 if (node is TreeViewStringFragment treeViewStringFragment)
308 {
309 var innerTreeViewCommandArgs = new TreeViewCommandArgs(
310 commandArgs.TreeViewService,
311 commandArgs.TreeViewContainer,
312 node,
313 commandArgs.RestoreFocusToTreeView,
314 commandArgs.ContextMenuFixedPosition,
315 commandArgs.MouseEventArgs,
316 commandArgs.KeyboardEventArgs);
317
318 menuOption = new(
319 treeViewStringFragment.Item.Value,
320 MenuOptionKind.Other,
321 SubMenu: await GetMenuRecord(innerTreeViewCommandArgs, true).ConfigureAwait(false));
322
323 var copyRunAllOnClicksWithinSelection = runAllOnClicksWithinSelection;
324
325 runAllOnClicksWithinSelection = async () =>
326 {
327 await copyRunAllOnClicksWithinSelection.Invoke().ConfigureAwait(false);
328
329 if (menuOption.SubMenu?.MenuOptionList.Single().OnClickFunc is not null)
330 {
331 await menuOption.SubMenu.MenuOptionList
332 .Single().OnClickFunc!
333 .Invoke()
334 .ConfigureAwait(false);
335 }
336 };
337
338 runAllOnClicksWithinSelectionHasEffect = true;
339 }
340 else
341 {
342 menuOption = new(
343 node.GetType().Name,
344 MenuOptionKind.Other,
345 SubMenu: MenuRecord.Empty);
346 }
347
348 menuOptionRecordList.Add(menuOption);
349 }
350
351 if (runAllOnClicksWithinSelectionHasEffect)
352 {
353 menuOptionRecordList.Insert(0, new(
354 "Run all OnClicks within selection",
355 MenuOptionKind.Create,
356 OnClickFunc: runAllOnClicksWithinSelection));
357 }
358
359 if (!menuOptionRecordList.Any())
360 return MenuRecord.Empty;
361
362 return new MenuRecord(menuOptionRecordList.ToImmutableArray());
363 }

365 private void RunTestByFullyQualifiedName(
366 TreeViewStringFragment treeViewStringFragment,
367 string fullyQualifiedName,
368 string? directoryNameForTestDiscovery)
369 {
370 var dotNetTestByFullyQualifiedNameFormattedCommand = DotNetCliCommandFormatter
371 .FormatDotNetTestByFullyQualifiedName(fullyQualifiedName);
372
373 if (string.IsNullOrWhiteSpace(directoryNameForTestDiscovery) ||
374 string.IsNullOrWhiteSpace(fullyQualifiedName))
375 {
376 return;
377 }
378
379 var terminalCommandRequest = new TerminalCommandRequest(
380 dotNetTestByFullyQualifiedNameFormattedCommand.Value,
381 directoryNameForTestDiscovery,
382 treeViewStringFragment.Item.DotNetTestByFullyQualifiedNameFormattedTerminalCommandRequestKey)
383 {
384 BeginWithFunc = parsedCommand =>
385 {
386 treeViewStringFragment.Item.TerminalCommandParsed = parsedCommand;
387 TreeViewService.ReRenderNode(TestExplorerState.TreeViewTestExplorerKey, treeViewStringFragment);
388 return Task.CompletedTask;
389 },
390 ContinueWithFunc = parsedCommand =>
391 {
392 treeViewStringFragment.Item.TerminalCommandParsed = parsedCommand;
393 var output = treeViewStringFragment.Item.TerminalCommandParsed?.OutputCache.ToString() ?? null;
394
395 if (output is not null && output.Contains("Duration:"))
396 {
397 if (output.Contains("Passed!"))
398 {
399 Dispatcher.Dispatch(new TestExplorerState.WithAction(inState => inState with
400 {
401 PassedTestHashSet = inState.PassedTestHashSet.Add(fullyQualifiedName),
402 NotRanTestHashSet = inState.NotRanTestHashSet.Remove(fullyQualifiedName),
403 FailedTestHashSet = inState.FailedTestHashSet.Remove(fullyQualifiedName),
404 }));
405 }
406 else
407 {
408 Dispatcher.Dispatch(new TestExplorerState.WithAction(inState => inState with
409 {
410 FailedTestHashSet = inState.FailedTestHashSet.Add(fullyQualifiedName),
411 NotRanTestHashSet = inState.NotRanTestHashSet.Remove(fullyQualifiedName),
412 PassedTestHashSet = inState.PassedTestHashSet.Remove(fullyQualifiedName),
413 }));
414 }
415 }
416
417 TreeViewService.ReRenderNode(TestExplorerState.TreeViewTestExplorerKey, treeViewStringFragment);
418 return Task.CompletedTask;
419 }
420 };
421
422 treeViewStringFragment.Item.TerminalCommandRequest = terminalCommandRequest;
423 TerminalStateWrap.Value.TerminalMap[TerminalFacts.EXECUTION_KEY].EnqueueCommand(terminalCommandRequest);
424 }
