121 public static string GetRelativeFromTwoAbsolutes(
122 IAbsolutePath startingPath,
123 IAbsolutePath endingPath,
124 IEnvironmentProvider environmentProvider)
125 {
126 var pathBuilder = new StringBuilder();
127 var commonPath = startingPath.AncestorDirectoryList.First().Value;
128
129 if ((startingPath.ParentDirectory?.Value ?? string.Empty) ==
130 (endingPath.ParentDirectory?.Value ?? string.Empty))
131 {
132 // TODO: Will this code break when the mounted drives are different, and parent directories share same name?
133
134 // Use './' because they share the same parent directory.
135 pathBuilder.Append($".{environmentProvider.DirectorySeparatorChar}");
136
137 commonPath = startingPath.AncestorDirectoryList.Last().Value;
138 }
139 else
140 {
141 // Use '../' but first calculate how many UpDir(s) are needed
142 int limitingIndex = Math.Min(
143 startingPath.AncestorDirectoryList.Count,
144 endingPath.AncestorDirectoryList.Count);
145
146 var i = 0;
147
148 for (; i < limitingIndex; i++)
149 {
150 var startingPathAncestor = environmentProvider.AbsolutePathFactory(
151 startingPath.AncestorDirectoryList[i].Value,
152 true);
153
154 var endingPathAncestor = environmentProvider.AbsolutePathFactory(
155 endingPath.AncestorDirectoryList[i].Value,
156 true);
157
158 if (startingPathAncestor.NameWithExtension == endingPathAncestor.NameWithExtension)
159 commonPath = startingPathAncestor.Value;
160 else
161 break;
162 }
163
164 var upDirCount = startingPath.AncestorDirectoryList.Count - i;
165
166 for (int appendUpDir = 0; appendUpDir < upDirCount; appendUpDir++)
167 {
168 pathBuilder.Append("../");
169 }
170 }
171
172 var notCommonPath = new string(endingPath.Value.Skip(commonPath.Length).ToArray());
173
174 return pathBuilder.Append(notCommonPath).ToString();
