175 public async Task SetFromLocalStorageAsync()
176 {
177 var optionsJsonString = await _storageService.GetValue(_textEditorService.StorageKey).ConfigureAwait(false) as string;
178
179 if (string.IsNullOrWhiteSpace(optionsJsonString))
180 return;
181
182 var optionsJson = JsonSerializer.Deserialize<TextEditorOptionsJsonDto>(optionsJsonString);
183
184 if (optionsJson is null)
185 return;
186
187 if (optionsJson.CommonOptionsJsonDto?.ThemeKey is not null)
188 {
189 var matchedTheme = _textEditorService.ThemeStateWrap.Value.ThemeList.FirstOrDefault(
190 x => x.Key == optionsJson.CommonOptionsJsonDto.ThemeKey);
191
192 SetTheme(matchedTheme ?? ThemeFacts.VisualStudioDarkThemeClone, false);
193 }
194
195 if (optionsJson.Keymap is not null)
196 {
197 var matchedKeymap = TextEditorKeymapFacts.AllKeymapsList.FirstOrDefault(
198 x => x.Key == optionsJson.Keymap.Key);
199
200 SetKeymap(matchedKeymap ?? TextEditorKeymapFacts.DefaultKeymap, false);
201
202 var activeKeymap = _textEditorService.OptionsStateWrap.Value.Options.Keymap;
203
204 if (activeKeymap is not null)
205 {
206 _dispatcher.Dispatch(new SetContextKeymapAction(
207 ContextFacts.TextEditorContext.ContextKey,
208 activeKeymap));
209 }
210 }
211
212 if (optionsJson.CommonOptionsJsonDto?.FontSizeInPixels is not null)
213 SetFontSize(optionsJson.CommonOptionsJsonDto.FontSizeInPixels.Value, false);
214
215 if (optionsJson.CursorWidthInPixels is not null)
216 SetCursorWidth(optionsJson.CursorWidthInPixels.Value, false);
217
218 if (optionsJson.TextEditorHeightInPixels is not null)
219 SetHeight(optionsJson.TextEditorHeightInPixels.Value, false);
220
221 if (optionsJson.ShowNewlines is not null)
222 SetShowNewlines(optionsJson.ShowNewlines.Value, false);
223
224 // TODO: OptionsSetUseMonospaceOptimizations will always get set to false (default for bool)
225 // for a first time user. This leads to a bad user experience since the proportional
226 // font logic is still being optimized. Therefore don't read in UseMonospaceOptimizations
227 // from local storage.
228 //
229 // OptionsSetUseMonospaceOptimizations(options.UseMonospaceOptimizations);
230
231 if (optionsJson.ShowWhitespace is not null)
232 SetShowWhitespace(optionsJson.ShowWhitespace.Value, false);
233 }
