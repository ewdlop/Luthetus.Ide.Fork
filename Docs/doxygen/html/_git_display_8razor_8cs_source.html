82 private MenuRecord ConstructMenu()
83 {
84 var localGitState = GitStateWrap.Value;
85
86 var menuOptionsList = new List<MenuOptionRecord>();
87
88 // Repo
89 {
90 string menuOptionDisplayName;
91 if (localGitState.Repo is null)
92 menuOptionDisplayName = "Pick Repo";
93 else
94 menuOptionDisplayName = "Change Repo";
95
96 var menuOption = new MenuOptionRecord(
97 menuOptionDisplayName,
98 MenuOptionKind.Other,
99 () => { ShowAddRepoDialog(); return Task.CompletedTask; });
100
101 var menuOptionNew = new MenuOptionRecord(
102 "Repo",
103 MenuOptionKind.Other,
104 SubMenu: new MenuRecord(new[] { menuOption }.ToImmutableArray()));
105
106 menuOptionsList.Add(menuOptionNew);
107 }
108
109 // Branch
110 if (localGitState.Repo is not null)
111 {
112 var branchNewMenuOption = GetBranchNewMenuOptionRecord(localGitState);
113 var branchCheckoutMenuOption = GetBranchCheckoutOptionRecord(localGitState);
114
115 var menuOptionNew = new MenuOptionRecord(
116 "Branch",
117 MenuOptionKind.Other,
118 SubMenu: new MenuRecord(new[]
119 {
120 branchNewMenuOption,
121 branchCheckoutMenuOption,
122 }.ToImmutableArray()));
123
124 menuOptionsList.Add(menuOptionNew);
125 }
126
127 // Actions
128 if (localGitState.Repo is not null)
129 {
130 var pushMenuOption = GetActionPushToOriginWithTrackingMenuOption(localGitState);
131 var pullMenuOption = GetActionPullMenuOption(localGitState);
132 var fetchMenuOption = GetActionFetchMenuOption(localGitState);
133
134 var menuOptionNew = new MenuOptionRecord(
135 "Actions",
136 MenuOptionKind.Other,
137 SubMenu: new MenuRecord(new[]
138 {
139 pushMenuOption,
140 pullMenuOption,
141 fetchMenuOption,
142 }.ToImmutableArray()));
143
144 menuOptionsList.Add(menuOptionNew);
145 }
146
147 return new MenuRecord(menuOptionsList.ToImmutableArray());
148 }
