123 private async Task LoadClasses(
124 CSharpProjectResource cSharpProjectResource,
125 SolutionVisualizationModel localSolutionVisualizationModel,
126 LuthetusTextEditorConfig textEditorConfig,
127 IServiceProvider serviceProvider)
128 {
129 if (textEditorConfig.RegisterModelFunc is null)
130 return;
131
132 var environmentProvider = serviceProvider.GetRequiredService<IEnvironmentProvider>();
133 var fileSystemProvider = serviceProvider.GetRequiredService<IFileSystemProvider>();
134 var commonComponentRenderers = serviceProvider.GetRequiredService<ICommonComponentRenderers>();
135 var dispatcher = serviceProvider.GetRequiredService<IDispatcher>();
136
137 var projectAbsolutePath = environmentProvider.AbsolutePathFactory(cSharpProjectResource.ResourceUri.Value, false);
138
139 var parentDirectory = projectAbsolutePath.ParentDirectory;
140 if (parentDirectory is null)
141 return;
142
143 var startingAbsolutePathForSearch = parentDirectory.Value;
144
145 var discoveredFileList = new List<string>();
146
147 await DiscoverFilesRecursively(startingAbsolutePathForSearch, discoveredFileList).ConfigureAwait(false);
148
149 NotificationHelper.DispatchInformative(
150 "SolutionVisualizationDrawing",
151 $"{projectAbsolutePath.NameWithExtension} discovery {discoveredFileList.Count} C# files",
152 commonComponentRenderers,
153 dispatcher,
154 TimeSpan.FromSeconds(6));
155
156 foreach (var file in discoveredFileList)
157 {
158 if (!localSolutionVisualizationModel.ParentMap.TryAdd(file, this))
159 localSolutionVisualizationModel.ParentMap[file] = this;
160 }
161
162 foreach (var file in discoveredFileList)
163 {
164 var resourceUri = new ResourceUri(file);
165
166 await textEditorConfig.RegisterModelFunc.Invoke(new RegisterModelArgs(
167 resourceUri,
168 serviceProvider))
169 .ConfigureAwait(false);
170 }

172 async Task DiscoverFilesRecursively(string directoryPathParent, List<string> discoveredFileList)
173 {
174 var directoryPathChildList = await fileSystemProvider.Directory.GetDirectoriesAsync(
175 directoryPathParent,
176 CancellationToken.None)
177 .ConfigureAwait(false);
178
179 var filePathChildList = await fileSystemProvider.Directory.GetFilesAsync(
180 directoryPathParent,
181 CancellationToken.None)
182 .ConfigureAwait(false);
183
184 foreach (var filePathChild in filePathChildList)
185 {
186 if (filePathChild.EndsWith(".cs"))
187 discoveredFileList.Add(filePathChild);
188 }
189
190 foreach (var directoryPathChild in directoryPathChildList)
191 {
192 if (directoryPathChild.Contains(".git") ||
193 directoryPathChild.Contains(".vs") ||
194 directoryPathChild.Contains(".vscode") ||
195 directoryPathChild.Contains(".idea") ||
196 directoryPathChild.Contains("bin") ||
197 directoryPathChild.Contains("obj"))
198 {
199 continue;
200 }
201
202 await DiscoverFilesRecursively(directoryPathChild, discoveredFileList).ConfigureAwait(false);
203 }
204 }
205 }
