89 public async Task HandleKeymapArgumentAsync(KeymapArgs keymapArgs)
90 {
91 var success = ContextRecord.Keymap.MapFirstOrDefault(keymapArgs, out var command);
92
93 if (success && command is not null)
94 await command.CommandFunc(keymapArgs).ConfigureAwait(false);
95 else if (ParentContextBoundary is not null)
96 await ParentContextBoundary.HandleKeymapArgumentAsync(keymapArgs).ConfigureAwait(false);
