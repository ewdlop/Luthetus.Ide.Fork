278 private void Report(
279 TextEditorDiagnosticLevel diagnosticLevel,
280 string message,
281 TextEditorTextSpan textSpan,
282 Guid diagnosticId)
283 {
284 var compilerServiceDiagnosticDecorationKind = diagnosticLevel switch
285 {
286 TextEditorDiagnosticLevel.Hint => CompilerServiceDiagnosticDecorationKind.DiagnosticHint,
287 TextEditorDiagnosticLevel.Suggestion => CompilerServiceDiagnosticDecorationKind.DiagnosticSuggestion,
288 TextEditorDiagnosticLevel.Warning => CompilerServiceDiagnosticDecorationKind.DiagnosticWarning,
289 TextEditorDiagnosticLevel.Error => CompilerServiceDiagnosticDecorationKind.DiagnosticError,
290 TextEditorDiagnosticLevel.Other => CompilerServiceDiagnosticDecorationKind.DiagnosticOther,
291 _ => CompilerServiceDiagnosticDecorationKind.DiagnosticOther,
292 };
293
294 textSpan = textSpan with
295 {
296 DecorationByte = (byte)compilerServiceDiagnosticDecorationKind
297 };
298
299 _diagnosticsBag.Add(new TextEditorDiagnostic(
300 diagnosticLevel,
301 message,
302 textSpan,
303 diagnosticId));
304 }
