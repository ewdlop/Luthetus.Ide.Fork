120 async Task RecursiveSearch(string directoryPath)
121 {
122 cancellationToken.ThrowIfCancellationRequested();
123
124 // Considering the use a breadth first algorithm
125
126 // Search Files
127 {
128 var childFileList = await fileSystemProvider.Directory
129 .GetFilesAsync(directoryPath)
130 .ConfigureAwait(false);
131
132 foreach (var childFile in childFileList)
133 {
134 // TODO: Don't hardcode file extensions here to avoid searching through them.
135 // Reason being, hardcoding them isn't going to work well as a long term solution.
136 // How does one detect if a file is not text?
137 // |
138 // I seem to get away with opening some non-text files, but I think a gif I opened
139 // had 1 million characters in it? So this takes 2 million bytes in a 2byte char?
140 // I'm not sure exactly what happened, I opened the gif and the app froze,
141 // I saw the character only at a glance. (2024-07-20)
142 if (!childFile.EndsWith(".jpg") &&
143 !childFile.EndsWith(".png") &&
144 !childFile.EndsWith(".pdf") &&
145 !childFile.EndsWith(".gif"))
146 {
147 await PerformSearchFile(childFile).ConfigureAwait(false);
148 }
149
150 filesProcessedCount++;
151 ShowFilesProcessedCountOnUi(0);
152 }
153 }
154
155 // Recurse into subdirectories
156 {
157 var subdirectoryList = await fileSystemProvider.Directory
158 .GetDirectoriesAsync(directoryPath)
159 .ConfigureAwait(false);
160
161 foreach (var subdirectory in subdirectoryList)
162 {
163 if (subdirectory.Contains(".git") ||
164 subdirectory.Contains(".vs") ||
165 subdirectory.Contains(".vscode") ||
166 subdirectory.Contains(".idea") ||
167 subdirectory.Contains("bin") ||
168 subdirectory.Contains("obj"))
169 {
170 continue;
171 }
172
173 await RecursiveSearch(subdirectory).ConfigureAwait(false);
174 }
175 }

178 async Task PerformSearchFile(string filePath)
179 {
180 var text = await fileSystemProvider.File.ReadAllTextAsync(filePath).ConfigureAwait(false);
181 var query = textEditorFindAllState.SearchQuery;
182
183 var matchedTextSpanList = new List<TextEditorTextSpan>();
184
185 for (int outerI = 0; outerI < text.Length; outerI++)
186 {
187 if (outerI + query.Length <= text.Length)
188 {
189 int innerI = 0;
190 for (; innerI < query.Length; innerI++)
191 {
192 if (text[outerI + innerI] != query[innerI])
193 break;
194 }
195
196 if (innerI == query.Length)
197 {
198 // Then the entire query was matched
199 matchedTextSpanList.Add(new TextEditorTextSpan(
200 outerI,
201 outerI + innerI,
202 (byte)FindOverlayDecorationKind.LongestCommonSubsequence,
203 new ResourceUri(filePath),
204 text));
205 }
206 }
207 }
208
209 foreach (var matchedTextSpan in matchedTextSpanList)
210 {
211 textSpanList.Add(matchedTextSpan);
212 }
213 }
