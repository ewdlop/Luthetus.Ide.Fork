51 public override Task OnKeyDownAsync(TreeViewCommandArgs commandArgs)
52 {
53 if (commandArgs.KeyboardEventArgs is null)
54 return Task.CompletedTask;
55
56 base.OnKeyDownAsync(commandArgs);
57
58 switch (commandArgs.KeyboardEventArgs.Code)
59 {
60 case KeyboardKeyFacts.WhitespaceCodes.ENTER_CODE:
61 return InvokeOpenInEditor(commandArgs, true);
62 case KeyboardKeyFacts.WhitespaceCodes.SPACE_CODE:
63 return InvokeOpenInEditor(commandArgs, false);
64 }
65
66 if (commandArgs.KeyboardEventArgs.CtrlKey)
67 {
68 CtrlModifiedKeymap(commandArgs);
69 return Task.CompletedTask;
70 }
71 else if (commandArgs.KeyboardEventArgs.AltKey)
72 {
73 AltModifiedKeymap(commandArgs);
74 return Task.CompletedTask;
75 }
76
77 return Task.CompletedTask;
78 }

136 private Task InvokePasteClipboard(TreeViewCommandArgs commandArgs)
137 {

139
140 if (activeNode is not TreeViewNamespacePath treeViewNamespacePath)
141 return Task.CompletedTask;
142
143 MenuOptionRecord pasteMenuOptionRecord;
144
145 if (treeViewNamespacePath.Item.AbsolutePath.IsDirectory)

147 pasteMenuOptionRecord = _menuOptionsFactory.PasteClipboard(

152 SolutionExplorerContextMenu.ParentOfCutFile = null;

155 await ReloadTreeViewModel(localParentOfCutFile).ConfigureAwait(false);
156
157 await ReloadTreeViewModel(treeViewNamespacePath).ConfigureAwait(false);
158 });
159 }
160 else
161 {
162 var parentDirectory = treeViewNamespacePath.Item.AbsolutePath.AncestorDirectoryList.Last();
163 var parentDirectoryAbsolutePath = _environmentProvider.AbsolutePathFactory(parentDirectory.Value, true);
164
165 pasteMenuOptionRecord = _menuOptionsFactory.PasteClipboard(
166 parentDirectoryAbsolutePath,
167 async () =>
168 {
169 var localParentOfCutFile = SolutionExplorerContextMenu.ParentOfCutFile;
170 SolutionExplorerContextMenu.ParentOfCutFile = null;
171
172 if (localParentOfCutFile is not null)
173 await ReloadTreeViewModel(localParentOfCutFile).ConfigureAwait(false);
174
175 await ReloadTreeViewModel(treeViewNamespacePath).ConfigureAwait(false);
176 });
177 }
178
179 if (pasteMenuOptionRecord.OnClickFunc is null)
180 return Task.CompletedTask;
