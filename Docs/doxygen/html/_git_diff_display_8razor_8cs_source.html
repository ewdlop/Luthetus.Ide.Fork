102 private async Task<bool> TryCreateEditorIn(string logFileContent, ResourceUri originalResourceUri)
103 {
104 // Dispose any previously created state for the editor in model
105 if (InResourceUri is not null)
106 {
107 TextEditorService.ModelApi.Dispose(InResourceUri.Value);
108 InResourceUri = null;
109 }
110
111 // Create Model
112 var resourceUri = new ResourceUri(ResourceUriFacts.Diff_ReservedResourceUri_Prefix + originalResourceUri.Value);
113 InResourceUri = resourceUri;
114
115 var fileLastWriteTime = DateTime.UtcNow;
116 var content = logFileContent;
117
118 var absolutePath = EnvironmentProvider.AbsolutePathFactory(resourceUri.Value, false);
119 var decorationMapper = DecorationMapperRegistry.GetDecorationMapper(absolutePath.ExtensionNoPeriod);
120 var compilerService = CompilerServiceRegistry.GetCompilerService(absolutePath.ExtensionNoPeriod);
121
122 var model = new TextEditorModel(
123 resourceUri,
124 fileLastWriteTime,
125 absolutePath.ExtensionNoPeriod,
126 content,
127 decorationMapper,
128 compilerService);
129
130 var modelModifier = new TextEditorModelModifier(model);
131 modelModifier.PerformRegisterPresentationModelAction(CompilerServiceDiagnosticPresentationFacts.EmptyPresentationModel);
132 modelModifier.PerformRegisterPresentationModelAction(FindOverlayPresentationFacts.EmptyPresentationModel);
133 modelModifier.PerformRegisterPresentationModelAction(DiffPresentationFacts.EmptyInPresentationModel);
134
135 model = modelModifier.ToModel();
136
137 TextEditorService.ModelApi.RegisterCustom(model);
138
139 model.CompilerService.RegisterResource(
140 model.ResourceUri,
141 shouldTriggerResourceWasModified: true);
142
143 // Create ViewModel
144 try
145 {
146 // Dispose any previously created state for the editor in view model
147 TextEditorService.ViewModelApi.Dispose(InViewModelKey);
148
149 var viewModelKey = InViewModelKey;
150 var category = new Category("diff-in");
151
152 var viewModel = new TextEditorViewModel(
153 viewModelKey,
154 resourceUri,
155 TextEditorService,
156 Dispatcher,
157 DialogService,
158 JsRuntime,
159 VirtualizationResult<List<RichCharacter>>.GetEmptyRichCharacters(),
160 new TextEditorDimensions(0, 0, 0, 0),
161 new ScrollbarDimensions(0, 0, 0, 0, 0),
162 new CharAndLineMeasurements(0, 0),
163 false,
164 category);
165
166 var firstPresentationLayerKeys = new[]
167 {
168 CompilerServiceDiagnosticPresentationFacts.PresentationKey,
169 FindOverlayPresentationFacts.PresentationKey,
170 DiffPresentationFacts.InPresentationKey,
171 }.ToImmutableArray();
172
173 viewModel.UnsafeState.ShouldSetFocusAfterNextRender = false;
174
175 viewModel = viewModel with
176 {
177 GetTabDisplayNameFunc = _ => absolutePath.NameWithExtension,
178 FirstPresentationLayerKeysList = firstPresentationLayerKeys.ToImmutableList()
179 };
180
181 TextEditorService.ViewModelApi.Register(viewModel);
182 }
183 catch (Exception e)
184 {
185 NotificationHelper.DispatchError(
186 nameof(TryCreateEditorIn),
187 e.ToString(),
188 CommonComponentRenderers,
189 Dispatcher,
190 TimeSpan.FromSeconds(6));
191
192 return false;
193 }
194
195 return true;
196 }

265 private async Task CreateEditorFromLog(
266 GitCliOutputParser gitCliOutputParser,
267 GitState localGitState,
268 GitFile localGitFile,
269 string logFileContent)
270 {
271 /*
272 Goal: Diff Editor (2024-08-26)
273 ==============================
274 Log -> string
275 -> Model (caching?)
276 -> new ResourceUri(ResourceUriFacts.Diff + originalResourceUri)
277 -> ViewModel (caching?)
278 -> new Category("diff-in")
279 -> Add(DiffPresentationLayer)
280 Actual -> string
281 -> Model (more likely to already exist than the log's model?)
282 -> ViewModel (caching?)
283 -> new Category("diff-out")
284 -> Add(DiffPresentationLayer)
285
286 Previously I had been adding the diff presentation layer to the ViewModels of the category: "main".
287 I think this diff presentation layer can and should be removed from ViewModels of the category: "main".
288 */
289
290 TextEditorService.DiffApi.Register(
291 _textEditorDiffModelKey,
292 InViewModelKey,
293 OutViewModelKey);
294
295 _logFileContent = logFileContent;
296 if (_logFileContent is null)
297 return;
298
299 var absolutePathStringStandardized = await TextEditorConfig.AbsolutePathStandardizeFunc.Invoke(
300 localGitFile.AbsolutePath.Value,
301 ServiceProvider);
302
303 var originalResourceUri = new ResourceUri(absolutePathStringStandardized);
304
305 if (!(await TryCreateEditorIn(logFileContent, originalResourceUri)))
306 return;
307
308 if (!(await TryCreateEditorOut(originalResourceUri, localGitFile.AbsolutePath)))
309 return;

331 Console.WriteLine("modelModifier is null || viewModelModifier is null");

346 Console.WriteLine("outPresentationModel.PendingCalculation is null");
347 return Task.CompletedTask;

356

358
359 Console.WriteLine($"lineIndex: {lineIndex} | ({lineInformation.StartPositionIndexInclusive} to {lineInformation.UpperLineEnd.StartPositionIndexInclusive})");

362 lineInformation.StartPositionIndexInclusive,

370 DiffPresentationFacts.OutPresentationKey,
371 DiffPresentationFacts.EmptyOutPresentationModel,
372 outResultTextSpanList.ToImmutableArray());
373 Console.WriteLine("modelModifier.CompletePendingCalculatePresentationModel");
