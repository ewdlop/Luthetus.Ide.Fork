48 private void SelectedProjectToModifyChanged(ChangeEventArgs changeEventArgs, DotNetSolutionState dotNetSolutionState)
49 {
50 if (changeEventArgs.Value is null || dotNetSolutionState.DotNetSolutionModel is null)
51 return;
