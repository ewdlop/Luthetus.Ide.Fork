40 private MenuRecord GetMenuRecord(TreeViewCommandArgs commandArgs)
41 {
42 if (_previousGetMenuRecordInvocation.treeViewCommandArgs == commandArgs)
43 return _previousGetMenuRecordInvocation.menuRecord;
44
45 if (commandArgs.NodeThatReceivedMouseEvent is null)
46 {
47 var menuRecord = MenuRecord.Empty;
48 _previousGetMenuRecordInvocation = (commandArgs, menuRecord);
49 return menuRecord;
50 }
51
52 var menuRecordsList = new List<MenuOptionRecord>();
53
54 var treeViewModel = commandArgs.NodeThatReceivedMouseEvent;
55 var parentTreeViewModel = treeViewModel.Parent;
56
57 var parentTreeViewAbsolutePath = parentTreeViewModel as TreeViewAbsolutePath;
58
59 if (treeViewModel is not TreeViewAbsolutePath treeViewAbsolutePath)
60 {
61 var menuRecord = MenuRecord.Empty;
62 _previousGetMenuRecordInvocation = (commandArgs, menuRecord);
63 return menuRecord;
64 }
65
66 if (treeViewAbsolutePath.Item.IsDirectory)
67 {
68 menuRecordsList.AddRange(GetFileMenuOptions(treeViewAbsolutePath, parentTreeViewAbsolutePath)
69 .Union(GetDirectoryMenuOptions(treeViewAbsolutePath))
70 .Union(GetDebugMenuOptions(treeViewAbsolutePath)));
71 }
72 else
73 {
74 menuRecordsList.AddRange(GetFileMenuOptions(treeViewAbsolutePath, parentTreeViewAbsolutePath)
75 .Union(GetDebugMenuOptions(treeViewAbsolutePath)));
76 }
77
78 // Default case
79 {
80 var menuRecord = new MenuRecord(menuRecordsList.ToImmutableArray());
81 _previousGetMenuRecordInvocation = (commandArgs, menuRecord);
82 return menuRecord;
83 }
84 }
