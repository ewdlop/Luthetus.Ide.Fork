51 private async Task SaveFileAsync(
52 IAbsolutePath absolutePath,
53 string content,
54 Func<DateTime?, Task> onAfterSaveCompletedWrittenDateTimeFunc,
55 CancellationToken cancellationToken = default)
56 {
57 if (cancellationToken.IsCancellationRequested)
58 return;
59
60 var absolutePathString = absolutePath.Value;
61
62 if (absolutePathString is not null &&
63 await _fileSystemProvider.File.ExistsAsync(absolutePathString).ConfigureAwait(false))
64 {
65 await _fileSystemProvider.File.WriteAllTextAsync(absolutePathString, content).ConfigureAwait(false);
66 }
67 else
68 {
69 // TODO: Save As to make new file
70 NotificationHelper.DispatchInformative("Save Action", "File not found. TODO: Save As", _commonComponentRenderers, _dispatcher, TimeSpan.FromSeconds(7));
71 }
72
73 DateTime? fileLastWriteTime = null;
74
75 if (absolutePathString is not null)
76 {
77 fileLastWriteTime = await _fileSystemProvider.File.GetLastWriteTimeAsync(
78 absolutePathString,
79 CancellationToken.None)
80 .ConfigureAwait(false);
81 }
82
83 if (onAfterSaveCompletedWrittenDateTimeFunc is not null)
84 await onAfterSaveCompletedWrittenDateTimeFunc.Invoke(fileLastWriteTime);
85 }
