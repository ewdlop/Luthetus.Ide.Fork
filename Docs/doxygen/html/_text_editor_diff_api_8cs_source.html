42 public Func<ITextEditorEditContext, Task> CalculateFactory(
43 Key<TextEditorDiffModel> diffModelKey,
44 CancellationToken cancellationToken)
45 {
46 return editContext =>
47 {
48 if (cancellationToken.IsCancellationRequested)
49 return Task.CompletedTask;
50
51 var diffModelModifier = editContext.GetDiffModelModifier(diffModelKey);
52
53 if (diffModelModifier is null)
54 return Task.CompletedTask;
55
56 var inViewModelModifier = editContext.GetViewModelModifier(diffModelModifier.DiffModel.InViewModelKey);
57 var outViewModelModifier = editContext.GetViewModelModifier(diffModelModifier.DiffModel.OutViewModelKey);
58
59 if (inViewModelModifier is null || outViewModelModifier is null)
60 return Task.CompletedTask;
61
62 var inModelModifier = editContext.GetModelModifier(inViewModelModifier.ViewModel.ResourceUri);
63 var outModelModifier = editContext.GetModelModifier(outViewModelModifier.ViewModel.ResourceUri);
64
65 if (inModelModifier is null || outModelModifier is null)
66 return Task.CompletedTask;
67
68 // In
69 editContext.TextEditorService.ModelApi.StartPendingCalculatePresentationModel(
70 editContext,
71 inModelModifier,
72 DiffPresentationFacts.InPresentationKey,
73 DiffPresentationFacts.EmptyInPresentationModel);
74 var inPresentationModel = inModelModifier.PresentationModelList.First(
75 x => x.TextEditorPresentationKey == DiffPresentationFacts.InPresentationKey);
76 if (inPresentationModel.PendingCalculation is null)
77 return Task.CompletedTask;
78 var inText = inPresentationModel.PendingCalculation.ContentAtRequest;
79
80 // Out
81 editContext.TextEditorService.ModelApi.StartPendingCalculatePresentationModel(
82 editContext,
83 outModelModifier,
84 DiffPresentationFacts.OutPresentationKey,
85 DiffPresentationFacts.EmptyOutPresentationModel);
86 var outPresentationModel = outModelModifier.PresentationModelList.First(
87 x => x.TextEditorPresentationKey == DiffPresentationFacts.OutPresentationKey);
88 if (outPresentationModel.PendingCalculation is null)
89 return Task.CompletedTask;
90 var outText = outPresentationModel.PendingCalculation.ContentAtRequest;
91
92 var diffResult = TextEditorDiffResult.Calculate(
93 inModelModifier.ResourceUri,
94 inText,
95 outModelModifier.ResourceUri,
96 outText);
97
98 inModelModifier.CompletePendingCalculatePresentationModel(
99 DiffPresentationFacts.InPresentationKey,
100 DiffPresentationFacts.EmptyInPresentationModel,
101 diffResult.InResultTextSpanList.ToImmutableArray());
102
103 outModelModifier.CompletePendingCalculatePresentationModel(
104 DiffPresentationFacts.OutPresentationKey,
105 DiffPresentationFacts.EmptyOutPresentationModel,
106 diffResult.OutResultTextSpanList.ToImmutableArray());
107
108 return Task.CompletedTask;
109 };
110 }
