245 public void ApplyDecorationRange(
246 ITextEditorEditContext editContext,
247 TextEditorModelModifier modelModifier,
248 IEnumerable<TextEditorTextSpan> textSpans)
249 {
250 var localRichCharacterList = modelModifier.RichCharacterList;
251
252 var positionsPainted = new HashSet<int>();
253
254 foreach (var textEditorTextSpan in textSpans)
255 {
256 for (var i = textEditorTextSpan.StartingIndexInclusive; i < textEditorTextSpan.EndingIndexExclusive; i++)
257 {
258 if (i < 0 || i >= localRichCharacterList.Length)
259 continue;
260
261 modelModifier.__SetDecorationByte(i, textEditorTextSpan.DecorationByte);
262 positionsPainted.Add(i);
263 }
264 }
265
266 for (var i = 0; i < localRichCharacterList.Length - 1; i++)
267 {
268 if (!positionsPainted.Contains(i))
269 {
270 // DecorationByte of 0 is to be 'None'
271 modelModifier.__SetDecorationByte(i, 0);
272 }
273 }
274
275 modelModifier.ShouldReloadVirtualizationResult = true;
276 }

278 public void ApplySyntaxHighlighting(
279 ITextEditorEditContext editContext,
280 TextEditorModelModifier modelModifier)
281 {
282 var syntacticTextSpansList = modelModifier.CompilerService.GetTokenTextSpansFor(modelModifier.ResourceUri);
283 var symbolsList = modelModifier.CompilerService.GetSymbolsFor(modelModifier.ResourceUri);
284
285 var symbolTextSpansList = symbolsList.Select(s => s.TextSpan);
286
287 var textSpanList = new List<TextEditorTextSpan>();
288 textSpanList.AddRange(syntacticTextSpansList);
289 textSpanList.AddRange(symbolTextSpansList);
290
291 ApplyDecorationRange(
292 editContext,
293 modelModifier,
294 textSpanList);
295 }
