42 ResourceUri inResourceUri,
43 string inText,
44 ResourceUri outResourceUri,
45 string outText)
46 {
47 // Need to build a square two dimensional array.
48 //
49 // Envisioning that:
50 // -Each character of 'inText' represents a row.
51 // -Each character of 'outText' represents a column.
52
53 var squaredSize = Math.Max(inText.Length, outText.Length);
54
55 var diffMatrix = new TextEditorDiffCell[squaredSize, squaredSize];
56
57 (int sourceWeight, int inIndex, int outIndex) highestSourceWeightTuple = (-1, -1, -1);
58
59 for (int inIndex = 0; inIndex < inText.Length; inIndex++)
60 {
61 char inCharValue = inText[inIndex];
62
63 for (int outIndex = 0; outIndex < outText.Length; outIndex++)
64 {
65 char outCharValue = outText[outIndex];
66
67 var weight = GetLargestWeightPriorToCurrentPosition(diffMatrix, inIndex, outIndex);
68 var isSourceWeight = false;
69
70 if (inCharValue == outCharValue)
71 {
72 var cellSourceWeight = weight + 1;
73
74 if (cellSourceWeight > weight)
75 {
76 weight = cellSourceWeight;
77 isSourceWeight = true;
78 }
79
80 if (cellSourceWeight > highestSourceWeightTuple.sourceWeight)
81 highestSourceWeightTuple = (cellSourceWeight, inIndex, outIndex);
82 }
83
84 diffMatrix[inIndex, outIndex] = new TextEditorDiffCell(
85 inCharValue, outCharValue, weight, isSourceWeight);
86 }
87
88 for (int fabricatedOutIndex = outText.Length; fabricatedOutIndex < squaredSize; fabricatedOutIndex++)
89 {
90 // This for loop sets the cells in the fabricated column in order to create a square matrix
91 // in cases where (outTextLength < inTextLength)
92
93 var rowLargestWeightPriorToCurrentColumn = GetLargestWeightPriorToCurrentPosition(
94 diffMatrix,
95 inIndex,
96 fabricatedOutIndex);
97
98 diffMatrix[inIndex, fabricatedOutIndex] = new TextEditorDiffCell(
99 inCharValue,
100 null,
101 rowLargestWeightPriorToCurrentColumn,
102 false);
103 }
104 }
105
106 for (int fabricatedInIndex = inText.Length; fabricatedInIndex < squaredSize; fabricatedInIndex++)
107 {
108 // This for loop sets the cells in the fabricated row in order to create a square matrix
109 // in cases where (inTextLength < outTextLength)
110 //
111 // TODO: This logic should to be removed. Instead of looking at this algorithm from the perspective of 'before' and 'after' text. It might be best to have the perspective be 'vertical' and 'horizontal' text. Then the 'vertical' text is equal to the longer of the two string inputs. By having the row count defined by the longer of the two strings you only would ever fabricate columns. This would allow for a lot of code to be removed.
112
113 for (int fabricatedOutIndex = 0; fabricatedOutIndex < squaredSize; fabricatedOutIndex++)
114 {
115 var rowLargestWeightPriorToCurrentColumn = GetLargestWeightPriorToCurrentPosition(
116 diffMatrix,
117 fabricatedInIndex,
118 fabricatedOutIndex);
119
120 diffMatrix[fabricatedInIndex, fabricatedOutIndex] = new TextEditorDiffCell(
121 null,
122 null,
123 rowLargestWeightPriorToCurrentColumn,
124 false);
125 }
126 }

128 // The abbreviation "lcs" is to mean "Longest Common Subsequence"
129 var lcsBuilder = new StringBuilder();
130
131 var inPositionIndicesLcsHashSet = new HashSet<int>();
132 var outPositionIndicesLcsHashSet = new HashSet<int>();
133
134 var inPositionIndicesOfDeletionHashSet = new HashSet<int>();
135 var outPositionIndicesInsertionHashSet = new HashSet<int>();
136
137 var inPositionIndicesOfModificationHashSet = new HashSet<int>();
138 var outPositionIndicesOfModificationHashSet = new HashSet<int>();
139
140 // Read the LongestCommonSubsequence by backtracking to the highest weights
141 {
142 var runningRowIndex = highestSourceWeightTuple.inIndex;
143 var runningColumnIndex = highestSourceWeightTuple.outIndex;
144
145 var targetSourceWeight = highestSourceWeightTuple.sourceWeight;
146
147 var foundLargestWeight = false;
148
149 var restoreColumnIndex = runningColumnIndex;
150
151 while (runningRowIndex != -1 && runningColumnIndex != -1)
152 {
153 if (!foundLargestWeight)
154 restoreColumnIndex = runningColumnIndex;
155
156 var cell = diffMatrix[runningRowIndex, runningColumnIndex];
157
158 if (cell.IsSourceOfRowWeight &&
159 cell.Weight == targetSourceWeight)
160 {
161 targetSourceWeight--;
162
163 if (cell.BeforeCharValue != cell.AfterCharValue)
164 throw new LuthetusTextEditorException($"The {nameof(cell.BeforeCharValue)}:'{cell.BeforeCharValue}' was not equal to the {nameof(cell.AfterCharValue)}:'{cell.AfterCharValue}'");
165
166 lcsBuilder.Append(cell.BeforeCharValue);
167
168 // Decoration logic for longest common subsequence
169 {
170 inPositionIndicesLcsHashSet.Add(runningRowIndex);
171 outPositionIndicesLcsHashSet.Add(runningColumnIndex);
172 }
173
174 restoreColumnIndex = runningColumnIndex - 1;
175
176 // Forces the row index decrementation then a column index restoration.
177 runningColumnIndex = 0;
178 }
179 else
180 {
181 // Decoration logic
182 {
183 if (outText.Length > inText.Length && !outPositionIndicesLcsHashSet.Contains(runningColumnIndex))
184 outPositionIndicesInsertionHashSet.Add(runningColumnIndex); // Insertion
185 else if (inText.Length > outText.Length && runningRowIndex >= outText.Length)
186 inPositionIndicesOfDeletionHashSet.Add(runningRowIndex); // Deletion
187 // TODO: Else if for modification is not working.
188 //
189 // else if (cell.BeforeCharValue != cell.AfterCharValue)
190 // {
191 // // Modification
192 // beforePositionIndicesOfModificationHashSet.Add(runningRowIndex);
193 // afterPositionIndicesOfModificationHashSet.Add(runningColumnIndex);
194 // }
195 }
196 }
197
198 if (runningColumnIndex == 0)
199 {
200 runningColumnIndex = restoreColumnIndex;
201 runningRowIndex--;
202 }
203 else
204 {
205 runningColumnIndex--;
206 }
207 }
208 }
209
210 var longestCommonSubsequenceValue = new string(lcsBuilder
211 .ToString()
212 .Reverse()
213 .ToArray());
214
215 var inTextSpans = new List<TextEditorTextSpan>();
216 var outTextSpans = new List<TextEditorTextSpan>();
217
218 // Decoration logic
219 {
220 // Longest common subsequence
221 {
222 inTextSpans.AddRange(GetTextSpans(
223 inResourceUri,
224 inText,
225 inPositionIndicesLcsHashSet,
226 (byte)TextEditorDiffDecorationKind.LongestCommonSubsequence));
227
228 outTextSpans.AddRange(GetTextSpans(
229 outResourceUri,
230 outText,
231 outPositionIndicesLcsHashSet,
232 (byte)TextEditorDiffDecorationKind.LongestCommonSubsequence));
233 }
234
235 // Insertion
236 {
237 outTextSpans.AddRange(GetTextSpans(
238 outResourceUri,
239 outText,
240 outPositionIndicesInsertionHashSet,
241 (byte)TextEditorDiffDecorationKind.Insertion));
242 }
243
244 // Deletion
245 {
246 inTextSpans.AddRange(GetTextSpans(
247 inResourceUri,
248 inText,
249 inPositionIndicesOfDeletionHashSet,
250 (byte)TextEditorDiffDecorationKind.Deletion));
251 }
252
253 // Modification
254 {
255 inTextSpans.AddRange(GetTextSpans(
256 inResourceUri,
257 inText,
258 inPositionIndicesOfModificationHashSet,
259 (byte)TextEditorDiffDecorationKind.Modification));
260
261 outTextSpans.AddRange(GetTextSpans(
262 outResourceUri,
263 outText,
264 outPositionIndicesOfModificationHashSet,
265 (byte)TextEditorDiffDecorationKind.Modification));
266 }
267 }
268
269 var diffResult = new TextEditorDiffResult(
270 inText,
271 outText,
272 diffMatrix,
273 highestSourceWeightTuple,
274 longestCommonSubsequenceValue,
275 inTextSpans.ToImmutableList(),
276 outTextSpans.ToImmutableList());
277
278 return diffResult;
279 }

281 private static int GetLargestWeightPriorToCurrentPosition(
282 TextEditorDiffCell[,] diffMatrix,
283 int inIndex,
284 int outIndex)
285 {
286 var largestWeight = 0;
287
288 for (int rowI = 0; rowI < inIndex; rowI++)
289 {
290 for (int columnI = 0; columnI < outIndex; columnI++)
291 {
292 var currentWeight = diffMatrix[rowI, columnI].Weight;
293
294 if (currentWeight > largestWeight)
295 largestWeight = currentWeight;
296 }
297 }
298
299 return largestWeight;
300 }

302 private static List<TextEditorTextSpan> GetTextSpans(
303 ResourceUri resourceUri,
304 string sourceText,
305 HashSet<int> positionIndicesHashSet,
306 byte decorationByte)
307 {
308 var matchTextSpans = new List<TextEditorTextSpan>();
309
310 if (!positionIndicesHashSet.Any())
311 return matchTextSpans;
312
313 var sortedPositionIndicesThatMatch = positionIndicesHashSet
314 .OrderBy(x => x)
315 .ToList();
316
317 // The foreach loop coalesces contiguous indices into a single TextEditorTextSpan
318 // and the logic that does the coalescing will not write out the final index without this.
319 sortedPositionIndicesThatMatch.Add(int.MaxValue);
320
321 var startingIndexInclusive = sortedPositionIndicesThatMatch.First();
322 var endingIndexExclusive = startingIndexInclusive + 1;
323
324 var skipFirstIndex = sortedPositionIndicesThatMatch
325 .Skip(1)
326 .ToArray();
327
328 foreach (var index in skipFirstIndex)
329 {
330 if (index == endingIndexExclusive)
331 {
332 endingIndexExclusive++;
333 }
334 else
335 {
336 var textSpan = new TextEditorTextSpan(
337 startingIndexInclusive,
338 endingIndexExclusive,
339 decorationByte,
340 resourceUri,
341 sourceText);
342
343 matchTextSpans.Add(textSpan);
344
345 startingIndexInclusive = index;
346 endingIndexExclusive = index + 1;
347 }
348 }
349
350 return matchTextSpans;
351 }
