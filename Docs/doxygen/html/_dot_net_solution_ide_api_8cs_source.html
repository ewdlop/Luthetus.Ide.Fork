314 private Task ParseSolution(Key<DotNetSolutionModel> dotNetSolutionModelKey)
315 {
316 var dotNetSolutionState = _dotNetSolutionStateWrap.Value;
317
318 var dotNetSolutionModel = dotNetSolutionState.DotNetSolutionsList.FirstOrDefault(
319 x => x.Key == dotNetSolutionModelKey);
320
321 if (dotNetSolutionModel is null)
322 return Task.CompletedTask;
323
324 _ = Task.Run(async () =>
325 {
326 var cancellationTokenSource = new CancellationTokenSource();
327 var cancellationToken = cancellationTokenSource.Token;
328
329 var progressBarModel = new ProgressBarModel(0, "parsing...")
330 {
331 OnCancelFunc = () =>
332 {
333 cancellationTokenSource.Cancel();
334 cancellationTokenSource.Dispose();
335 return Task.CompletedTask;
336 }
337 };
338
339 NotificationHelper.DispatchProgress(
340 $"Parse: {dotNetSolutionModel.AbsolutePath.NameWithExtension}",
341 progressBarModel,
342 _commonComponentRenderers,
343 _dispatcher,
344 TimeSpan.FromMilliseconds(-1));
345
346 // var progressThrottle = new Throttle(TimeSpan.FromMilliseconds(100));
347 var progressThrottle = new ThrottleOptimized<(ParseSolutionStageKind StageKind, double? Progress, string? MessageOne, string? MessageTwo)>(TimeSpan.FromMilliseconds(200), (tuple, _) =>
348 {
349 switch (tuple.StageKind)
350 {
351 case ParseSolutionStageKind.A:
352 progressBarModel.SetProgress(tuple.Progress, tuple.MessageOne);
353 return Task.CompletedTask;
354 case ParseSolutionStageKind.B:
355 progressBarModel.SetProgress(tuple.Progress, tuple.MessageOne, tuple.MessageTwo);
356 progressBarModel.Dispose();
357 return Task.CompletedTask;
358 case ParseSolutionStageKind.C:
359 progressBarModel.SetProgress(tuple.Progress, tuple.MessageOne);
360 progressBarModel.Dispose();
361 return Task.CompletedTask;
362 case ParseSolutionStageKind.D:
363 progressBarModel.SetProgress(tuple.Progress, tuple.MessageOne, tuple.MessageTwo);
364 return Task.CompletedTask;
365 case ParseSolutionStageKind.E:
366 progressBarModel.SetProgress(tuple.Progress, tuple.MessageOne, tuple.MessageTwo);
367 return Task.CompletedTask;
368 default:
369 return Task.CompletedTask;
370 }
371 });
372
373 try
374 {
375 if (_textEditorService.TextEditorConfig.RegisterModelFunc is null)
376 return;
377
378 progressThrottle.Run((ParseSolutionStageKind.A, 0.05, "Discovering projects...", null));
379
380 foreach (var project in dotNetSolutionModel.DotNetProjectList)
381 {
382 RegisterStartupControl(project);
383
384 var resourceUri = new ResourceUri(project.AbsolutePath.Value);
385
386 if (!await _fileSystemProvider.File.ExistsAsync(resourceUri.Value))
387 continue; // TODO: This can still cause a race condition exception if the file is removed before the next line runs.
388
389 var registerModelArgs = new RegisterModelArgs(resourceUri, _serviceProvider)
390 {
391 ShouldBlockUntilBackgroundTaskIsCompleted = true,
392 };
393
394 await _textEditorService.TextEditorConfig.RegisterModelFunc.Invoke(registerModelArgs)
395 .ConfigureAwait(false);
396 }
397
398 var previousStageProgress = 0.05;
399 var dotNetProjectListLength = dotNetSolutionModel.DotNetProjectList.Length;
400 var projectsParsedCount = 0;
401 foreach (var project in dotNetSolutionModel.DotNetProjectList)
402 {
403 // foreach project in solution
404 // foreach C# file in project
405 // EnqueueBackgroundTask(async () =>
406 // {
407 // ParseCSharpFile();
408 // UpdateProgressBar();
409 // });
410 //
411 // Treat every project as an equal weighting with relation to remaining percent to complete
412 // on the progress bar.
413 //
414 // If the project were to be parsed, how much would it move the percent progress completed by?
415 //
416 // Then, in order to see progress while each C# file in the project gets parsed,
417 // multiply the percent progress this project can provide by the proportion
418 // of the project's C# files which have been parsed.
419 var maximumProgressAvailableToProject = (1 - previousStageProgress) * ((double)1.0 / dotNetProjectListLength);
420 var currentProgress = Math.Min(1.0, previousStageProgress + maximumProgressAvailableToProject * projectsParsedCount);
421
422 // This 'SetProgress' is being kept out the throttle, since it sets message 1
423 // whereas the per class progress updates set message 2.
424 //
425 // Otherwise an update to message 2 could result in this message 1 update never being written.
426 progressBarModel.SetProgress(
427 currentProgress,
428 $"{projectsParsedCount + 1}/{dotNetProjectListLength}: {project.AbsolutePath.NameWithExtension}");
429
430 cancellationToken.ThrowIfCancellationRequested();
431
432 await DiscoverClassesInProject(project, progressBarModel, progressThrottle, currentProgress, maximumProgressAvailableToProject);
433 projectsParsedCount++;
434 }
435
436 progressThrottle.Run((ParseSolutionStageKind.B, 1, $"Finished parsing: {dotNetSolutionModel.AbsolutePath.NameWithExtension}", string.Empty));
437 }
438 catch (Exception e)
439 {
440 if (e is OperationCanceledException)
441 progressBarModel.IsCancelled = true;
442
443 var currentProgress = progressBarModel.GetProgress();
444
445 progressThrottle.Run((ParseSolutionStageKind.C, currentProgress, e.ToString(), null));
446 }
447 });
448
449 return Task.CompletedTask;
450 }
