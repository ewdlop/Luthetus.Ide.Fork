238 public static void HandleForTokenKeyword(CSharpParserModel model)
239 {
240 var forKeywordToken = (KeywordToken)model.TokenWalker.Consume();
241
242 var openParenthesisToken = (OpenParenthesisToken)model.TokenWalker.Match(SyntaxKind.OpenParenthesisToken);
243
244 // Initialization Case One
245 // ;
246 var initializationExpressionNode = (IExpressionNode)new EmptyExpressionNode(CSharpFacts.Types.Void.ToTypeClause());
247 var initializationStatementDelimiterToken = (StatementDelimiterToken)model.TokenWalker.Match(SyntaxKind.StatementDelimiterToken);
248 var badStateInitialization = false;
249
250 if (initializationStatementDelimiterToken.IsFabricated)
251 {
252 // Initialization Case Two
253 // i = 0;
254 var identifierToken = (IdentifierToken)model.TokenWalker.Match(SyntaxKind.IdentifierToken);
255
256 if (identifierToken.IsFabricated)
257 {
258 // Initialization Case Three
259 // int i = 0;
260 var typeClauseNode = model.TokenWalker.MatchTypeClauseNode(model);
261 var isCaseThree = !typeClauseNode.IsFabricated;
262
263 if (isCaseThree)
264 {
265 identifierToken = (IdentifierToken)model.TokenWalker.Match(SyntaxKind.IdentifierToken);
266 }
267 else
268 {
269 // Initialization Case Four
270 // bad syntax?
271 badStateInitialization = true;
272 }
273 }
274
275 if (!badStateInitialization)
276 {
277 // Read the remainder
278 // = 0;
279 var equalsToken = (EqualsToken)model.TokenWalker.Match(SyntaxKind.EqualsToken);
280
281 model.ExpressionList.Add((SyntaxKind.CloseParenthesisToken, null));
282 initializationExpressionNode = ParseOthers.ParseExpression(model);
283
284 initializationStatementDelimiterToken = (StatementDelimiterToken)model.TokenWalker.Match(SyntaxKind.StatementDelimiterToken);
285 }
286 }
287
288 // Condition Case One
289 // ;
290 var conditionExpressionNode = (IExpressionNode)new EmptyExpressionNode(CSharpFacts.Types.Void.ToTypeClause());
291 var conditionStatementDelimiterToken = (StatementDelimiterToken)model.TokenWalker.Match(SyntaxKind.StatementDelimiterToken);
292
293 if (conditionStatementDelimiterToken.IsFabricated)
294 {
295 // Condition Case Two
296 // i < 10;
297
298 model.ExpressionList.Add((SyntaxKind.CloseParenthesisToken, null));
299 conditionExpressionNode = ParseOthers.ParseExpression(model);
300
301 conditionStatementDelimiterToken = (StatementDelimiterToken)model.TokenWalker.Match(SyntaxKind.StatementDelimiterToken);
302 }
303
304 // Updation Case One
305 // )
306 var updationExpressionNode = (IExpressionNode)new EmptyExpressionNode(CSharpFacts.Types.Void.ToTypeClause());
307 var closeParenthesisToken = (CloseParenthesisToken)model.TokenWalker.Match(SyntaxKind.CloseParenthesisToken);
308
309 if (closeParenthesisToken.IsFabricated)
310 {
311 model.ExpressionList.Add((SyntaxKind.CloseParenthesisToken, null));
312 updationExpressionNode = ParseOthers.ParseExpression(model);
313
314 closeParenthesisToken = (CloseParenthesisToken)model.TokenWalker.Match(SyntaxKind.CloseParenthesisToken);
315
316 if (closeParenthesisToken.IsFabricated)
317 {
318 while (!model.TokenWalker.IsEof)
319 {
320 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CloseParenthesisToken)
321 break;
322
323 _ = model.TokenWalker.Consume();
324 }
325 }
326 }
327
328 var forStatementNode = new ForStatementNode(
329 forKeywordToken,
330 openParenthesisToken,
331 ImmutableArray<ISyntax>.Empty,
332 initializationStatementDelimiterToken,
333 conditionExpressionNode,
334 conditionStatementDelimiterToken,
335 updationExpressionNode,
336 closeParenthesisToken,
337 codeBlockNode: null);
338
339 model.SyntaxStack.Push(forStatementNode);
340 model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner = forStatementNode;
341 }
