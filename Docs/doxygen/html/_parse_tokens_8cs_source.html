22 public static void ParseIdentifierToken(CSharpParserModel model)
23 {
24 var originalTokenIndex = model.TokenWalker.Index;
25
26 model.TryParseExpressionSyntaxKindList.Add(SyntaxKind.TypeClauseNode);
27 model.TryParseExpressionSyntaxKindList.Add(SyntaxKind.VariableDeclarationNode);
28 model.TryParseExpressionSyntaxKindList.Add(SyntaxKind.VariableReferenceNode);
29 model.TryParseExpressionSyntaxKindList.Add(SyntaxKind.ConstructorInvocationExpressionNode);
30
31 if (model.CurrentCodeBlockBuilder.CodeBlockOwner is not null &&
32 model.CurrentCodeBlockBuilder.CodeBlockOwner.SyntaxKind != SyntaxKind.TypeDefinitionNode)
33 {
34 // There is a syntax conflict between a ConstructorDefinitionNode and a FunctionInvocationNode.
35 //
36 // Disambiguation is done based on the 'CodeBlockOwner' until a better solution is found.
37 //
38 // If the supposed "ConstructorDefinitionNode" does not have the same name as
39 // the CodeBlockOwner.
40 //
41 // Then, it perhaps should be treated as a function invocation (or function definition).
42 // The main case for this being someone typing out pseudo code within a CodeBlockOwner
43 // that is a TypeDefinitionNode.
44 model.TryParseExpressionSyntaxKindList.Add(SyntaxKind.FunctionInvocationNode);
45 }
46
47 var successParse = ParseOthers.TryParseExpression(null, model, out var expressionNode);
48
49 if (!successParse)
50 {
51 expressionNode = ParseOthers.ParseExpression(model);
52 model.StatementBuilder.ChildList.Add(expressionNode);
53 return;
54 }
55
56 switch (expressionNode.SyntaxKind)
57 {
58 case SyntaxKind.TypeClauseNode:
59 MoveToHandleTypeClauseNode(originalTokenIndex, (TypeClauseNode)expressionNode, model);
60 return;
61 case SyntaxKind.VariableDeclarationNode:
62 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenParenthesisToken ||
63 model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenAngleBracketToken)
64 {
65 MoveToHandleFunctionDefinition((VariableDeclarationNode)expressionNode, model);
66 return;
67 }
68
69 MoveToHandleVariableDeclarationNode((VariableDeclarationNode)expressionNode, model);
70 return;
71 case SyntaxKind.VariableReferenceNode:
72 case SyntaxKind.FunctionInvocationNode:
73 case SyntaxKind.ConstructorInvocationExpressionNode:
74 model.StatementBuilder.ChildList.Add(expressionNode);
75 return;
76 default:
77 model.DiagnosticBag.ReportTodoException(model.TokenWalker.Current.TextSpan, $"nameof(ParseIdentifierToken) default case");
78 return;
79 }
80 }

91 public static void MoveToHandleVariableDeclarationNode(IVariableDeclarationNode variableDeclarationNode, IParserModel model)
92 {
93 var variableKind = VariableKind.Local;
94
95 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenBraceToken ||
96 (model.TokenWalker.Current.SyntaxKind == SyntaxKind.EqualsToken &&
97 model.TokenWalker.Next.SyntaxKind == SyntaxKind.CloseAngleBracketToken))
98 {
99 variableKind = VariableKind.Property;
100 }
101 else if (model.CurrentCodeBlockBuilder.CodeBlockOwner is not null &&
102 model.CurrentCodeBlockBuilder.CodeBlockOwner.SyntaxKind == SyntaxKind.TypeDefinitionNode)
103 {
104 variableKind = VariableKind.Field;
105 }
106
107 ((VariableDeclarationNode)variableDeclarationNode).VariableKind = variableKind;
108
109 model.Binder.BindVariableDeclarationNode(variableDeclarationNode, model);
110 model.CurrentCodeBlockBuilder.ChildList.Add(variableDeclarationNode);
111 model.StatementBuilder.ChildList.Add(variableDeclarationNode);
112
113 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenBraceToken)
114 {
115 ParsePropertyDefinition((CSharpParserModel)model);
116 }
117 else if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.EqualsToken &&
118 model.TokenWalker.Next.SyntaxKind == SyntaxKind.CloseAngleBracketToken)
119 {
120 ParsePropertyDefinition_ExpressionBound((CSharpParserModel)model);
121 }

124 public static void MoveToHandleTypeClauseNode(int originalTokenIndex, TypeClauseNode typeClauseNode, IParserModel model)
125 {
126 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.StatementDelimiterToken ||
127 model.TokenWalker.Current.SyntaxKind == SyntaxKind.EndOfFileToken ||
128 model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenBraceToken ||
129 model.TokenWalker.Current.SyntaxKind == SyntaxKind.CloseBraceToken)
130 {
131 model.StatementBuilder.ChildList.Add(typeClauseNode);
132 }
133 else if (model.CurrentCodeBlockBuilder.CodeBlockOwner is TypeDefinitionNode typeDefinitionNode &&
134 UtilityApi.IsConvertibleToIdentifierToken(typeClauseNode.TypeIdentifierToken.SyntaxKind) &&
135 model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenParenthesisToken &&
136 typeDefinitionNode.TypeIdentifierToken.TextSpan.GetText() == typeClauseNode.TypeIdentifierToken.TextSpan.GetText())
137 {
138 // ConstructorDefinitionNode
139
140 var identifierToken = UtilityApi.ConvertToIdentifierToken(typeClauseNode.TypeIdentifierToken, model);
141
142 ParseFunctions.HandleConstructorDefinition(
143 typeDefinitionNode,
144 identifierToken,
145 (CSharpParserModel)model);
146 }
147 else
148 {
149 model.StatementBuilder.ChildList.Add(typeClauseNode);
150 }
151
152 return;
153 }

155 public static void ParsePropertyDefinition(CSharpParserModel model)
156 {
157 #if DEBUG
158 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = true;
159 #endif
160
161 var openBraceToken = (OpenBraceToken)model.TokenWalker.Consume();
162
163 var openBraceCounter = 1;
164
165 while (true)
166 {
167 if (model.TokenWalker.IsEof)
168 break;
169
170 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenBraceToken)
171 {
172 ++openBraceCounter;
173 }
174 else if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CloseBraceToken)
175 {
176 if (--openBraceCounter <= 0)
177 break;
178 }
179
180 _ = model.TokenWalker.Consume();
181 }
182
183 var closeTokenIndex = model.TokenWalker.Index;
184 var closeBraceToken = (CloseBraceToken)model.TokenWalker.Match(SyntaxKind.CloseBraceToken);
185
186 #if DEBUG
187 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = false;
188 #endif
189 }

226 public static void ParseOpenBraceToken(OpenBraceToken openBraceToken, CSharpParserModel model)
227 {
228 if (model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner is null)
229 {
230 var arbitraryCodeBlockNode = new ArbitraryCodeBlockNode(model.CurrentCodeBlockBuilder.CodeBlockOwner);
231 model.SyntaxStack.Push(arbitraryCodeBlockNode);
232 model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner = arbitraryCodeBlockNode;
233 }
234
235 model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner.SetOpenBraceToken(openBraceToken, model);
236
237 var parentScopeDirection = model.CurrentCodeBlockBuilder?.CodeBlockOwner?.ScopeDirectionKind ?? ScopeDirectionKind.Both;
238 if (parentScopeDirection == ScopeDirectionKind.Both)
239 {
240 if (!model.CurrentCodeBlockBuilder.PermitInnerPendingCodeBlockOwnerToBeParsed)
241 {
242 model.TokenWalker.DeferParsingOfChildScope(openBraceToken, model);
243 return;
244 }
245
246 model.CurrentCodeBlockBuilder.PermitInnerPendingCodeBlockOwnerToBeParsed = false;
247 }
248
249 var nextCodeBlockOwner = model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner;
250 var nextReturnTypeClauseNode = nextCodeBlockOwner.GetReturnTypeClauseNode();
251
252 model.Binder.OpenScope(nextCodeBlockOwner, nextReturnTypeClauseNode, openBraceToken.TextSpan, model);
253 model.CurrentCodeBlockBuilder = new(parent: model.CurrentCodeBlockBuilder, codeBlockOwner: nextCodeBlockOwner);
254 nextCodeBlockOwner.OnBoundScopeCreatedAndSetAsCurrent(model);
255 }

298 public static void ParseOpenSquareBracketToken(CSharpParserModel model)
299 {
300 var openSquareBracketToken = (OpenSquareBracketToken)model.TokenWalker.Consume();
301
302 if (model.StatementBuilder.ChildList.Count != 0)
303 {
304 model.DiagnosticBag.ReportTodoException(
305 openSquareBracketToken.TextSpan,
306 $"Unexpected '{nameof(OpenSquareBracketToken)}'");
307 return;
308 }
309 var openSquareBracketCounter = 1;
310 var corruptState = false;
311
312 #if DEBUG
313 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = true;
314 #endif
315
316 while (!model.TokenWalker.IsEof)
317 {
318 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenSquareBracketToken)
319 {
320 ++openSquareBracketCounter;
321 }
322 else if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CloseSquareBracketToken)
323 {
324 if (--openSquareBracketCounter <= 0)
325 break;
326 }
327 else if (!corruptState)
328 {
329 var tokenIndexOriginal = model.TokenWalker.Index;
330
331 model.ExpressionList.Add((SyntaxKind.CloseSquareBracketToken, null));
332 model.ExpressionList.Add((SyntaxKind.CommaToken, null));
333 var expression = ParseOthers.ParseExpression(model);
334
335 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CommaToken)
336 _ = model.TokenWalker.Consume();
337
338 if (tokenIndexOriginal < model.TokenWalker.Index)
339 continue; // Already consumed so avoid the one at the end of the while loop
340 }
341
342 _ = model.TokenWalker.Consume();
343 }
344
345 var closeTokenIndex = model.TokenWalker.Index;
346 var closeSquareBracketToken = (CloseSquareBracketToken)model.TokenWalker.Match(SyntaxKind.CloseSquareBracketToken);
347
348 #if DEBUG
349 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = false;
350 #endif
351 }

395 public static void ParseStatementDelimiterToken(StatementDelimiterToken statementDelimiterToken, CSharpParserModel model)
396 {
397 if (model.SyntaxStack.TryPeek(out var syntax) && syntax.SyntaxKind == SyntaxKind.NamespaceStatementNode)
398 {
399 var closureCurrentCompilationUnitBuilder = model.CurrentCodeBlockBuilder;
400 ICodeBlockOwner? nextCodeBlockOwner = null;
401 TypeClauseNode? scopeReturnTypeClauseNode = null;
402
403 var namespaceStatementNode = (NamespaceStatementNode)model.SyntaxStack.Pop();
404 nextCodeBlockOwner = namespaceStatementNode;
405
406 namespaceStatementNode.SetStatementDelimiterToken(statementDelimiterToken, model);
407
408 model.Binder.OpenScope(
409 nextCodeBlockOwner,
410 scopeReturnTypeClauseNode,
411 statementDelimiterToken.TextSpan,
412 model);
413
414 model.Binder.AddNamespaceToCurrentScope(
415 namespaceStatementNode.IdentifierToken.TextSpan.GetText(),
416 model);
417
418 model.CurrentCodeBlockBuilder = new(model.CurrentCodeBlockBuilder, nextCodeBlockOwner);
419 }
420 else if (model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner is not null &&
421 !model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner.OpenBraceToken.ConstructorWasInvoked)
422 {
423 var pendingChild = model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner;
424
425 model.Binder.OpenScope(pendingChild, CSharpFacts.Types.Void.ToTypeClause(), statementDelimiterToken.TextSpan, model);
426 model.CurrentCodeBlockBuilder = new(model.CurrentCodeBlockBuilder, pendingChild);
427 pendingChild.OnBoundScopeCreatedAndSetAsCurrent(model);
428
429 model.Binder.CloseScope(statementDelimiterToken.TextSpan, model);
430
431 if (model.CurrentCodeBlockBuilder.Parent is not null)
432 model.CurrentCodeBlockBuilder = model.CurrentCodeBlockBuilder.Parent;
433
434 model.CurrentCodeBlockBuilder.InnerPendingCodeBlockOwner = null;
435 }
