45 public CompilationUnit Parse()
46 {
47 var globalCodeBlockBuilder = new CodeBlockBuilder(null, null);
48 var currentCodeBlockBuilder = globalCodeBlockBuilder;
49 var diagnosticBag = new DiagnosticBag();
50
51 var model = new CSharpParserModel(
52 Binder,
53 BinderSession,
54 new TokenWalker(Lexer.SyntaxTokenList, diagnosticBag),
55 new Stack<ISyntax>(),
56 diagnosticBag,
57 globalCodeBlockBuilder,
58 currentCodeBlockBuilder);
59
60 #if DEBUG
61 model.TokenWalker.ProtectedTokenSyntaxKindList = new() { SyntaxKind.StatementDelimiterToken, SyntaxKind.OpenBraceToken, SyntaxKind.CloseBraceToken, };
62 #endif
63
64 while (true)
65 {
66 // The last statement in this while loop is conditionally: '_ = model.TokenWalker.Consume();'.
67 // Knowing this to be the case is extremely important.
68 var token = model.TokenWalker.Current;
69
70 #if DEBUG
71 Console.WriteLine(token.SyntaxKind + "___" + token.TextSpan.GetText() + "___" + model.TokenWalker.Index);
72 #endif
73
74 switch (token.SyntaxKind)
75 {
76 case SyntaxKind.NumericLiteralToken:
77 case SyntaxKind.CharLiteralToken:
78 case SyntaxKind.StringLiteralToken:
79 case SyntaxKind.PlusToken:
80 case SyntaxKind.PlusPlusToken:
81 case SyntaxKind.MinusToken:
82 case SyntaxKind.StarToken:
83 case SyntaxKind.DollarSignToken:
84 case SyntaxKind.AtToken:
85 if (model.StatementBuilder.ChildList.Count == 0)
86 {
87 ParseOthers.StartStatement_Expression(model);
88 }
89 else
90 {
91 var expressionNode = ParseOthers.ParseExpression(model);
92 model.StatementBuilder.ChildList.Add(expressionNode);
93 }
94 break;
95 case SyntaxKind.PreprocessorDirectiveToken:
96 ParseTokens.ParsePreprocessorDirectiveToken((PreprocessorDirectiveToken)token, model);
97 break;
98 case SyntaxKind.IdentifierToken:
99 ParseTokens.ParseIdentifierToken(model);
100 break;
101 case SyntaxKind.OpenBraceToken:
102 model.StatementBuilder.FinishStatement(model);
103
104 #if DEBUG
105 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = true;
106 #endif
107
108 var openBraceToken = (OpenBraceToken)model.TokenWalker.Consume();
109
110 #if DEBUG
111 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = false;
112 #endif
113
114 ParseTokens.ParseOpenBraceToken(openBraceToken, model);
115 break;
116 case SyntaxKind.CloseBraceToken:
117 model.StatementBuilder.FinishStatement(model);
118
119 #if DEBUG
120 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = true;
121 #endif
122
123 var closeBraceToken = (CloseBraceToken)model.TokenWalker.Consume();
124
125 #if DEBUG
126 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = false;
127 #endif
128
129 ParseTokens.ParseCloseBraceToken(closeBraceToken, model);
130 break;
131 case SyntaxKind.OpenParenthesisToken:
132 if (model.StatementBuilder.ChildList.Count == 0)
133 ParseOthers.StartStatement_Expression(model);
134 else
135 ParseTokens.ParseOpenParenthesisToken(model);
136 break;
137 case SyntaxKind.CloseParenthesisToken:
138 ParseTokens.ParseCloseParenthesisToken((CloseParenthesisToken)token, model);
139 break;
140 case SyntaxKind.OpenAngleBracketToken:
141 if (model.StatementBuilder.ChildList.Count == 0)
142 ParseOthers.StartStatement_Expression(model);
143 else
144 ParseTokens.ParseOpenAngleBracketToken((OpenAngleBracketToken)token, model);
145 break;
146 case SyntaxKind.CloseAngleBracketToken:
147 ParseTokens.ParseCloseAngleBracketToken((CloseAngleBracketToken)token, model);
148 break;
149 case SyntaxKind.OpenSquareBracketToken:
150 ParseTokens.ParseOpenSquareBracketToken(model);
151 break;
152 case SyntaxKind.CloseSquareBracketToken:
153 ParseTokens.ParseCloseSquareBracketToken((CloseSquareBracketToken)token, model);
154 break;
155 case SyntaxKind.ColonToken:
156 ParseTokens.ParseColonToken(model);
157 break;
158 case SyntaxKind.MemberAccessToken:
159 ParseTokens.ParseMemberAccessToken((MemberAccessToken)token, model);
160 break;
161 case SyntaxKind.EqualsToken:
162 ParseTokens.ParseEqualsToken(model);
163 break;
164 case SyntaxKind.StatementDelimiterToken:
165 model.StatementBuilder.FinishStatement(model);
166
167 #if DEBUG
168 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = true;
169 #endif
170
171 var statementDelimiterToken = (StatementDelimiterToken)model.TokenWalker.Consume();
172
173 #if DEBUG
174 model.TokenWalker.SuppressProtectedSyntaxKindConsumption = false;
175 #endif
176
177 ParseTokens.ParseStatementDelimiterToken(statementDelimiterToken, model);
178 break;
179 case SyntaxKind.EndOfFileToken:
180 if (model.SyntaxStack.TryPeek(out var syntax) &&
181 syntax is EndOfFileToken)
182 {
183 _ = model.SyntaxStack.Pop();
184 }
185
186 if (model.SyntaxStack.TryPop(out var notUsedSyntax))
187 {
188 if (notUsedSyntax is null)
189 {
190 }
191 else if (notUsedSyntax is IExpressionNode)
192 {
193 model.CurrentCodeBlockBuilder.ChildList.Add(notUsedSyntax);
194 }
195 else if (notUsedSyntax.SyntaxKind == SyntaxKind.AmbiguousIdentifierNode)
196 {
197 var ambiguousIdentifierNode = (AmbiguousIdentifierNode)notUsedSyntax;
198 model.CurrentCodeBlockBuilder.ChildList.Add(notUsedSyntax);
199 model.DiagnosticBag.ReportUndefinedTypeOrNamespace(
200 ambiguousIdentifierNode.IdentifierToken.TextSpan,
201 ambiguousIdentifierNode.IdentifierToken.TextSpan.GetText());
202 }
203 }
204 else if (model.StatementBuilder.ChildList.Any())
205 {
206 foreach (var item in model.StatementBuilder.ChildList)
207 {
208 model.CurrentCodeBlockBuilder.ChildList.Add(item);
209 }
210 }
211 break;
212 default:
213 if (UtilityApi.IsContextualKeywordSyntaxKind(token.SyntaxKind))
214 ParseTokens.ParseKeywordContextualToken(model);
215 else if (UtilityApi.IsKeywordSyntaxKind(token.SyntaxKind))
216 ParseTokens.ParseKeywordToken(model);
217 break;
218 }
219
220 if (token.SyntaxKind == SyntaxKind.EndOfFileToken)
221 {
222 if (model.CurrentCodeBlockBuilder.ParseChildScopeQueue.TryDequeue(out var deferredChildScope))
223 deferredChildScope.PrepareMainParserLoop(model.TokenWalker.Index, model);
224 else
225 break;
226 }
227
228 if (model.TokenWalker.ConsumeCounter == 0)
229 {
230 // This means either:
231 // - None of the methods for syntax could make sense of the token, so they didn't consume it.
232 // - For whatever reason the method that handled the syntax made sense of the token, but never consumed it.
233 // - The token was consumed, then for some reason a backtrack occurred.
234 //
235 // To avoid an infinite loop, this will ensure at least 1 token is consumed each iteration of the while loop.
236 //
237 // (and that the token index increased by at least 1 from the previous loop; this is implicitly what is implied).
238 _ = model.TokenWalker.Consume();
239 }
240 else if (model.TokenWalker.ConsumeCounter < 0)
241 {
242 // This means that a syntax invoked 'model.TokenWalker.Backtrack()'.
243 // Without invoking an equal amount of 'model.TokenWalker.Consume()' to avoid an infinite loop.
244 throw new LuthetusTextEditorException($"model.TokenWalker.ConsumeCounter:{model.TokenWalker.ConsumeCounter} < 0");
245 }
246
247 model.TokenWalker.ConsumeCounterReset();
248 }
249
250 if (model.CurrentCodeBlockBuilder.Parent is not null)
251 {
252 // The current token here would be the EOF token.
253 Binder.CloseScope(model.TokenWalker.Current.TextSpan, model);
254 }
255
256 DiagnosticsList = DiagnosticsList.AddRange(model.DiagnosticBag.ToImmutableArray());
257
258 var topLevelStatementsCodeBlock = model.CurrentCodeBlockBuilder.Build(
259 DiagnosticsList
260 .Union(Binder.DiagnosticsList)
261 .Union(Lexer.DiagnosticList)
262 .ToImmutableArray());
263
264 Binder.FinalizeBinderSession(BinderSession);
265 return new CompilationUnit(
266 topLevelStatementsCodeBlock,
267 Lexer,
268 this,
269 Binder);
270 }
271}

275//{
276 // In regards to whether to '_ = model.TokenWalker.Consume();'
277 // at the start of the while loop, or at the end:
278 //
279 // The end is better because:
280 //
281 // - Anytime one wants to check the previous token,
282 // with the 'Consume()' at the end of the loop then one can use 'model.TokenWalker.Previous'.
283 // But, if 'Consume()' is at the start of the loop then one has to use 'model.TokenWalker.Peek(-2)' and that feels very janky.
284 //
285 // - As well, helps ensure a "handle-syntax" does not 'Consume()' someone elses syntax.
286 // If an individual "syntax" cannot make sense of the current token,
287 // then it can just return to the main loop and the main loop will deal with it.
288 //
289 // On a different note:
290 //
291 // 'should return to the main loop'
292 // - The end of every statement
293 // - The start of a code block
294 // - The end of a code block
295 //
296 // The goal with every iteration of the main loop,
297 // is to narrow down the possible syntax that is occuring.
298 // Then to invoke some other method that will handle that syntax
299 // entirely, and then return to the main loop for the next syntax to start being parsed.
300 //
301 // For example, multiple consecutive keywords might be an ambiguous syntax.
302 // Therefore, the main loop should push these keywords onto the 'StatementStack'.
303 // Then, eventually a disambiguating token will be encountered.
304 // At this point, the corresponding method to handle the syntax can be invoked,
305 // and the method can pop off the 'StatementStack' to see the "related" tokens.
306 //
307 // There are two loops:
308 // - Statement Loop
309 // - Expression Loop
310 //
311 // Statement loop exists in CSharpParser.cs (this file).
312 // Expression loop exists in ParseOthers.cs (a relative path from this file: './Internals/ParseOthers.cs').
313 //
314 // The parser should gracefully handle syntax that does not compile.
315 // i.e.: if a syntactical error can be recovered from, then
316 // the parser is expected to recover.
317 //
318 // For example, "(;"
319 // For the sake of the example, this is presumed to start a ParenthesizedExpressionNode.
320 // It goes on to see a StatementDelimiterToken before it sees a CloseParenthesisToken.
321 // Instead of cascading syntactical errors to the text that comes after this (imagine this is part of a source code file)
322 // the ParenthesizedExpressionNode should report a diagnostic error,
323 // but then return back to the main loop.
324 // The main loop will see a StatementDelimiterToken, and then a new statement will begin
325 // as if nothing went wrong. This allows the rest of the text file to parse with the "same result".
326 //
327 // The parsing is at times done 'breadth first'.
328 // Because, a class definition might be:
329 //
330 // ````public class MyClass
331 // ````{
332 // ```` public MyClass(string firstName)
333 // ```` {
334 // ```` FirstName = firstName;
335 // ```` }
336 // ````
337 // ```` public string FirstName { get; set; }
338 // ````}
339 //
340 // In this scenario the 'FirstName' property is being referenced from within the class's constructor.
341 // Yet, the constructor is defined above the property it references.
342 //
343 // Instead of parsing 'depth first', in this scenario the index of the token that
344 // starts the code block body of the constructor definition is remembered,
345 // then the parser skips until it finds the matching close brace
346 // (by counting the open brace and close braces until it evens out).
347 //
348 // Once the class definition is finished being parsed, then the deferred parsing will start.
349 // This will modify the 'TokenWalker' such that it now is pointing at the previously "remembered"
350 // token that starts the code block of the constructor definition.
351 //
352 // As well, the 'TokenWalker' is told what index to return to once finished parsing the child scope.
353 // In this case the 'TokenWalker' would return to the index that represents the 'CloseBraceToken'
354 // of the class definition.
355 //
356 // If there were many child scopes that needed to be 'defer parsed' then the same
357 // code will be hit as was initially, and each time a child scope is 'defer parsed'
358 // it is removed from the list of child scopes that need to be 'defer parsed'.
359 //
360 // Regarding StatementDelimiterToken, OpenBraceToken, and CloseBraceToken:
361 // - These syntax hold a greater meaning than the "individual syntax".
362 // For example: "1;".
363 // In the example '1' is a NumericLiteralToken.
364 // Furthermore, a NumericLiteralToken can be parsed as a 'LiteralExpressionNode'.
365 // But, the ending StatementDelimiterToken does not belong to the 'LiteralExpressionNode'.
366 // It is to be "provided" to the main statement loop, to be processed there.
367 //
368 // In short, syntax shouldn't consume tokens that don't belong to them.
369 // And it is being stated that 'StatementDelimiterToken, OpenBraceToken, and CloseBraceToken'
370 // in general do not belong to a syntax.
371 //
372 // Some exceptions to this would be a property definition "public string FirstName { get; set; }".
373 // Here, the OpenBraceToken and CloseBraceToken are not being used in the same sense
374 // as they are when defining scope.
375 //
376 // So, the property definition should consume those OpenBraceToken and CloseBraceToken.
377 //
378 // However, consider this example:
379 //
380 // ````public string FirstName
381 // ````{
382 // ```` get
383 // ```` {
384 // ```` return _firstName;
385 // ```` }
386 // ```` set => _firstName = value;
387 // ````}
388 //
389 // Here, the property's "getter" is defining a code block.
390 // So, the brace tokens that deliminate the "getter", ought to
391 // "visit" the main loop for a moment. Then it can continue on with the property syntax.
392//}
