183 private void __SplitIntoTwoPartitions(int partitionIndex)
184 {
185 var originalPartition = _partitionList[partitionIndex];
186
187 var firstUnevenSplit = PartitionSize / 2 + PartitionSize % 2;
188 var secondUnevenSplit = PartitionSize / 2;
189
190 // Validate multi-byte characters go on same partition (i.e.: '\r\n')
191 {
192 // firstUnevenSplit is a count so -1 to make it an index
193 if (originalPartition.RichCharacterList[firstUnevenSplit - 1].Value == '\r')
194 {
195 if (originalPartition.RichCharacterList[(firstUnevenSplit - 1) + 1].Value == '\n')
196 {
197 firstUnevenSplit += 1;
198 secondUnevenSplit -= 1;
199 }
200 }
201
202 // TODO: If the partition to split ends in '\r' and the cause for the split
203 // is to create space in order to insert a '\n',
204 // |
205 // Then this works out as a "happy accident" of sorts.
206 // This is not ideal, it should be more concrete than "oops it worked".
207 // |
208 // The reason it works is because a split won't check if the next partition
209 // has space (? source needed) and will always move the '\r' to the new partition,
210 // then return to the insert and put the '\n' immediately after.
211
212 // One of the reasons for not having a multi-byte character span multiple partitions,
213 // is that if a partition has capacity 4,096 but a count of 2,048,
214 // one cannot insert between the bytes of a multi-byte character
215 // so the first partition with only half its capacity used, would be unable to be used
216 // any further than 2,048 because it would mean writing between its multi-byte character
217 // than spans into the next partition.
218 }
219
220 // Replace old
221 {
222 var partition = new TextEditorPartition(originalPartition.RichCharacterList
223 .Skip(0)
224 .Take(firstUnevenSplit)
225 .ToList());
226
227 _partitionList = _partitionList.SetItem(
228 partitionIndex,
229 partition);
230 }
231
232 // Insert new
233 {
234 var partition = new TextEditorPartition(originalPartition.RichCharacterList
235 .Skip(firstUnevenSplit)
236 .Take(secondUnevenSplit)
237 .ToList());
238
239 _partitionList = _partitionList.Insert(
240 partitionIndex + 1,
241 partition);
242 }

340
