168 public Task UnsafeCreateDirectoryAsync(
169 string absolutePathString,
170 CancellationToken cancellationToken = default)
171 {
172 var existingFile = _inMemoryFileSystemProvider._files.FirstOrDefault(f =>
173 f.AbsolutePath.Value == absolutePathString &&
174 f.IsDirectory);
175
176 if (existingFile is not null)
177 return Task.CompletedTask;
178
179 var absolutePath = _environmentProvider.AbsolutePathFactory(absolutePathString, true);
180
181 var outDirectory = new InMemoryFile(
182 string.Empty,
183 absolutePath,
184 DateTime.UtcNow,
185 true);
186
187 _inMemoryFileSystemProvider._files.Add(outDirectory);
188
189 _environmentProvider.DeletionPermittedRegister(new SimplePath(
190 absolutePathString,
191 IS_DIRECTORY_RESPONSE));

193 return Task.CompletedTask;

196 public async Task UnsafeDeleteAsync(
197 string absolutePathString,
198 bool recursive,
199 CancellationToken cancellationToken = default)
200 {
201 _environmentProvider.AssertDeletionPermitted(absolutePathString, IS_DIRECTORY_RESPONSE);
202
203 var indexOfExistingFile = _inMemoryFileSystemProvider._files.FindIndex(f =>
204 f.AbsolutePath.Value == absolutePathString &&
205 f.IsDirectory);
206
207 if (indexOfExistingFile == -1)
208 return;
209
210 var childFileList = _inMemoryFileSystemProvider._files.Where(imf =>
211 imf.AbsolutePath.Value.StartsWith(absolutePathString) &&
212 imf.AbsolutePath.Value != absolutePathString)
213 .ToArray();
214
215 foreach (var child in childFileList)
216 {
217 if (child.IsDirectory)
218 {
219 await _inMemoryFileSystemProvider._directory.UnsafeDeleteAsync(
220 child.AbsolutePath.Value,
221 false,
222 cancellationToken)
223 .ConfigureAwait(false);
224 }
225 else
226 {
227 await _inMemoryFileSystemProvider._file.UnsafeDeleteAsync(
228 child.AbsolutePath.Value,
229 cancellationToken)
230 .ConfigureAwait(false);
231 }
232 }
233
234 _inMemoryFileSystemProvider._files.RemoveAt(indexOfExistingFile);
235 }

237 public async Task UnsafeCopyAsync(
238 string sourceAbsolutePathString,
239 string destinationAbsolutePathString,
240 CancellationToken cancellationToken = default)
241 {
242 var indexOfExistingFile = _inMemoryFileSystemProvider._files.FindIndex(f =>
243 f.AbsolutePath.Value == sourceAbsolutePathString &&
244 f.IsDirectory);
245
246 if (indexOfExistingFile == -1)
247 return;
248
249 var sourceAbsolutePath = _environmentProvider.AbsolutePathFactory(sourceAbsolutePathString, true);
250
251 var childDirectories = (await GetDirectoriesAsync(sourceAbsolutePathString, cancellationToken).ConfigureAwait(false))
252 .Select(x => _environmentProvider.AbsolutePathFactory(x, true))
253 .ToArray();
254
255 var childFiles = (await GetFilesAsync(sourceAbsolutePathString, cancellationToken).ConfigureAwait(false))
256 .Select(x => _environmentProvider.AbsolutePathFactory(x, false))
257 .ToArray();
258
259 var children = childDirectories.Union(childFiles);
260
261 var destinationExists = await UnsafeExistsAsync(destinationAbsolutePathString, cancellationToken).ConfigureAwait(false);
262
263 if (destinationExists)
264 {
265 _environmentProvider.AssertDeletionPermitted(destinationAbsolutePathString, IS_DIRECTORY_RESPONSE);
266 }
267 else
268 {
269 var destinationAbsolutePath = _environmentProvider.AbsolutePathFactory(
270 destinationAbsolutePathString,
271 true);
272
273 var destinationFile = new InMemoryFile(
274 string.Empty,
275 destinationAbsolutePath,
276 DateTime.UtcNow,
277 true);
278
279 _inMemoryFileSystemProvider._files.Add(destinationFile);
280 }
281
282 foreach (var child in children)
283 {
284 var innerDestinationPath = _environmentProvider.JoinPaths(
285 destinationAbsolutePathString,
286 sourceAbsolutePath.NameWithExtension);
287
288 var destinationChild = _environmentProvider.JoinPaths(
289 innerDestinationPath,
290 child.NameWithExtension);
291
292 if (child.IsDirectory)
293 {
294 await _inMemoryFileSystemProvider._directory.UnsafeCopyAsync(
295 child.Value,
296 innerDestinationPath,
297 cancellationToken)
298 .ConfigureAwait(false);
299 }
300 else
301 {
302 await _inMemoryFileSystemProvider._file.UnsafeCopyAsync(
303 child.Value,
304 destinationChild,
305 cancellationToken)
306 .ConfigureAwait(false);
307 }
308 }

335 public Task<string[]> UnsafeGetDirectoriesAsync(
336 string absolutePathString,
337 CancellationToken cancellationToken = default)
338 {
339 var existingFile = _inMemoryFileSystemProvider._files.FirstOrDefault(f =>
340 f.AbsolutePath.Value == absolutePathString &&
341 f.IsDirectory);
342
343 if (existingFile is null)
344 return Task.FromResult(Array.Empty<string>());
345
346 var childrenFromAllGenerationsList = _inMemoryFileSystemProvider._files.Where(
347 f => f.AbsolutePath.Value.StartsWith(absolutePathString) &&
348 f.AbsolutePath.Value != absolutePathString)
349 .ToArray();
350
351 var directChildren = childrenFromAllGenerationsList.Where(
352 f =>
353 {
354 var withoutParentPrefix = new string(
355 f.AbsolutePath.Value
356 .Skip(absolutePathString.Length)
357 .ToArray());
358
359 return withoutParentPrefix.EndsWith("/") &&
360 withoutParentPrefix.Count(x => x == '/') == 1;
361 })
362 .Select(f => f.AbsolutePath.Value)
363 .ToArray();
364
365 return Task.FromResult(directChildren);
366 }
