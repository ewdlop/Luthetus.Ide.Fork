64 public override async Task LoadChildListAsync()
65 {
66 // Codebehinds: Children need to
67 // - new instance
68 // - try map to old instance
69 // - opportunity for child to do something (like take siblings as their children)
70
71 try
72 {
73 var previousChildren = new List<TreeViewNoType>(ChildList);
74 var newChildList = new List<TreeViewNoType>();
75
76 // new instance
77 if (Item.AbsolutePath.IsDirectory)
78 {
79 newChildList = await TreeViewHelperNamespacePathDirectory.LoadChildrenAsync(this).ConfigureAwait(false);
80 }
81 else
82 {
83 switch (Item.AbsolutePath.ExtensionNoPeriod)
84 {
85 case ExtensionNoPeriodFacts.DOT_NET_SOLUTION:
86 return;
87 case ExtensionNoPeriodFacts.C_SHARP_PROJECT:
88 newChildList = await TreeViewHelperCSharpProject.LoadChildrenAsync(this).ConfigureAwait(false);
89 break;
90 case ExtensionNoPeriodFacts.RAZOR_MARKUP:
91 newChildList = await TreeViewHelperRazorMarkup.LoadChildrenAsync(this).ConfigureAwait(false);
92 break;
93 }
94 }
95
96 // try map to old instance
97 ChildList = newChildList;
98 LinkChildren(previousChildren, ChildList);
99
100 // opportunity for child to do something (like take siblings as their children)
101 {
102 var shouldPermitChildToTakeSiblingsAsChildren = false;
103
104 if (Item.AbsolutePath.IsDirectory)
105 {
106 shouldPermitChildToTakeSiblingsAsChildren = true;
107 }
108 else
109 {
110 switch (Item.AbsolutePath.ExtensionNoPeriod)
111 {
112 case ExtensionNoPeriodFacts.C_SHARP_PROJECT:
113 shouldPermitChildToTakeSiblingsAsChildren = true;
114 break;
115 }
116 }
117
118 if (shouldPermitChildToTakeSiblingsAsChildren)
119 {
120 // Codebehind logic
121 var copyOfChildrenToFindRelatedFiles = new List<TreeViewNoType>(newChildList);
122
123 // Note that this loops over the original, and passes the copy
124 foreach (var child in newChildList)
125 {
126 child.RemoveRelatedFilesFromParent(copyOfChildrenToFindRelatedFiles);
127 }
128
129 // The parent directory gets what is left over after the
130 // children take their respective 'code behinds'
131 newChildList = copyOfChildrenToFindRelatedFiles;
132
133 // This time, 'LinkChildren(...)' is invoked
134 // in order to wire up the 'TreeViewNoType.IndexAmongSiblings'.
135 //
136 // This index is used by the keyboard events to move
137 // throughout the tree view.
138 ChildList = newChildList;
139 LinkChildren(ChildList, ChildList);
140 }
141 }
142 }
143 catch (Exception exception)
144 {
145 ChildList = new List<TreeViewNoType>
146 {
147 new TreeViewException(exception, false, false, CommonComponentRenderers)
148 {
149 Parent = this,
150 IndexAmongSiblings = 0,
151 }
152 };
153 }
154
155 TreeViewChangedKey = Key<TreeViewChanged>.NewKey();
156 }
