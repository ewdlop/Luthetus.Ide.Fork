292 {

294 IExpressionNode? previousDelimiterExpressionNode = null;
295

306 model.ExpressionList.RemoveAt(i);
307
308 if (delimiterExpressionTuple.ExpressionNode is null)
309 break; // This implies to forcibly return back to the statement while loop.

312 if (Object.ReferenceEquals(triggeredDelimiterTuple.ExpressionNode, delimiterExpressionTuple.ExpressionNode) &&
313 triggeredDelimiterTuple.DelimiterSyntaxKind != delimiterExpressionTuple.DelimiterSyntaxKind)
314 {
315 continue; // This implies that the triggered syntax kind was not the first syntax kind found for the given 'triggeredDelimiterTuple.ExpressionNode'.
