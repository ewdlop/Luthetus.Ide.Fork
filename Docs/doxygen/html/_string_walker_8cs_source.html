266 public (TextEditorTextSpan textSpan, string value) ReadWordTuple(ImmutableArray<char>? additionalCharactersToBreakOnList = null)

269
270 // The wordBuilder is appended to everytime a character is consumed.

275

282 }
283

285 {
286 // This is the start of a word as opposed to the continuation of a word
287 wordBuilderStartingIndexInclusive = PositionIndex;
288 }
289
290 wordBuilder.Append(CurrentCharacter);
291
292 _ = ReadCharacter();
293 }
294

296 wordBuilderStartingIndexInclusive,

298 0,
299 ResourceUri,
