25 public async Task CreateDirectoryAsync(
26 string absolutePathString,
27 CancellationToken cancellationToken = default)
28 {
29 // This method will add the path to IEnvironmentProvider.DeletionPermittedPaths
30 // and therefore, if the directory already exists, then return early.
31 //
32 // An example of the concern is that someone tries to create the
33 // root directory. It already existed, so the 'create' part did nothing.
34 // But now they're allowed to delete the root directory.
35 // (note: there are double checks in place such that this described
36 // situation couldn't happen regardless)
37 if (await ExistsAsync(absolutePathString, cancellationToken).ConfigureAwait(false))
38 return;
39
40 Directory.CreateDirectory(absolutePathString);
41
42 _environmentProvider.DeletionPermittedRegister(
43 new SimplePath(absolutePathString, IS_DIRECTORY_RESPONSE));
44 }
45
46 public Task DeleteAsync(
47 string absolutePathString,
48 bool recursive,
49 CancellationToken cancellationToken = default)
50 {
51 try
52 {
53 _environmentProvider.AssertDeletionPermitted(absolutePathString, IS_DIRECTORY_RESPONSE);
54 Directory.Delete(absolutePathString, recursive);
55 }
56 catch (Exception exception)

58 NotifyUserOfException(exception);
59 throw;
60 }
61
62 return Task.CompletedTask;
63 }

89 public async Task MoveAsync(
90 string sourceAbsolutePathString,

92 CancellationToken cancellationToken = default)
93 {
94 try
95 {
96 _environmentProvider.AssertDeletionPermitted(sourceAbsolutePathString, IS_DIRECTORY_RESPONSE);

98 if (await ExistsAsync(destinationAbsolutePathString).ConfigureAwait(false))
99 _environmentProvider.AssertDeletionPermitted(destinationAbsolutePathString, IS_DIRECTORY_RESPONSE);

101 Directory.Move(

103 destinationAbsolutePathString);
104
105 _environmentProvider.DeletionPermittedRegister(

112 }
