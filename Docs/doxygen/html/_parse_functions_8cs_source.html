119 public static FunctionArgumentsListingNode HandleFunctionArguments(CSharpParserModel model)
120 {
121 var openParenthesisToken = (OpenParenthesisToken)model.TokenWalker.Consume();
122 var functionArgumentEntryNodeList = new List<FunctionArgumentEntryNode>();
123 var openParenthesisCount = 1;
124 var corruptState = false;
125
126 while (!model.TokenWalker.IsEof)
127 {
128 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenParenthesisToken)
129 {
130 openParenthesisCount++;
131 }
132 else if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CloseParenthesisToken)
133 {
134 openParenthesisCount--;
135
136 if (openParenthesisCount == 0)
137 {
138 break;
139 }
140 }
141 else if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.OpenBraceToken)
142 {
143 break;
144 }
145 else if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.EqualsToken &&
146 model.TokenWalker.Next.SyntaxKind == SyntaxKind.CloseAngleBracketToken)
147 {
148 break;
149 }
150 else if (!corruptState)
151 {
152 var tokenIndexOriginal = model.TokenWalker.Index;
153 var successTypeClauseNode = ParseOthers.TryParseExpression(SyntaxKind.TypeClauseNode, model, out var typeClauseNode);
154 var successName = false;
155
156 if (successTypeClauseNode)
157 {
158 // 'TypeClauseNode' or 'VariableDeclarationNode'
159 var successNameableToken = false;
160
161 if (UtilityApi.IsConvertibleToIdentifierToken(model.TokenWalker.Current.SyntaxKind))
162 {
163 var identifierToken = UtilityApi.ConvertToIdentifierToken(model.TokenWalker.Consume(), model);
164 successNameableToken = true;
165
166 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.EqualsToken)
167 {
168 // Optional
169 }
170
171 var variableDeclarationNode = new VariableDeclarationNode(
172 (TypeClauseNode)typeClauseNode,
173 identifierToken,
174 VariableKind.Local,
175 false);
176
177 var functionArgumentEntryNode = new FunctionArgumentEntryNode(
178 variableDeclarationNode,
179 optionalCompileTimeConstantToken: null,
180 isOptional: false,
181 hasParamsKeyword: false,
182 hasOutKeyword: false,
183 hasInKeyword: false,
184 hasRefKeyword: false);
185
186 functionArgumentEntryNodeList.Add(functionArgumentEntryNode);
187
188 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CommaToken)
189 _ = model.TokenWalker.Consume();
190
191 if (tokenIndexOriginal < model.TokenWalker.Index)
192 continue; // Already consumed so avoid the one at the end of the while loop
193 }
194
195 if (!successNameableToken)
196 corruptState = true;
197 }
198 else
199 {
200 corruptState = true;
201 }
202 }
203
204 _ = model.TokenWalker.Consume();
205 }
206
207 var closeParenthesisToken = default(CloseParenthesisToken);
208
209 if (model.TokenWalker.Current.SyntaxKind == SyntaxKind.CloseParenthesisToken)
210 closeParenthesisToken = (CloseParenthesisToken)model.TokenWalker.Consume();
211
212 return new FunctionArgumentsListingNode(
213 openParenthesisToken,
214 functionArgumentEntryNodeList,
215 closeParenthesisToken);
216 }
