89 private async Task PassAlongSizeIfNoActiveTab()
90 {
91 var panelState = PanelStateWrap.Value;
92 var panelGroup = panelState.PanelGroupList.FirstOrDefault(x => x.Key == PanelGroupKey);
93
94 if (panelGroup is not null)
95 {
96 var activePanelTab = panelGroup.TabList.FirstOrDefault(
97 x => x.Key == panelGroup.ActiveTabKey);
98
99 var adjacentElementSizeDimensionAttribute = AdjacentElementDimensions.DimensionAttributeList.First(
100 x => x.DimensionAttributeKind == DimensionAttributeKind);
101
102 var indexOfPreviousPassAlong = adjacentElementSizeDimensionAttribute.DimensionUnitList.FindIndex(
103 x => x.Purpose == DimensionAttributeModificationPurpose);
104
105 if (activePanelTab is null && indexOfPreviousPassAlong == -1)
106 {
107 var panelGroupSizeDimensionsAttribute = panelGroup.ElementDimensions.DimensionAttributeList.First(
108 x => x.DimensionAttributeKind == DimensionAttributeKind);
109
110 var panelGroupPercentageSize = panelGroupSizeDimensionsAttribute.DimensionUnitList.First(
111 x => x.DimensionUnitKind == DimensionUnitKind.Percentage);
112
113 adjacentElementSizeDimensionAttribute.DimensionUnitList.Add(new DimensionUnit
114 {
115 Value = panelGroupPercentageSize.Value,
116 DimensionUnitKind = panelGroupPercentageSize.DimensionUnitKind,
117 DimensionOperatorKind = DimensionOperatorKind.Add,
118 Purpose = DimensionAttributeModificationPurpose
119 });
120
121 await ReRenderSelfAndAdjacentElementDimensionsFunc
122 .Invoke()
123 .ConfigureAwait(false);
124 }
125 else if (activePanelTab is not null && indexOfPreviousPassAlong != -1)
126 {
127 adjacentElementSizeDimensionAttribute.DimensionUnitList.RemoveAt(indexOfPreviousPassAlong);
128
129 await ReRenderSelfAndAdjacentElementDimensionsFunc
130 .Invoke()
131 .ConfigureAwait(false);
132 }
133 }
134 }
