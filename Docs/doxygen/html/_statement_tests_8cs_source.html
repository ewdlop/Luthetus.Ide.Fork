434 private void WriteChildrenIndented(ISyntaxNode node, string name = "node")
435 {
436 Console.WriteLine($"foreach (var child in {name}.GetChildList())");
437 foreach (var child in node.GetChildList())
438 {
439 Console.WriteLine("\t" + child.SyntaxKind);
440 }
441 Console.WriteLine();
442 }
443
444 private void WriteChildrenIndentedRecursive(ISyntaxNode node, string name = "node", int indentation = 0)
445 {
446 var indentationStringBuilder = new StringBuilder();
447 for (int i = 0; i < indentation; i++)
448 indentationStringBuilder.Append('\t');
449
450 Console.WriteLine($"{indentationStringBuilder.ToString()}{node.SyntaxKind}");
451
452 // For the child tokens
453 indentationStringBuilder.Append('\t');
454 var childIndentation = indentationStringBuilder.ToString();
455
456 foreach (var child in node.GetChildList())
457 {
458 if (child is ISyntaxNode syntaxNode)
459 {
460 WriteChildrenIndentedRecursive(syntaxNode, "node", indentation + 1);
461 }
462 else if (child is ISyntaxToken syntaxToken)
463 {
464 Console.WriteLine($"{childIndentation}{child.SyntaxKind}__{syntaxToken.TextSpan.GetText()}");
465 }
466 }
467
468 if (indentation == 0)
469 Console.WriteLine();
470 }
