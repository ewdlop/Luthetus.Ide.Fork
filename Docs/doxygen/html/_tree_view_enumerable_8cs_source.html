110 private static Type GetGenericArgument(Type type)
111 {
112 // Type is Array
113 // short-circuit if you expect lots of arrays
114 if (type.IsArray)
115 return type.GetElementType()!;
116
117 // type is IEnumerable<T>;
118 if (type.IsGenericType && type.GetGenericTypeDefinition() == typeof(IEnumerable<>))
119 return type.GetGenericArguments()[0];
120
121 // type implements/extends IEnumerable<T>;
122 var enumType = type.GetInterfaces()
123 .Where(t => t.IsGenericType &&
124 t.GetGenericTypeDefinition() == typeof(IEnumerable<>))
125 .Select(t => t.GenericTypeArguments[0]).FirstOrDefault();
126 return enumType ?? type;
127 }
