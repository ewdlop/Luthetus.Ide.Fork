59 private string CommandUiInputBinding
60 {
61 get => _command;
62 set
63 {
64 _command = value;
65
66 // If not browsing history, cache what the user is typing
67 //
68 // This means that while the '_cachedCommand' protects against
69 // accidental pressing of the ArrowUp key,
70 //
71 // If one is modifying a command that was populated into the
72 // input element by browsing history,
73 // then until the enter key is pressed to stop browsing history,
74 // nothing will be cached and an accidental ArrowUp here could lose information.
75 //
76 // But, the priority for protecting against lost information by accidental ArrowUp
77 // is on an original newly being typed command, not the history.
78 if (_terminalCommandRequestHistory is null)
79 _cachedCommand = value;
80 }

127 private void HandleOnKeyDown(KeyboardEventArgs keyboardEventArgs)
128 {
129 if (keyboardEventArgs.Code == "Enter")
130 {
131 var commandLocal = _command;
132 _terminalCommandRequestHistory = null;
133 _indexHistory = 0;
134
135 Terminal.EnqueueCommand(new TerminalCommandRequest(
136 commandText: commandLocal,
137 workingDirectory: null));
138 }
139 else if (keyboardEventArgs.Key == "ArrowUp")
140 {
141 if (_terminalCommandRequestHistory is null)
142 {
143 _terminalCommandRequestHistory = Terminal.TerminalInteractive.GetTerminalCommandRequestHistory();
144 _indexHistory = 0;
145
146 var commandAtIndex = GetTerminalCommandRequestAtIndexHistory(
147 _indexHistory,
148 _terminalCommandRequestHistory);
149
150 if (commandAtIndex is not null)
151 _command = commandAtIndex.CommandText;
152 }
153 else
154 {
155 if (_indexHistory < _terminalCommandRequestHistory.Count - 1)
156 _indexHistory++;
157
158 var commandAtIndex = GetTerminalCommandRequestAtIndexHistory(
159 _indexHistory,
160 _terminalCommandRequestHistory);
161
162 if (commandAtIndex is not null)
163 _command = commandAtIndex.CommandText;
164 }
165 }
166 else if (keyboardEventArgs.Key == "ArrowDown")
167 {
168 if (_terminalCommandRequestHistory is not null)
169 {
170 if (_indexHistory == 0)
171 {
172 _command = _cachedCommand;
173 _terminalCommandRequestHistory = null;
174 }
175 else
176 {
177 if (_indexHistory < 0)
178 _indexHistory = 0;
179 else
180 _indexHistory--;
181
182 var commandAtIndex = GetTerminalCommandRequestAtIndexHistory(
183 _indexHistory,
184 _terminalCommandRequestHistory);
185
186 if (commandAtIndex is not null)
187 _command = commandAtIndex.CommandText;
188 }
189 }
190 }

198 private void OnWriteOutput()
199 {
200 _throttle.Run(_ =>
201 {
202 TextEditorService.PostUnique(
203 nameof(TerminalOutput),
204 editContext =>
205 {
206 var formatter = Terminal.TerminalOutput.OutputFormatterList.FirstOrDefault(
207 x => x.Name == nameof(TerminalOutputFormatterExpand));
208
209 if (formatter is not TerminalOutputFormatterExpand terminalOutputFormatterExpand)
210 return Task.CompletedTask;
211
212 var modelModifier = editContext.GetModelModifier(terminalOutputFormatterExpand.TextEditorModelResourceUri);
213 var viewModelModifier = editContext.GetViewModelModifier(terminalOutputFormatterExpand.TextEditorViewModelKey);
214 var cursorModifierBag = editContext.GetCursorModifierBag(viewModelModifier?.ViewModel);
215 var primaryCursorModifier = editContext.GetPrimaryCursorModifier(cursorModifierBag);
216
217 if (modelModifier is null || viewModelModifier is null || cursorModifierBag is null || primaryCursorModifier is null)
218 return Task.CompletedTask;
219
220 var localTerminal = Terminal;
221
222 var outputFormatted = (TerminalOutputFormattedTextEditor)localTerminal.TerminalOutput
223 .GetOutputFormatted(nameof(TerminalOutputFormatterExpand));
224
225 modelModifier.SetContent(outputFormatted.Text);
226
227 var lineIndexOriginal = primaryCursorModifier.LineIndex;
228 var columnIndexOriginal = primaryCursorModifier.ColumnIndex;
229
230 // Move Cursor, try to preserve the current cursor position.
231 {
232 if (primaryCursorModifier.LineIndex > modelModifier.LineCount - 1)
233 primaryCursorModifier.LineIndex = modelModifier.LineCount - 1;
234
235 var lineInformation = modelModifier.GetLineInformation(primaryCursorModifier.LineIndex);
236
237 if (primaryCursorModifier.ColumnIndex > lineInformation.LastValidColumnIndex)
238 primaryCursorModifier.SetColumnIndexAndPreferred(lineInformation.LastValidColumnIndex);
239
240 if (lineIndexOriginal != primaryCursorModifier.LineIndex ||
241 columnIndexOriginal != primaryCursorModifier.ColumnIndex)
242 {
243 viewModelModifier.ViewModel.UnsafeState.ShouldRevealCursor = true;
244 }
245 }
246
247 var compilerServiceResource = modelModifier.CompilerService.GetCompilerServiceResourceFor(
248 terminalOutputFormatterExpand.TextEditorModelResourceUri);
249
250 if (compilerServiceResource is TerminalResource terminalResource)
251 {
252 terminalResource.ManualDecorationTextSpanList.Clear();
253 terminalResource.ManualDecorationTextSpanList.AddRange(
254 outputFormatted.SymbolList.Select(x => x.TextSpan));
255
256 terminalResource.ManualSymbolList.Clear();
257 terminalResource.ManualSymbolList.AddRange(outputFormatted.SymbolList);
258
259 editContext.TextEditorService.ModelApi.ApplyDecorationRange(
260 editContext,
261 modelModifier,
262 terminalResource.GetTokenTextSpans());
263
264 editContext.TextEditorService.ModelApi.ApplyDecorationRange(
265 editContext,
266 modelModifier,
267 outputFormatted.TextSpanList);
268 }
269
270 return Task.CompletedTask;
271 });
272 return Task.CompletedTask;
273 });
274 }
