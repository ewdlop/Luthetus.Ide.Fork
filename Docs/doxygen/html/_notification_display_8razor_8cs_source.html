101 {
102 var styleBuilder = new StringBuilder();
103
104 var widthInFontWidthInvariantCulture = WIDTH_IN_FONT_WIDTH.ToCssValue();
105 //var widthInPixelsInvariantCulture = WIDTH_IN_PIXELS.ToCssValue();

109 styleBuilder.Append($" width: {widthInFontWidthInvariantCulture}ch;");
110 styleBuilder.Append($" height: {heightInFontHeightInvariantCulture}em;");
111
112 var rightOffsetInPixelsInvariantCulture = RIGHT_OFFSET_IN_PIXELS.ToCssValue();
113
114 styleBuilder.Append($" right: {rightOffsetInPixelsInvariantCulture}px;");
115
116 var bottomOffsetDueToHeightInFontHeight = HEIGHT_IN_FONT_HEIGHT * Index;
117 //var bottomOffsetDueToHeight = HEIGHT_IN_FONT_HEIGHT * Index;
118
119 var bottomOffsetDueToBottomOffsetInPixels = BOTTOM_OFFSET_IN_PIXELS * (1 + Index);
120 //var bottomOffsetDueToBottomOffset = BOTTOM_OFFSET_IN_PIXELS * (1 + Index);
121
122 // The bottom panel has:
123 // height="calc(var(--luth_ide_panel-tabs-font-size) + var(--luth_ide_panel-tabs-margin) + var(--luth_ide_panel-tabs-bug-are-not-aligning-need-to-fix-todo))"
124 //
125 // The height for the bottom panel might need to change? This is pretty gross.
126 // I want to add a notifications counter in the bottom panel on the horizontal end.
127 // So I'm putting this here to align things.
128 //
129 // And changing:
130 // 'styleBuilder.Append($" bottom: {totalBottomOffsetInvariantCulture}px;");'
131 //
132 // to include the bottom panel height in the calculation.
133 var panelBottomHeight = "0.85em + 4px + 0.7em";
134
135 var bottomOffsetDueToHeightInFontHeightInvariantCulture = bottomOffsetDueToHeightInFontHeight.ToCssValue();
136 var bottomOffsetDueToBottomOffsetInPixelsInvariantCulture = bottomOffsetDueToBottomOffsetInPixels.ToCssValue();
137
138 //var totalBottomOffsetInvariantCulture = totalBottomOffset.ToCssValue();
139
140 styleBuilder.Append($" bottom: calc({bottomOffsetDueToHeightInFontHeightInvariantCulture}em + {bottomOffsetDueToBottomOffsetInPixelsInvariantCulture}px + {panelBottomHeight}) ;");
141
142 return styleBuilder.ToString();
143 }
