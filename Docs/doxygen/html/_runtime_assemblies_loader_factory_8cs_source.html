266 string name,
267 Type[] genericParameters)

269 var typeTextSpan = TextEditorTextSpan.FabricateTextSpan(name);
270 var typeIdentifierToken = new IdentifierToken(typeTextSpan);
271
272 GenericParametersListingNode? genericParametersListingNode = null;
273
274 if (genericParameters.Any())

276 // Generic Parameters Open Angle Bracket
277 var openAngleBracketTextSpan = TextEditorTextSpan.FabricateTextSpan("<");
278 var openAngleBracketToken = new OpenAngleBracketToken(openAngleBracketTextSpan);
279
280 // Generic Parameter Entries
281 var genericParameterEntryNodeList = new List<GenericParameterEntryNode>();
282
283 foreach (var genericParameter in genericParameters)
284 {
285 var genericParameterEntryNode = new GenericParameterEntryNode(
286 GetTypeClauseNode(
287 genericParameter.Name,
288 genericParameter.GenericTypeArguments));
289
290 genericParameterEntryNodeList.Add(genericParameterEntryNode);
291 }
292
293 // Generic Parameters Close Parenthesis
294 var closeAngleBracketTextSpan = TextEditorTextSpan.FabricateTextSpan(">");
295 var closeAngleBracketToken = new CloseAngleBracketToken(closeAngleBracketTextSpan);
296
297 // Generic Parameters Listing
298 genericParametersListingNode = new GenericParametersListingNode(
299 openAngleBracketToken,
300 genericParameterEntryNodeList,
301 closeAngleBracketToken);
302 }
303
304 return new TypeClauseNode(
305 typeIdentifierToken,
306 null,
