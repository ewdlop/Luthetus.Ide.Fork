122 private MenuRecord GetMenuRecord()
123 {
124 var renderBatch = TextEditorViewModelDisplay._storedRenderBatchTuple.Validated;
125 if (renderBatch is null)
126 return NoResultsMenuRecord;
127
128 try
129 {
130 var cursorList = new TextEditorCursor[] { renderBatch.ViewModel.PrimaryCursor }.ToImmutableArray();
131
132 var primaryCursor = cursorList.First(x => x.IsPrimaryCursor);
133
134 if (primaryCursor.ColumnIndex > 0)
135 {
136 var word = renderBatch.Model.ReadPreviousWordOrDefault(
137 primaryCursor.LineIndex,
138 primaryCursor.ColumnIndex);
139
140 List<MenuOptionRecord> menuOptionRecordsList = new();
141
142 if (word is not null)
143 {
144 var autocompleteWordsList = AutocompleteService.GetAutocompleteOptions(word);
145
146 var autocompleteEntryList = autocompleteWordsList
147 .Select(aw => new AutocompleteEntry(aw, AutocompleteEntryKind.Word, null))
148 .ToArray();
149
150 // (2023-08-09) Looking into using an ICompilerService for autocompletion.
151 {
152 var positionIndex = renderBatch.Model.GetPositionIndex(primaryCursor);
153
154 var textSpan = new TextEditorTextSpan(
155 positionIndex,
156 positionIndex + 1,
157 0,
158 renderBatch.Model.ResourceUri,
159 // TODO: renderBatch.Model.GetAllText() probably isn't needed here. Maybe a useful optimization is to remove it somehow?
160 renderBatch.Model.GetAllText());
161
162 var compilerServiceAutocompleteEntryList = renderBatch.Model.CompilerService.GetAutocompleteEntries(
163 word,
164 textSpan);
165
166 if (compilerServiceAutocompleteEntryList.Any())
167 {
168 autocompleteEntryList = compilerServiceAutocompleteEntryList
169 .AddRange(autocompleteEntryList)
170 .ToArray();
171 }
172 }
173
174 menuOptionRecordsList = autocompleteEntryList.Select(entry => new MenuOptionRecord(
175 entry.DisplayName,
176 MenuOptionKind.Other,
177 () => SelectMenuOption(() =>
178 {
179 if (entry.AutocompleteEntryKind != AutocompleteEntryKind.Snippet)
180 InsertAutocompleteMenuOption(word, entry, renderBatch.ViewModel);
181
182 return entry.SideEffectFunc?.Invoke();
183 }),
184 WidgetParameterMap: new Dictionary<string, object?>
185 {
186 {
187 nameof(AutocompleteEntry),
188 entry
189 }
190 }))
191 .ToList();
192 }
193
194 if (!menuOptionRecordsList.Any())
195 menuOptionRecordsList.Add(new MenuOptionRecord("No results", MenuOptionKind.Other));
196
197 return new MenuRecord(menuOptionRecordsList.ToImmutableArray());
198 }
199
200 return NoResultsMenuRecord;
201 }
202 // Catching 'InvalidOperationException' is for the currently occurring case: "Collection was modified; enumeration operation may not execute."
203 catch (Exception e) when (e is LuthetusTextEditorException || e is InvalidOperationException)
204 {
205 return NoResultsMenuRecord;
206 }
207 }

248 private Task InsertAutocompleteMenuOption(
249 string word,

251 TextEditorViewModel viewModel)
252 {
253 var renderBatch = TextEditorViewModelDisplay._storedRenderBatchTuple.Validated;
254 if (renderBatch is null)
255 return Task.CompletedTask;
256
257 TextEditorService.PostUnique(
258 nameof(InsertAutocompleteMenuOption),
259 editContext =>
260 {
261 var modelModifier = editContext.GetModelModifier(viewModel.ResourceUri);
262 var viewModelModifier = editContext.GetViewModelModifier(viewModel.ViewModelKey);
263 var cursorModifierBag = editContext.GetCursorModifierBag(viewModelModifier?.ViewModel);
264 var primaryCursorModifier = editContext.GetPrimaryCursorModifier(cursorModifierBag);
265
266 if (modelModifier is null || viewModelModifier is null || cursorModifierBag is null || primaryCursorModifier is null)
267 return Task.CompletedTask;
268
269 TextEditorService.ModelApi.InsertText(
270 editContext,
271 modelModifier,
272 cursorModifierBag,
273 autocompleteEntry.DisplayName.Substring(word.Length),
274 CancellationToken.None);
275
276 return renderBatch.ViewModel.FocusAsync();
277 });
278
279 return renderBatch.ViewModel.FocusAsync();
