102 private async Task HandleCommand(TerminalCommandRequest terminalCommandRequest)
103 {
104 TerminalOutput.ClearHistoryWhenExistingOutputTooLong();
105
106 var parsedCommand = await TerminalInteractive.TryHandleCommand(terminalCommandRequest);
107 ActiveTerminalCommandParsed = parsedCommand;
108
109 if (parsedCommand is null)
110 return;
111
112 var cliWrapCommand = Cli.Wrap(parsedCommand.TargetFileName);
113
114 cliWrapCommand = cliWrapCommand.WithWorkingDirectory(TerminalInteractive.WorkingDirectory);
115
116 if (!string.IsNullOrWhiteSpace(parsedCommand.Arguments))
117 cliWrapCommand = cliWrapCommand.WithArguments(parsedCommand.Arguments);
118
119 // TODO: Decide where to put invocation of 'parsedCommand.SourceTerminalCommandRequest.BeginWithFunc'...
120 // ...and invocation of 'cliWrapCommand'
121 // and invocation of 'parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc'
122 // |
123 // This comment is referring to the 'try/catch' block logic.
124 // If the 'BeginWithFunc' throws an exception should the 'cliWrapCommand' run?
125 // If the 'cliWrapCommand' throws an exception should the 'ContinueWithFunc' run?
126
127 try
128 {
129 HasExecutingProcess = true;
130
131 if (parsedCommand.SourceTerminalCommandRequest.BeginWithFunc is not null)
132 {
133 await parsedCommand.SourceTerminalCommandRequest.BeginWithFunc
134 .Invoke(parsedCommand)
135 .ConfigureAwait(false);
136 }
137
138 await cliWrapCommand
139 .Observe(_commandCancellationTokenSource.Token)
140 .ForEachAsync(HandleOutput)
141 .ConfigureAwait(false);
142 }
143 catch (Exception e)
144 {
145 // TODO: This will erroneously write 'StartedCommandEvent' out twice...
146 // ...unless a check is added to see WHEN the exception was thrown.
147 TerminalOutput.WriteOutput(
148 parsedCommand,
149 new StartedCommandEvent(-1));
150
151 TerminalOutput.WriteOutput(
152 parsedCommand,
153 new StandardErrorCommandEvent(
154 parsedCommand.SourceTerminalCommandRequest.CommandText +
155 " threw an exception" +
156 "\n"));
157
158 NotificationHelper.DispatchError("Terminal Exception", e.ToString(), _commonComponentRenderers, _dispatcher, TimeSpan.FromSeconds(14));
159 }
160 finally
161 {
162 if (parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc is not null)
163 {
164 await parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc
165 .Invoke(parsedCommand)
166 .ConfigureAwait(false);
167 }
168
169 HasExecutingProcess = false;
170 }
171 }
