357 private void InitializeOptionsApiTests(
358 out ITextEditorService textEditorService,
359 out IState<TextEditorOptionsState> textEditorOptionsState,
360 out IDispatcher dispatcher,
361 out ServiceProvider serviceProvider)
362 {
363 var backgroundTaskService = new BackgroundTaskServiceSynchronous();
364
365 var serviceCollection = new ServiceCollection()
366 .AddScoped<IJSRuntime, DoNothingJsRuntime>()
367 .AddLuthetusTextEditor(new LuthetusHostingInformation(
368 LuthetusHostingKind.UnitTestingSynchronous,
369 LuthetusPurposeKind.TextEditor,
370 backgroundTaskService))
371 .AddFluxor(options => options.ScanAssemblies(
372 typeof(LuthetusCommonConfig).Assembly,
373 typeof(LuthetusTextEditorConfig).Assembly));
374
375 serviceProvider = serviceCollection.BuildServiceProvider();
376
377 var store = serviceProvider.GetRequiredService<IStore>();
378 store.InitializeAsync().Wait();
379
380 dispatcher = serviceProvider.GetRequiredService<IDispatcher>();
381 textEditorService = serviceProvider.GetRequiredService<ITextEditorService>();
382 textEditorOptionsState = serviceProvider.GetRequiredService<IState<TextEditorOptionsState>>();
383 }
