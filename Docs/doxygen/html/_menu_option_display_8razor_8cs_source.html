132 private Task HandleOnKeyDown(KeyboardEventArgs keyboardEventArgs)
133 {
134 switch (keyboardEventArgs.Key)
135 {
136 case KeyboardKeyFacts.MovementKeys.ARROW_RIGHT:
137 case KeyboardKeyFacts.AlternateMovementKeys.ARROW_RIGHT:
138 var localSubMenu = MenuOptionRecord.SubMenu;
139 if (localSubMenu is not null)
140 return RenderDropdownAsync(localSubMenu);
141 break;
142 }
143
144 switch (keyboardEventArgs.Code)
145 {
146 case KeyboardKeyFacts.WhitespaceCodes.ENTER_CODE:
147 case KeyboardKeyFacts.WhitespaceCodes.SPACE_CODE:
148 return HandleOnClick();
149 }
150
151 return Task.CompletedTask;
152 }
