261 public static ImmutableArray<TextEditorTextSpan> FindMatches(this ITextEditorModel model, string query)
262 {
263 var text = model.GetAllText();
264 var matchedTextSpans = new List<TextEditorTextSpan>();
265
266 for (int outerI = 0; outerI < text.Length; outerI++)
267 {
268 if (outerI + query.Length <= text.Length)
269 {
270 int innerI = 0;
271 for (; innerI < query.Length; innerI++)
272 {
273 if (text[outerI + innerI] != query[innerI])
274 break;
275 }
276
277 if (innerI == query.Length)
278 {
279 // Then the entire query was matched
280 matchedTextSpans.Add(new TextEditorTextSpan(
281 outerI,
282 outerI + innerI,
283 (byte)FindOverlayDecorationKind.LongestCommonSubsequence,
284 model.ResourceUri,
285 text));
286 }
287 }
288 }
289
290 return matchedTextSpans.ToImmutableArray();
291 }

307 public static LineInformation GetLineInformation(this ITextEditorModel model, int lineIndex)
308 {
309 model.AssertLineIndex(lineIndex);

311 LineEnd GetLineEndLower(int lineIndex)
312 {
313 // Large index? Then set the index to the last index.
314 lineIndex = Math.Min(lineIndex, model.LineEndList.Count - 1);
315
316 // Small index? Then return StartOfFile.
317 if (lineIndex <= 0)
318 return new(0, 0, LineEndKind.StartOfFile);
319
320 // In-range index? Then return the previous line's line ending.
321 return model.LineEndList[lineIndex - 1];
322 }
323
324 LineEnd GetLineEndUpper(int lineIndex)
325 {
326 // Large index? Then set the index to the last index.
327 lineIndex = Math.Min(lineIndex, model.LineEndList.Count - 1);
328
329 // Small index? Then return the first LineEnd
330 if (lineIndex <= 0)
331 return model.LineEndList[0];
332
333 // In-range index? Then return the LineEnd at that index.
334 return model.LineEndList[lineIndex];
335 }
336
337 var lineEndLower = GetLineEndLower(lineIndex);
338 var lineEndUpper = GetLineEndUpper(lineIndex);
339
340 return new LineInformation(
341 lineIndex,
342 lineEndLower.EndPositionIndexExclusive,
343 lineEndUpper.EndPositionIndexExclusive,
344 lineEndLower,
345 lineEndUpper);

348 public static LineInformation GetLineInformationFromPositionIndex(this ITextEditorModel model, int positionIndex)
349 {
350 model.AssertPositionIndex(positionIndex);
351
352 int GetLineIndexFromPositionIndex()
353 {
354 // StartOfFile
355 if (model.LineEndList[0].EndPositionIndexExclusive > positionIndex)
356 return 0;
357
358 // EndOfFile
359 if (model.LineEndList[^1].EndPositionIndexExclusive <= positionIndex)
360 return model.LineEndList.Count - 1;
361
362 // In-between
363 for (var i = 1; i < model.LineEndList.Count; i++)
364 {
365 var lineEndTuple = model.LineEndList[i];
366
367 if (lineEndTuple.EndPositionIndexExclusive > positionIndex)
368 return i;
369 }
370
371 // Fallback return StartOfFile
372 return 0;
373 }
374
375 return model.GetLineInformation(GetLineIndexFromPositionIndex());

383 public static int GetColumnIndexOfCharacterWithDifferingKind(
384 this ITextEditorModel model,
385 int lineIndex,
386 int columnIndex,
387 bool moveBackwards)
388 {
389 var iterateBy = moveBackwards
390 ? -1
391 : 1;
392
393 var lineStartPositionIndex = model.GetLineInformation(lineIndex).StartPositionIndexInclusive;
394
395 if (lineIndex > model.LineEndList.Count - 1)
396 return -1;
397
398 var lastPositionIndexOnRow = model.LineEndList[lineIndex].EndPositionIndexExclusive - 1;
399 var positionIndex = model.GetPositionIndex(lineIndex, columnIndex);
400
401 if (moveBackwards)
402 {
403 if (positionIndex <= lineStartPositionIndex)
404 return -1;
405
406 positionIndex -= 1;
407 }
408
409 if (positionIndex < 0 || positionIndex >= model.RichCharacterList.Length)
410 return -1;
411
412 var startCharacterKind = CharacterKindHelper.CharToCharacterKind(model.RichCharacterList[positionIndex].Value);
413
414 while (true)
415 {
416 if (positionIndex >= model.RichCharacterList.Length ||
417 positionIndex > lastPositionIndexOnRow ||
418 positionIndex < lineStartPositionIndex)
419 {
420 return -1;
421 }
422
423 var currentCharacterKind = CharacterKindHelper.CharToCharacterKind(model.RichCharacterList[positionIndex].Value);
424
425 if (currentCharacterKind != startCharacterKind)
426 break;
427
428 positionIndex += iterateBy;
429 }
430
431 if (moveBackwards)
432 positionIndex += 1;
433
434 return positionIndex - lineStartPositionIndex;
435 }

461 public static string? ReadPreviousWordOrDefault(
462 this ITextEditorModel model,
463 int lineIndex,
464 int columnIndex,
465 bool isRecursiveCall = false)
466 {
467 var wordPositionIndexEndExclusive = model.GetPositionIndex(lineIndex, columnIndex);
468 var wordCharacterKind = model.GetCharacterKind(wordPositionIndexEndExclusive - 1);
469
470 if (wordCharacterKind == CharacterKind.Punctuation && !isRecursiveCall)
471 {
472 // If previous previous word is a punctuation character, then perhaps
473 // the user hit { 'Ctrl' + 'Space' } to trigger the autocomplete
474 // and was at a MemberAccessToken (or a period '.')
475 //
476 // So, read the word previous to the punctuation.
477
478 var anotherAttemptColumnIndex = columnIndex - 1;
479
480 if (anotherAttemptColumnIndex >= 0)
481 return model.ReadPreviousWordOrDefault(lineIndex, anotherAttemptColumnIndex, true);
482 }
483
484 if (wordCharacterKind == CharacterKind.LetterOrDigit)
485 {
486 var wordColumnIndexStartInclusive = model.GetColumnIndexOfCharacterWithDifferingKind(
487 lineIndex,
488 columnIndex,
489 true);
490
491 if (wordColumnIndexStartInclusive == -1)
492 wordColumnIndexStartInclusive = 0;
493
494 var wordLength = columnIndex - wordColumnIndexStartInclusive;
495 var wordPositionIndexStartInclusive = wordPositionIndexEndExclusive - wordLength;
496
497 return model.GetString(wordPositionIndexStartInclusive, wordLength);
498 }
499
500 return null;
501 }
