51 public void IncrementLinks()
52 {
53 var becameDisplayed = false;
54
55 lock (_linksLock)
56 {
57 Links++;
58
59 if (Links == 1)
60 {
61 // This ViewModel was not being displayed until this point.
62 //
63 // The goal is to lazily update the UI, i.e.: only when a given view model is being displayed on the UI.
64 //
65 // Therefore, presume that the UI this newly rendered view model is outdated,
66 // (perhaps the font-size was changed for eample)
67 IsFirstDisplay = true;
68 becameDisplayed = true;
69
70 // Furthermore, subscribe to the events which indicate that the UI has changed (for example font-size)
71 // so that these events are immediately handled, considering that this view model is being rendered on the UI.
72 _textEditorService.AppDimensionStateWrap.StateChanged += AppDimensionStateWrap_StateChanged;
73 }
74 else if (Links > 1)
75 {
76 // TODO: This exception is getting commented out for now because dragging a text editor
77 // tab off the text editor group display is using the same view model for the dialog
78 // as it was in the group tab.
79 //
80 //throw new LuthetusFatalException($"{nameof(DisplayTracker)} detected a {nameof(TextEditorViewModel)}" +
81 // " was being displayed in two places simultaneously." +
82 // " A {nameof(TextEditorViewModel)} can only be displayed by a single" +
83 // " {nameof(TextEditorViewModelDisplay)} at a time.");
84 }
85 }
86
87 if (becameDisplayed)
88 {
89 // TODO: If the 'AppDimensionStateWrap_StateChanged(...)' is invoking 'PostScrollAndRemeasure()' too...
90 // ... and it is believed that it is being done there to measure the UI, as the
91 // newly sized font wouldn't be guaranteed to have been rendered in time,
92 // then why is this invoked here.
93 // |
94 // The response is that the very first render of the text editor will ignore the view model,
95 // and only serves to measure the font size.
96 // |
97 // As a result when a view model becomes displayed, the text editor will already know the font-size.
98 // And it will just "be available" and this invocation is less of a re-measure, and more of a
99 // setting of what the current font measurements are on the view model being displayed.
100
101 // Tell the view model what the (already known) font-size measurements and text-editor measurements are.
102 PostScrollAndRemeasure();
103 }
