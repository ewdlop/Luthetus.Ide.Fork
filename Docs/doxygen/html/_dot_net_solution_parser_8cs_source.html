50 public CompilationUnit Parse()
51 {
52 while (true)
53 {
54 var consumedToken = _tokenWalker.Consume();
55
56 switch (consumedToken.SyntaxKind)
57 {
58 case SyntaxKind.AssociatedNameToken:
59 ParseAssociatedNameToken((AssociatedNameToken)consumedToken);
60 break;
61 case SyntaxKind.AssociatedValueToken:
62 ParseAssociatedValueToken((AssociatedValueToken)consumedToken);
63 break;
64 case SyntaxKind.OpenAssociatedGroupToken:
65 ParseOpenAssociatedGroupToken((OpenAssociatedGroupToken)consumedToken);
66 break;
67 case SyntaxKind.CloseAssociatedGroupToken:
68 ParseCloseAssociatedGroupToken((CloseAssociatedGroupToken)consumedToken);
69 break;
70 default:
71 break;
72 }
73
74 if (consumedToken.SyntaxKind == SyntaxKind.EndOfFileToken)
75 break;
76 }
77
78 var globalSectionNestedProjects = DotNetSolutionGlobal.DotNetSolutionGlobalSectionList.FirstOrDefault(x =>
79 {
80 return (x.GlobalSectionArgument?.TextSpan.GetText() ?? string.Empty) ==
81 LexSolutionFacts.GlobalSectionNestedProjects.START_TOKEN;
82 });
83
84 if (globalSectionNestedProjects is not null)
85 {
86 foreach (var associatedEntry in globalSectionNestedProjects.AssociatedEntryGroup.AssociatedEntryList)
87 {
88 switch (associatedEntry.AssociatedEntryKind)
89 {
90 case AssociatedEntryKind.Pair:
91 var pair = (AssociatedEntryPair)associatedEntry;
92
93 if (Guid.TryParse(pair.AssociatedNameToken.TextSpan.GetText(),
94 out var childProjectIdGuid))
95 {
96 if (Guid.TryParse(pair.AssociatedValueToken.TextSpan.GetText(),
97 out var solutionFolderIdGuid))
98 {
99 var nestedProjectEntry = new NestedProjectEntry(
100 childProjectIdGuid,
101 solutionFolderIdGuid);
102
103 _nestedProjectEntryList.Add(nestedProjectEntry);
104 }
105 }
106
107 break;
108 default:
109 break;
110 }
111 }
112 }
113
114 return new CompilationUnit(
115 null,
116 Lexer,
117 this,
118 null);
119 }

121 public void ParseAssociatedNameToken(AssociatedNameToken associatedNameToken)
122 {
123 if (!_hasReadHeader)
124 {
125 foreach (var wellKnownAssociatedName in LexSolutionFacts.Header.WellKnownAssociatedNamesList)
126 {
127 if (associatedNameToken.TextSpan.GetText() == wellKnownAssociatedName)
128 {
129 var associatedValueToken = (AssociatedValueToken)_tokenWalker.Match(SyntaxKind.AssociatedValueToken);
130
131 switch (wellKnownAssociatedName)
132 {
133 case LexSolutionFacts.Header.EXACT_VISUAL_STUDIO_VERSION_START_TOKEN:
134 _dotNetSolutionHeader = _dotNetSolutionHeader with
135 {
136 ExactVisualStudioVersionPair = new AssociatedEntryPair(
137 associatedNameToken, associatedValueToken),
138 };
139 break;
140 case LexSolutionFacts.Header.FORMAT_VERSION_START_TOKEN:
141 _dotNetSolutionHeader = _dotNetSolutionHeader with
142 {
143 FormatVersionPair = new AssociatedEntryPair(
144 associatedNameToken, associatedValueToken),
145 };
146 break;
147 case LexSolutionFacts.Header.HASHTAG_VISUAL_STUDIO_VERSION_START_TOKEN:
148 _dotNetSolutionHeader = _dotNetSolutionHeader with
149 {
150 HashtagVisualStudioVersionPair = new AssociatedEntryPair(
151 associatedNameToken, associatedValueToken),
152 };
153 break;
154 case LexSolutionFacts.Header.MINIMUM_VISUAL_STUDIO_VERSION_START_TOKEN:
155 _dotNetSolutionHeader = _dotNetSolutionHeader with
156 {
157 MinimumVisualStudioVersionPair = new AssociatedEntryPair(
158 associatedNameToken, associatedValueToken),
159 };
160 break;
161 }
162 }
163 }
164 }
165 else
166 {
167 var associatedValueToken = (AssociatedValueToken)_tokenWalker.Match(SyntaxKind.AssociatedValueToken);
168 var associatedEntryPair = new AssociatedEntryPair(associatedNameToken, associatedValueToken);
169
170 _associatedEntryGroupBuilderStack.Peek().AssociatedEntryList.Add(associatedEntryPair);
171 }
172 }

186 private void ParseOpenAssociatedGroupToken(OpenAssociatedGroupToken openAssociatedGroupToken)
187 {
188 // Presumption is made here, that the header only contains AssociatedEntryPairs
189 _hasReadHeader = true;
190
191 var success = _associatedEntryGroupBuilderStack.TryPeek(out var parent);
192
193 if (openAssociatedGroupToken.TextSpan.GetText() == LexSolutionFacts.Global.START_TOKEN)
194 {
195 _dotNetSolutionGlobal = _dotNetSolutionGlobal with
196 {
197 WasFound = true,
198 OpenAssociatedGroupToken = openAssociatedGroupToken
199 };
200 }
201 else if (openAssociatedGroupToken.TextSpan.GetText() == LexSolutionFacts.GlobalSection.START_TOKEN)
202 {
203 // TODO: Should this be re-written without using a closure hack?
204 var localDotNetSolutionGlobalSectionBuilder = new DotNetSolutionGlobalSectionBuilder();
205
206 _associatedEntryGroupBuilderStack.Push(new AssociatedEntryGroupBuilder(
207 openAssociatedGroupToken,
208 builtGroup =>
209 {
210 localDotNetSolutionGlobalSectionBuilder.AssociatedEntryGroup = builtGroup;
211
212 var outList = _dotNetSolutionGlobal.DotNetSolutionGlobalSectionList.Add(
213 localDotNetSolutionGlobalSectionBuilder.Build());
214
215 _dotNetSolutionGlobal = _dotNetSolutionGlobal with
216 {
217 DotNetSolutionGlobalSectionList = outList
218 };
219 }));
220
221 localDotNetSolutionGlobalSectionBuilder.GlobalSectionArgument =
222 (AssociatedValueToken)_tokenWalker.Match(SyntaxKind.AssociatedValueToken);
223
224 localDotNetSolutionGlobalSectionBuilder.GlobalSectionOrder =
225 (AssociatedValueToken)_tokenWalker.Match(SyntaxKind.AssociatedValueToken);
226 }
227 else if (openAssociatedGroupToken.TextSpan.GetText() == LexSolutionFacts.Project.PROJECT_DEFINITION_START_TOKEN)
228 {
229 _associatedEntryGroupBuilderStack.Push(new AssociatedEntryGroupBuilder(
230 openAssociatedGroupToken,
231 builtGroup =>
232 {
233 if (builtGroup.AssociatedEntryList.Length == 4)
234 {
235 var i = 0;
236
237 var projectTypeGuidAssociatedPair = builtGroup.AssociatedEntryList[i++] as AssociatedEntryPair;
238 var displayNameAssociatedPair = builtGroup.AssociatedEntryList[i++] as AssociatedEntryPair;
239 var relativePathFromSolutionFileStringAssociatedPair = builtGroup.AssociatedEntryList[i++] as AssociatedEntryPair;
240 var projectIdGuidAssociatedPair = builtGroup.AssociatedEntryList[i++] as AssociatedEntryPair;
241
242 _ = Guid.TryParse(projectTypeGuidAssociatedPair.AssociatedValueToken.TextSpan.GetText(), out var projectTypeGuid);
243 var displayName = displayNameAssociatedPair.AssociatedValueToken.TextSpan.GetText();
244 var relativePathFromSolutionFileString = relativePathFromSolutionFileStringAssociatedPair.AssociatedValueToken.TextSpan.GetText();
245 _ = Guid.TryParse(projectIdGuidAssociatedPair.AssociatedValueToken.TextSpan.GetText(), out var projectIdGuid);
246
247 IDotNetProject dotNetProject;
248
249 if (projectTypeGuid == SolutionFolder.SolutionFolderProjectTypeGuid)
250 {
251 dotNetProject = new SolutionFolder(
252 displayName,
253 projectTypeGuid,
254 relativePathFromSolutionFileString,
255 projectIdGuid,
256 builtGroup.OpenAssociatedGroupToken,
257 builtGroup.CloseAssociatedGroupToken,
258 null);
259 }
260 else
261 {
262 dotNetProject = new CSharpProjectModel(
263 displayName,
264 projectTypeGuid,
265 relativePathFromSolutionFileString,
266 projectIdGuid,
267 builtGroup.OpenAssociatedGroupToken,
268 builtGroup.CloseAssociatedGroupToken,
269 null);
270 }
271
272 _dotNetProjectList.Add(dotNetProject);
273 }
274
275 _noParentHavingAssociatedEntryGroup = builtGroup;
276 }));
277 }
278 else if (parent is not null)
279 {
280 _associatedEntryGroupBuilderStack.Push(new AssociatedEntryGroupBuilder(
281 openAssociatedGroupToken,
282 builtGroup =>
283 {
284 parent.AssociatedEntryList.Add(builtGroup);
285 }));
286 }
287 }

289 private void ParseCloseAssociatedGroupToken(CloseAssociatedGroupToken closeAssociatedGroupToken)
290 {
291 if (closeAssociatedGroupToken.TextSpan.GetText() == LexSolutionFacts.Global.END_TOKEN)
292 {
293 _dotNetSolutionGlobal = _dotNetSolutionGlobal with
294 {
295 CloseAssociatedGroupToken = closeAssociatedGroupToken
296 };
297 }
298 else if (closeAssociatedGroupToken.TextSpan.GetText() == LexSolutionFacts.GlobalSection.END_TOKEN)
299 {
300 var associatedEntryGroupBuilder = _associatedEntryGroupBuilderStack.Pop();
301 associatedEntryGroupBuilder.CloseAssociatedGroupToken = closeAssociatedGroupToken;
302
303 associatedEntryGroupBuilder.Build();
304 }
305 else
306 {
307 var associatedEntryGroupBuilder = _associatedEntryGroupBuilderStack.Pop();
308 associatedEntryGroupBuilder.CloseAssociatedGroupToken = closeAssociatedGroupToken;
309
310 associatedEntryGroupBuilder.Build();
311 }
