95 private async Task HandleCommand(TerminalCommandRequest terminalCommandRequest)
96 {
97 TerminalOutput.ClearHistoryWhenExistingOutputTooLong();
98
99 var parsedCommand = await TerminalInteractive.TryHandleCommand(terminalCommandRequest);
100 ActiveTerminalCommandParsed = parsedCommand;
101
102 if (parsedCommand is null)
103 return;
104
105 try
106 {
107 HasExecutingProcess = true;
108
109 if (parsedCommand.SourceTerminalCommandRequest.BeginWithFunc is not null)
110 {
111 await parsedCommand.SourceTerminalCommandRequest.BeginWithFunc
112 .Invoke(parsedCommand)
113 .ConfigureAwait(false);
114 }
115
116 await ExecuteWebsiteCliAsync(parsedCommand, HandleOutput, _commandCancellationTokenSource.Token)
117 .ConfigureAwait(false);
118 }
119 catch (Exception e)
120 {
121 // TODO: This will erroneously write 'StartedCommandEvent' out twice...
122 // ...unless a check is added to see WHEN the exception was thrown.
123 TerminalOutput.WriteOutput(parsedCommand, new StartedCommandEvent(-1));
124
125 TerminalOutput.WriteOutput(
126 parsedCommand,
127 new StandardErrorCommandEvent(parsedCommand.SourceTerminalCommandRequest.CommandText + " threw an exception" + "\n"));
128
129 NotificationHelper.DispatchError("Terminal Exception", e.ToString(), _commonComponentRenderers, _dispatcher, TimeSpan.FromSeconds(14));
130 }
131 finally
132 {
133 if (parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc is not null)
134 {
135 await parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc
136 .Invoke(parsedCommand)
137 .ConfigureAwait(false);
138 }
139
140 HasExecutingProcess = false;
141 }

184 private Task DotNetWebsiteCliAsync(
185 TerminalCommandParsed parsedCommand,
186 Action<CommandEvent> handleOutputAction,
187 CancellationToken cancellationToken)
188 {
189 var argumentList = parsedCommand.Arguments.Trim().Split(' ');
190
191 var firstArgument = argumentList.FirstOrDefault();
192
193 if (string.IsNullOrWhiteSpace(firstArgument))
194 {
195 TerminalOutput.WriteOutput(parsedCommand, new StandardErrorCommandEvent($"firstArgument was null or whitespace."));
196 }
197
198 switch (firstArgument)
199 {
200 case TerminalWebsiteFacts.InitialArguments.RUN:
201 {
202 var projectPath = (string?)null;
203
204 if (argumentList.Length == 2)
205 {
206 projectPath = argumentList[1];
207 }
208 else if (argumentList.Length == 3)
209 {
210 if (argumentList[1] != TerminalWebsiteFacts.Options.PROJECT)
211 {
212 TerminalOutput.WriteOutput(parsedCommand, new StandardErrorCommandEvent($"argumentList[1]:{argumentList[1]} != {TerminalWebsiteFacts.Options.PROJECT}"));
213 return Task.CompletedTask;
214 }
215
216 projectPath = argumentList[2];
217 }
218 else
219 {
220 TerminalOutput.WriteOutput(parsedCommand, new StandardErrorCommandEvent($"Bad argument count: '{argumentList.Length}'."));
221 return Task.CompletedTask;
222 }
223
224 TerminalOutput.WriteOutput(parsedCommand, new StandardErrorCommandEvent($"projectPath:{projectPath}"));
225 return Task.CompletedTask;
226 }
227 default:
228 {
229 TerminalOutput.WriteOutput(parsedCommand, new StandardErrorCommandEvent($"First argument: '{firstArgument}' was not recognized."));
230 return Task.CompletedTask;
231 }
232 }
233
234 throw new LuthetusIdeException($"The method '{nameof(DotNetWebsiteCliAsync)}' should return on each branch of the switch statement.");
235 }
