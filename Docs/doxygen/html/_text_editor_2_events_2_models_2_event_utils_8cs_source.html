124 public static bool CheckIfKeymapArgsMapsToCommand(
125 TextEditorComponentData componentData,
126 KeymapArgs keymapArgs,
127 bool hasSelection,
128 ITextEditorService textEditorService,
129 out Key<KeymapLayer> layerKey,
130 out bool success,
131 out CommandNoType command)
132 {
133 layerKey = ((ITextEditorKeymap)componentData.Options.Keymap!).GetLayer(hasSelection);
134
135 keymapArgs.LayerKey = layerKey;
136
137 success = ((ITextEditorKeymap)componentData.Options.Keymap!).TryMap(
138 keymapArgs,
139 componentData,
140 out command);
141
142 if (!success && keymapArgs.LayerKey != TextEditorKeymapDefaultFacts.DefaultLayer.Key)
143 {
144 keymapArgs.LayerKey = TextEditorKeymapDefaultFacts.DefaultLayer.Key;
145
146 _ = ((ITextEditorKeymap)componentData.Options.Keymap!).TryMap(
147 keymapArgs,
148 componentData,
149 out command);
150 }
151
152 if (KeyboardKeyFacts.WhitespaceCodes.ENTER_CODE == keymapArgs.Code && keymapArgs.ShiftKey)
153 command = TextEditorCommandDefaultFacts.NewLineBelow;
154
155 return command is not null;
156 }

163 public static bool IsAutocompleteMenuInvoker(KeymapArgs keymapArgs)
164 {
165 // LetterOrDigit was hit without Ctrl being held
166 return !keymapArgs.CtrlKey &&
167 !KeyboardKeyFacts.IsWhitespaceCode(keymapArgs.Code) &&
168 !KeyboardKeyFacts.IsMetaKey(keymapArgs);

171 public static bool IsSyntaxHighlightingInvoker(KeymapArgs keymapArgs)
172 {
173 return keymapArgs.Key == ";" ||
174 KeyboardKeyFacts.IsWhitespaceCode(keymapArgs.Code) ||
175 keymapArgs.CtrlKey && keymapArgs.Key == "s" ||
176 keymapArgs.CtrlKey && keymapArgs.Key == "v" ||
177 keymapArgs.CtrlKey && keymapArgs.Key == "z" ||
178 keymapArgs.CtrlKey && keymapArgs.Key == "y";

193 public static async Task<(int rowIndex, int columnIndex)> CalculateRowAndColumnIndex(
194 ResourceUri resourceUri,
195 Key<TextEditorViewModel> viewModelKey,
196 MouseEventArgs mouseEventArgs,
197 TextEditorComponentData componentData,
198 ITextEditorEditContext editContext)
199 {
200 var modelModifier = editContext.GetModelModifier(resourceUri);
201 var viewModelModifier = editContext.GetViewModelModifier(viewModelKey);
202 var globalTextEditorOptions = editContext.TextEditorService.OptionsStateWrap.Value.Options;
203
204 if (modelModifier is null || viewModelModifier is null)
205 return (0, 0);
206
207 var charMeasurements = viewModelModifier.ViewModel.CharAndLineMeasurements;
208 var textEditorDimensions = viewModelModifier.ViewModel.TextEditorDimensions;
209 var scrollbarDimensions = viewModelModifier.ViewModel.ScrollbarDimensions;
210
211 var relativeX = mouseEventArgs.ClientX - textEditorDimensions.BoundingClientRectLeft;
212 var relativeY = mouseEventArgs.ClientY - textEditorDimensions.BoundingClientRectTop;
213
214 var positionX = relativeX;
215 var positionY = relativeY;
216
217 // Scroll position offset
218 {
219 positionX += scrollbarDimensions.ScrollLeft;
220 positionY += scrollbarDimensions.ScrollTop;
221 }
222
223 var rowIndex = (int)(positionY / charMeasurements.LineHeight);
224
225 rowIndex = rowIndex > modelModifier.LineCount - 1
226 ? modelModifier.LineCount - 1
227 : rowIndex;
228
229 int columnIndexInt;
230
231
232 var columnIndexDouble = positionX / charMeasurements.CharacterWidth;
233 columnIndexInt = (int)Math.Round(columnIndexDouble, MidpointRounding.AwayFromZero);
234
235 var lengthOfRow = modelModifier.GetLineLength(rowIndex);
236
237 // Tab key column offset
238 {
239 var parameterForGetTabsCountOnSameRowBeforeCursor = columnIndexInt > lengthOfRow
240 ? lengthOfRow
241 : columnIndexInt;
242
243 int tabsOnSameRowBeforeCursor;
244
245 try
246 {
247 tabsOnSameRowBeforeCursor = modelModifier.GetTabCountOnSameLineBeforeCursor(
248 rowIndex,
249 parameterForGetTabsCountOnSameRowBeforeCursor);
250 }
251 catch (LuthetusTextEditorException)
252 {
253 tabsOnSameRowBeforeCursor = 0;
254 }
255
256 // 1 of the character width is already accounted for
257 var extraWidthPerTabKey = TextEditorModel.TAB_WIDTH - 1;
258
259 columnIndexInt -= extraWidthPerTabKey * tabsOnSameRowBeforeCursor;
260 }
261
262 columnIndexInt = columnIndexInt > lengthOfRow
263 ? lengthOfRow
264 : columnIndexInt;
265
266 rowIndex = Math.Max(rowIndex, 0);
267 columnIndexInt = Math.Max(columnIndexInt, 0);
268
269 return (rowIndex, columnIndexInt);
270 }
