32 public void SelectRowEndingKindOnChange(ChangeEventArgs changeEventArgs)
33 {
34 var renderBatchLocal = _renderBatch;
35 if (renderBatchLocal is null)
36 return;
37
38 var model = renderBatchLocal.Model;
39 var viewModel = renderBatchLocal.ViewModel;
40
41 if (model is null || viewModel is null)
42 return;
43
44 var rowEndingKindString = (string)(changeEventArgs.Value ?? string.Empty);
45
46 if (Enum.TryParse<LineEndKind>(rowEndingKindString, out var rowEndingKind))
47 {
48 _root.TextEditorService.PostRedundant(
49 nameof(TextEditorService.ModelApi.SetUsingLineEndKind),
50 viewModel.ResourceUri,
51 viewModel.ViewModelKey,
52 editContext =>
53 {
54 var modelModifier = editContext.GetModelModifier(viewModel.ResourceUri);
55
56 if (modelModifier is null)
57 return Task.CompletedTask;
58
59 _root.TextEditorService.ModelApi.SetUsingLineEndKind(
60 editContext,
61 modelModifier,
62 rowEndingKind);
63 return Task.CompletedTask;
64 });
65 }
