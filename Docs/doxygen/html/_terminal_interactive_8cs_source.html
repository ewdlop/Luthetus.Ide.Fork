30 public async Task<TerminalCommandParsed?> TryHandleCommand(TerminalCommandRequest terminalCommandRequest)

37
38 _terminalCommandRequestHistory.Insert(0, terminalCommandRequest);

41 var parsedCommand = Parse(terminalCommandRequest);
42
43 // To set the working directory, is not mutually exclusive
44 // to the "cd" command. Do not combine these.
45 if (terminalCommandRequest.WorkingDirectory is not null &&
46 terminalCommandRequest.WorkingDirectory != WorkingDirectory)
47 {
48 SetWorkingDirectory(terminalCommandRequest.WorkingDirectory);
49 }
50
51 if (parsedCommand.TargetFileName.StartsWith(RESERVED_TARGET_FILENAME_PREFIX))

53 _terminal.TerminalOutput.WriteOutput(

55 new StartedCommandEvent(-1));

58 return null;
59 }
60
61 switch (parsedCommand.TargetFileName)
62 {
63 case "cd":
64 _terminal.TerminalOutput.WriteOutput(
65 parsedCommand,
66 new StartedCommandEvent(-1));
67
68 SetWorkingDirectory(parsedCommand.Arguments);

70 _terminal.TerminalOutput.WriteOutput(
71 parsedCommand,
72 new StandardOutputCommandEvent($"WorkingDirectory set to: '{parsedCommand.Arguments}'\n"));
73 return null;
74 case "clear":
75 _terminal.TerminalOutput.ClearOutput();
76 return null;
77 default:

99 public TerminalCommandParsed Parse(TerminalCommandRequest terminalCommandRequest)
100 {
101 try
102 {
103 var stringWalker = new StringWalker(ResourceUri.Empty, terminalCommandRequest.CommandText);
104
105 // Get target file name
106 string targetFileName;
107 {
108 var targetFileNameBuilder = new StringBuilder();
109 var startPositionIndex = stringWalker.PositionIndex;
110
111 while (!stringWalker.IsEof)
112 {
113 if (WhitespaceFacts.ALL_LIST.Contains(stringWalker.CurrentCharacter))
114 break;
115 else
116 targetFileNameBuilder.Append(stringWalker.CurrentCharacter);
117
118 _ = stringWalker.ReadCharacter();
119 }
120
121 targetFileName = targetFileNameBuilder.ToString();
122 }
123
124 // Get arguments
125 stringWalker.ReadWhitespace();
126 var arguments = stringWalker.RemainingText;
127
128 return new TerminalCommandParsed(

130 arguments,
131 terminalCommandRequest);
132 }
