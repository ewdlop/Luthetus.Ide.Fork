65 protected override void OnInitialized()
66 {
67 BackgroundTaskService.Enqueue(
68 Key<IBackgroundTask>.NewKey(),
69 ContinuousBackgroundTaskWorker.GetQueueKey(),
70 nameof(LuthetusIdeInitializer),
71 async () =>
72 {
73 if (TextEditorConfig.CustomThemeRecordList is not null)
74 {
75 foreach (var themeRecord in TextEditorConfig.CustomThemeRecordList)
76 {
77 Dispatcher.Dispatch(new ThemeState.RegisterAction(themeRecord));
78 }
79 }
80
81 foreach (var terminalKey in TerminalFacts.WELL_KNOWN_KEYS)
82 {
83 if (terminalKey == TerminalFacts.GENERAL_KEY)
84 AddGeneralTerminal();
85 else if (terminalKey == TerminalFacts.EXECUTION_KEY)
86 AddExecutionTerminal();
87 }
88
89 Dispatcher.Dispatch(new CodeSearchState.InitializeResizeHandleDimensionUnitAction(
90 new DimensionUnit
91 {
92 ValueFunc = () => AppOptionsStateWrap.Value.Options.ResizeHandleHeightInPixels / 2,
93 DimensionUnitKind = DimensionUnitKind.Pixels,
94 DimensionOperatorKind = DimensionOperatorKind.Subtract,
95 Purpose = DimensionUnitFacts.Purposes.RESIZABLE_HANDLE_ROW,
96 }));
97
98 InitializePanelResizeHandleDimensionUnit();
99 InitializePanelTabs();
100 CommandFactory.Initialize();
101 });
102
103 base.OnInitialized();
104 }
