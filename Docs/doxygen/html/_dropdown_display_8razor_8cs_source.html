67 protected override async Task OnAfterRenderAsync(bool firstRender)
68 {
69#if DEBUG
70 AssertInfiniteRenderLoop();
71#endif
72
73 if (firstRender)
74 {
75 // Force the initial invocation (as opposed to waiting for the event)
76 await RemeasureAndRerender();
77 }
78 else if (LuthetusHostingInformation.LuthetusPurposeKind == LuthetusPurposeKind.Ide && _hasPendingEvent)
79 {
80 /*
81 'LuthetusPurposeKind.Ide' hack for Text Editor NuGet Package (2024-08-28)
82 ========================================================================
83 The dropdown currently relies on the html element id: 'ContextFacts.GlobalContext.ContextElementId'.
84
85 This element id is used by the IDE, but its unlikely to appear in someone's app who is using the
86 text editor NuGet package.
87
88 TODO: Add a configuration option that allows the currently hardcoded, 'ContextFacts.GlobalContext.ContextElementId',
89 to instead be specified as other values.
90 */
91
92 if (_isOffScreenHorizontally || _isOffScreenVertically)
93 {
94 lock (_hasPendingEventLock)
95 {
96 _hasPendingEvent = false;
97 }
98
99 var inDropdown = Dropdown;
100
101 var outWidth = inDropdown.Width;
102 var outHeight = inDropdown.Height;
103 var outLeft = inDropdown.Left;
104 var outTop = inDropdown.Top;
105
106 if (_isOffScreenHorizontally) // These 'if'(s) are not mutually exclusive
107 {
108 var horizontalPadding = 5;
109
110 outLeft = Math.Max(
111 0,
112 (_globalHtmlElementDimensions?.WidthInPixels ?? 0) -
113 (_htmlElementDimensions?.WidthInPixels ?? 0) -
114 horizontalPadding);
115 }
116
117 if (_isOffScreenVertically) // These 'if'(s) are not mutually exclusive
118 {
119 var verticalPadding = 5;
120
121 outTop = Math.Max(
122 0,
123 (_globalHtmlElementDimensions?.HeightInPixels ?? 0) -
124 (_htmlElementDimensions?.HeightInPixels ?? 0) -
125 verticalPadding);
126 }
127
128 var outDropdown = inDropdown with
129 {
130 Width = outWidth,
131 Height = outHeight,
132 Left = outLeft,
133 Top = outTop
134 };
135
136 Dispatcher.Dispatch(new DropdownState.FitOnScreenAction(outDropdown));
137 }
138 }
139
140 await base.OnAfterRenderAsync(firstRender);
141 }
