307 out ServiceProvider serviceProvider)
308 {

319 typeof(LuthetusTextEditorConfig).Assembly));
320
321 serviceProvider = serviceCollection.BuildServiceProvider();
322
323 var store = serviceProvider.GetRequiredService<IStore>();
