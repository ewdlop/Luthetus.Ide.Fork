105 public async Task RegisterModelFunc(RegisterModelArgs registerModelArgs)
106 {
107 var model = _textEditorService.ModelApi.GetOrDefault(registerModelArgs.ResourceUri);
108
109 if (model is null)
110 {
111 var resourceUri = registerModelArgs.ResourceUri;
112
113 var fileLastWriteTime = await _fileSystemProvider.File
114 .GetLastWriteTimeAsync(resourceUri.Value)
115 .ConfigureAwait(false);
116
117 var content = await _fileSystemProvider.File
118 .ReadAllTextAsync(resourceUri.Value)
119 .ConfigureAwait(false);
120
121 var absolutePath = _environmentProvider.AbsolutePathFactory(resourceUri.Value, false);
122 var decorationMapper = _decorationMapperRegistry.GetDecorationMapper(absolutePath.ExtensionNoPeriod);
123 var compilerService = _compilerServiceRegistry.GetCompilerService(absolutePath.ExtensionNoPeriod);
124
125 model = new TextEditorModel(
126 resourceUri,
127 fileLastWriteTime,
128 absolutePath.ExtensionNoPeriod,
129 content,
130 decorationMapper,
131 compilerService);
132
133 var modelModifier = new TextEditorModelModifier(model);
134 modelModifier.PerformRegisterPresentationModelAction(CompilerServiceDiagnosticPresentationFacts.EmptyPresentationModel);
135 modelModifier.PerformRegisterPresentationModelAction(FindOverlayPresentationFacts.EmptyPresentationModel);
136 modelModifier.PerformRegisterPresentationModelAction(DiffPresentationFacts.EmptyInPresentationModel);
137 modelModifier.PerformRegisterPresentationModelAction(DiffPresentationFacts.EmptyOutPresentationModel);
138
139 model = modelModifier.ToModel();
140
141 _textEditorService.ModelApi.RegisterCustom(model);
142
143 model.CompilerService.RegisterResource(
144 model.ResourceUri,
145 shouldTriggerResourceWasModified: false);
146
147 var uniqueTextEditorWork = new UniqueTextEditorWork(
148 nameof(compilerService.ParseAsync),
149 editContext =>
150 {
151 var modelModifier = editContext.GetModelModifier(resourceUri);
152
153 if (modelModifier is null)
154 return Task.CompletedTask;
155
156 return compilerService.ParseAsync(editContext, modelModifier);
157 });
158
159 if (registerModelArgs.ShouldBlockUntilBackgroundTaskIsCompleted)
160 await _textEditorService.PostAsync(uniqueTextEditorWork).ConfigureAwait(false);
161 else
162 _textEditorService.Post(uniqueTextEditorWork);
163 }
164
165 await CheckIfContentsWereModifiedAsync(
166 _dispatcher,
167 registerModelArgs.ResourceUri.Value,
168 model)
169 .ConfigureAwait(false);
170 }

172 public Task<Key<TextEditorViewModel>> TryRegisterViewModelFunc(TryRegisterViewModelArgs registerViewModelArgs)
173 {
174 try
175 {
176 var model = _textEditorService.ModelApi.GetOrDefault(registerViewModelArgs.ResourceUri);
177
178 if (model is null)
179 {
180 NotificationHelper.DispatchDebugMessage(nameof(TryRegisterViewModelFunc), () => "model is null: " + registerViewModelArgs.ResourceUri.Value, _commonComponentRenderers, _dispatcher, TimeSpan.FromSeconds(4));
181 return Task.FromResult(Key<TextEditorViewModel>.Empty);
182 }
183
184 var viewModel = _textEditorService.ModelApi
185 .GetViewModelsOrEmpty(registerViewModelArgs.ResourceUri)
186 .FirstOrDefault(x => x.Category == registerViewModelArgs.Category);
187
188 if (viewModel is not null)
189 return Task.FromResult(viewModel.ViewModelKey);
190
191 var viewModelKey = Key<TextEditorViewModel>.NewKey();
192
193 viewModel = new TextEditorViewModel(
194 viewModelKey,
195 registerViewModelArgs.ResourceUri,
196 _textEditorService,
197 _dispatcher,
198 _dialogService,
199 _jsRuntime,
200 VirtualizationResult<List<RichCharacter>>.GetEmptyRichCharacters(),
201 new TextEditorDimensions(0, 0, 0, 0),
202 new ScrollbarDimensions(0, 0, 0, 0, 0),
203 new CharAndLineMeasurements(0, 0),
204 false,
205 registerViewModelArgs.Category);
206
207 var firstPresentationLayerKeys = new[]
208 {
209 CompilerServiceDiagnosticPresentationFacts.PresentationKey,
210 FindOverlayPresentationFacts.PresentationKey,
211 }.ToImmutableArray();
212
213 var absolutePath = _environmentProvider.AbsolutePathFactory(
214 registerViewModelArgs.ResourceUri.Value,
215 false);
216
217 viewModel.UnsafeState.ShouldSetFocusAfterNextRender = registerViewModelArgs.ShouldSetFocusToEditor;
218
219 viewModel = viewModel with
220 {
221 OnSaveRequested = HandleOnSaveRequested,
222 GetTabDisplayNameFunc = _ => absolutePath.NameWithExtension,
223 FirstPresentationLayerKeysList = firstPresentationLayerKeys.ToImmutableList()
224 };
225
226 _textEditorService.ViewModelApi.Register(viewModel);
227
228 return Task.FromResult(viewModelKey);

230 void HandleOnSaveRequested(ITextEditorModel innerTextEditor)
231 {
232 var innerContent = innerTextEditor.GetAllText();
233
234 var cancellationToken = model.TextEditorSaveFileHelper.GetCancellationToken();
235
236 _ideBackgroundTaskApi.FileSystem.SaveFile(
237 absolutePath,
238 innerContent,
239 writtenDateTime =>
240 {
241 if (writtenDateTime is not null)
242 {
243 _textEditorService.PostUnique(
244 nameof(HandleOnSaveRequested),
245 editContext =>
246 {
247 var modelModifier = editContext.GetModelModifier(innerTextEditor.ResourceUri);
248 if (modelModifier is null)
249 return Task.CompletedTask;
250
251 _textEditorService.ModelApi.SetResourceData(
252 editContext,
253 modelModifier,
254 writtenDateTime.Value);
255 return Task.CompletedTask;
256 });
257 }
258
259 return Task.CompletedTask;
260 },
261 cancellationToken);
262 }
263 }
264 catch (Exception e)
265 {
266 NotificationHelper.DispatchError(
267 nameof(TryRegisterViewModelFunc),
268 e.ToString(),
269 _commonComponentRenderers,
270 _dispatcher,
271 TimeSpan.FromSeconds(6));
272 return Task.FromResult(Key<TextEditorViewModel>.Empty);
273 }

276 public async Task<bool> TryShowViewModelFunc(TryShowViewModelArgs showViewModelArgs)
277 {
278 _textEditorService.GroupApi.Register(EditorTextEditorGroupKey);
279
280 var viewModel = _textEditorService.ViewModelApi.GetOrDefault(showViewModelArgs.ViewModelKey);
281
282 if (viewModel is null)
283 return false;
284
285 if (viewModel.Category == new Category("main") &&
286 showViewModelArgs.GroupKey == Key<TextEditorGroup>.Empty)
287 {
288 showViewModelArgs = new TryShowViewModelArgs(
289 showViewModelArgs.ViewModelKey,
290 EditorTextEditorGroupKey,
291 showViewModelArgs.ShouldSetFocusToEditor,
292 showViewModelArgs.ServiceProvider);
293 }
294
295 if (showViewModelArgs.ViewModelKey == Key<TextEditorViewModel>.Empty ||
296 showViewModelArgs.GroupKey == Key<TextEditorGroup>.Empty)
297 {
298 return false;
299 }
300
301 _textEditorService.GroupApi.AddViewModel(
302 showViewModelArgs.GroupKey,
303 showViewModelArgs.ViewModelKey);
304
305 _textEditorService.GroupApi.SetActiveViewModel(
306 showViewModelArgs.GroupKey,
307 showViewModelArgs.ViewModelKey);
308
309 if (showViewModelArgs.ShouldSetFocusToEditor)
310 {
311 _textEditorService.PostUnique(nameof(TryShowViewModelFunc), editContext =>
312 {
313 var viewModelModifier = editContext.GetViewModelModifier(showViewModelArgs.ViewModelKey);
314
315 viewModelModifier.ViewModel.UnsafeState.ShouldSetFocusAfterNextRender =
316 showViewModelArgs.ShouldSetFocusToEditor;
317
318 return viewModel.FocusAsync();
319 });
320 }
321
322 return true;
323 }

325 private async Task CheckIfContentsWereModifiedAsync(
326 IDispatcher dispatcher,
327 string inputFileAbsolutePathString,
328 TextEditorModel textEditorModel)
329 {
330 var fileLastWriteTime = await _fileSystemProvider.File
331 .GetLastWriteTimeAsync(inputFileAbsolutePathString)
332 .ConfigureAwait(false);
333
334 if (fileLastWriteTime > textEditorModel.ResourceLastWriteTime &&
335 _ideComponentRenderers.BooleanPromptOrCancelRendererType is not null)
336 {
337 var notificationInformativeKey = Key<IDynamicViewModel>.NewKey();
338
339 var notificationInformative = new NotificationViewModel(
340 notificationInformativeKey,
341 "File contents were modified on disk",
342 _ideComponentRenderers.BooleanPromptOrCancelRendererType,
343 new Dictionary<string, object?>
344 {
345 {
346 nameof(IBooleanPromptOrCancelRendererType.Message),
347 "File contents were modified on disk"
348 },
349 {
350 nameof(IBooleanPromptOrCancelRendererType.AcceptOptionTextOverride),
351 "Reload"
352 },
353 {
354 nameof(IBooleanPromptOrCancelRendererType.OnAfterAcceptFunc),
355 new Func<Task>(() =>
356 {
357 _backgroundTaskService.Enqueue(
358 Key<IBackgroundTask>.NewKey(),
359 ContinuousBackgroundTaskWorker.GetQueueKey(),
360 "Check If Contexts Were Modified",
361 async () =>
362 {
363 dispatcher.Dispatch(new NotificationState.DisposeAction(
364 notificationInformativeKey));
365
366 var content = await _fileSystemProvider.File
367 .ReadAllTextAsync(inputFileAbsolutePathString)
368 .ConfigureAwait(false);
369
370 _textEditorService.PostUnique(
371 nameof(CheckIfContentsWereModifiedAsync),
372 editContext =>
373 {
374 var modelModifier = editContext.GetModelModifier(textEditorModel.ResourceUri);
375 if (modelModifier is null)
376 return Task.CompletedTask;
377
378 _textEditorService.ModelApi.Reload(
379 editContext,
380 modelModifier,
381 content,
382 fileLastWriteTime);
383
384 editContext.TextEditorService.ModelApi.ApplySyntaxHighlighting(
385 editContext,
386 modelModifier);
387 return Task.CompletedTask;
388 });
389 });
390 return Task.CompletedTask;
391 })
392 },
393 {
394 nameof(IBooleanPromptOrCancelRendererType.OnAfterDeclineFunc),
395 new Func<Task>(() =>
396 {
397 dispatcher.Dispatch(new NotificationState.DisposeAction(
398 notificationInformativeKey));
399
400 return Task.CompletedTask;
401 })
402 },
403 },
404 TimeSpan.FromSeconds(20),
405 true,
406 null);
407
408 dispatcher.Dispatch(new NotificationState.RegisterAction(
409 notificationInformative));
410 }
