1134 private void WriteChildrenIndentedRecursive(ISyntaxNode node, string name = "node", int indentation = 0)
1135 {
1136 var indentationStringBuilder = new StringBuilder();
1137 for (int i = 0; i < indentation; i++)
1138 indentationStringBuilder.Append('\t');
1139
1140 Console.WriteLine($"{indentationStringBuilder.ToString()}{node.SyntaxKind}");
1141
1142 // For the child tokens
1143 indentationStringBuilder.Append('\t');
1144 var childIndentation = indentationStringBuilder.ToString();
1145
1146 foreach (var child in node.GetChildList())
1147 {
1148 if (child is ISyntaxNode syntaxNode)
1149 {
1150 WriteChildrenIndentedRecursive(syntaxNode, "node", indentation + 1);
1151 }
1152 else if (child is ISyntaxToken syntaxToken)
1153 {
1154 Console.WriteLine($"{childIndentation}{child.SyntaxKind}__{syntaxToken.TextSpan.GetText()}");
1155 }
1156 }
1157
1158 if (indentation == 0)
1159 Console.WriteLine();
1160 }
