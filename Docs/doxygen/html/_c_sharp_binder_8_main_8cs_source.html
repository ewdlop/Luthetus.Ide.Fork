175 }
176 }

180 }
181 }

665 public void CloseScope(
666 TextEditorTextSpan textSpan,
667 CSharpParserModel model)
668 {
669 // Check if it is the global scope, if so return early.
670 {
671 if (model.BinderSession.CurrentScopeIndexKey == 0)
672 return;
673 }
674
675 var inBuilder = model.CurrentCodeBlockBuilder;
676 var inOwner = inBuilder.CodeBlockOwner;
677
678 var outBuilder = model.CurrentCodeBlockBuilder.Parent;
679 var outOwner = outBuilder?.CodeBlockOwner;
680
681 // Update Scope
682 {
683 var scope = model.BinderSession.ScopeList[model.BinderSession.CurrentScopeIndexKey];
684 scope.EndingIndexExclusive = textSpan.EndingIndexExclusive;
685 model.BinderSession.ScopeList[model.BinderSession.CurrentScopeIndexKey] = scope;
686
687 // Restore Parent Scope
688 if (scope.ParentIndexKey is not null)
689 {
690 model.BinderSession.CurrentScopeIndexKey = scope.ParentIndexKey.Value;
691 }
692 }
693
694 // Update CodeBlockOwner
695 if (inOwner is not null)
696 {
697 inOwner.SetCodeBlockNode(inBuilder.Build(), model);
698
699 if (inOwner.SyntaxKind == SyntaxKind.NamespaceStatementNode)
700 model.Binder.BindNamespaceStatementNode((NamespaceStatementNode)inOwner, model);
701 else if (inOwner.SyntaxKind == SyntaxKind.TypeDefinitionNode)
702 model.Binder.BindTypeDefinitionNode((TypeDefinitionNode)inOwner, model, true);
703
704 // Restore Parent CodeBlockBuilder
705 if (outBuilder is not null)
706 {
707 model.CurrentCodeBlockBuilder = outBuilder;
708 outBuilder.InnerPendingCodeBlockOwner = null;
709
710 if (inOwner.SyntaxKind != SyntaxKind.TryStatementTryNode &&
711 inOwner.SyntaxKind != SyntaxKind.TryStatementCatchNode &&
712 inOwner.SyntaxKind != SyntaxKind.TryStatementFinallyNode)
713 {
714 outBuilder.ChildList.Add(inOwner);
715 }
716 }
717 }
