69 public DotNetSolutionModelBuilder AddDotNetProject(
70 IDotNetProject dotNetProject,
71 IEnvironmentProvider environmentProvider)
72 {
73 var solutionProjectEntry = GetSolutionProjectEntry(
74 dotNetProject,
75 environmentProvider);
76
77 lock (_tokensLock)
78 {
79 IDotNetProject? newProjectToken = null;
80 var allTokens = GetAllTokens();
81
82 if (DotNetProjectList.Any())
83 {
84 // If there is an existing project token, then insert the new
85 // project token immediately after the project token.
86
87 // This conditional branch needs a newline pre-pended.
88 solutionProjectEntry = '\n' + solutionProjectEntry;
89
90 IDotNetProject? lastValidProjectToken = null;
91
92 for (var i = DotNetProjectList.Length - 1; i >= 0; i--)
93 {
94 var entry = DotNetProjectList[i];
95
96 if (entry.CloseAssociatedGroupToken is not null)
97 {
98 lastValidProjectToken = entry;
99 break;
100 }
101 }
102
103 if (lastValidProjectToken is null || lastValidProjectToken.CloseAssociatedGroupToken is null)
104 return this;
105
106 var openStartingIndexInclusive =
107 lastValidProjectToken.CloseAssociatedGroupToken.Value.TextSpan.EndingIndexExclusive;
108
109 var openEndingIndexExclusive = openStartingIndexInclusive +
110 LexSolutionFacts.Project.PROJECT_DEFINITION_START_TOKEN.Length;
111
112 var openTextSpan = lastValidProjectToken.CloseAssociatedGroupToken.Value.TextSpan with
113 {
114 StartingIndexInclusive = openStartingIndexInclusive,
115 EndingIndexExclusive = openEndingIndexExclusive + solutionProjectEntry.Length
116 };
117
118 dotNetProject.OpenAssociatedGroupToken = new OpenAssociatedGroupToken(openTextSpan);
119
120 var closeEndingIndexExclusive =
121 lastValidProjectToken.CloseAssociatedGroupToken.Value.TextSpan.EndingIndexExclusive +
122 solutionProjectEntry.Length;
123
124 var closeStartingIndexInclusive = closeEndingIndexExclusive -
125 LexSolutionFacts.Project.PROJECT_DEFINITION_END_TOKEN.Length;
126
127 var closeTextSpan = lastValidProjectToken.CloseAssociatedGroupToken.Value.TextSpan with
128 {
129 StartingIndexInclusive = closeStartingIndexInclusive,
130 EndingIndexExclusive = closeEndingIndexExclusive
131 };
132
133 dotNetProject.CloseAssociatedGroupToken = new CloseAssociatedGroupToken(closeTextSpan);
134
135 newProjectToken = dotNetProject;
136 }
137 else
138 {
139 var global = DotNetSolutionGlobal;
140
141 if (global.OpenAssociatedGroupToken is null)
142 return this;
143
144 var newProjectTextSpanStartingIndexInclusive = global.OpenAssociatedGroupToken.Value.TextSpan.StartingIndexInclusive;
145
146 var newProjectTextSpan = global.OpenAssociatedGroupToken.Value.TextSpan with
147 {
148 StartingIndexInclusive = newProjectTextSpanStartingIndexInclusive,
149 EndingIndexExclusive = newProjectTextSpanStartingIndexInclusive + solutionProjectEntry.Length
150 };
151
152 dotNetProject.OpenAssociatedGroupToken = new OpenAssociatedGroupToken(newProjectTextSpan);
153 dotNetProject.CloseAssociatedGroupToken = new CloseAssociatedGroupToken(newProjectTextSpan);
154
155 newProjectToken = dotNetProject;
156 }
157
158 DotNetProjectList = DotNetProjectList.Add(newProjectToken);
159
160 SolutionFileContents = SolutionFileContents.Insert(
161 newProjectToken.OpenAssociatedGroupToken.TextSpan.StartingIndexInclusive,
162 solutionProjectEntry);
163
164 UnsafeShiftTextAfterInsertion(
165 allTokens,
166 newProjectToken.OpenAssociatedGroupToken.TextSpan);
167 }
168
169 return this;
170 }

186 private void UnsafeShiftTextAfterInsertion(
187 ImmutableArray<(ISyntaxToken token, Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?> withFunc)> tokenTuplesToShift,
188 TextEditorTextSpan insertedTextSpan)
189 {
190 for (int i = 0; i < tokenTuplesToShift.Length; i++)
191 {
192 var tokenTuple = tokenTuplesToShift[i];
193
194 if (tokenTuple.token.TextSpan.StartingIndexInclusive >= insertedTextSpan.StartingIndexInclusive)
195 {
196 var newTextSpan = tokenTuple.token.TextSpan with
197 {
198 StartingIndexInclusive = tokenTuple.token.TextSpan.StartingIndexInclusive + insertedTextSpan.Length,
199 EndingIndexExclusive = tokenTuple.token.TextSpan.EndingIndexExclusive + insertedTextSpan.Length,
200 };
201
202 tokenTuple.withFunc.Invoke(tokenTuple.token, newTextSpan);
203 }
204 }

207 private ImmutableArray<(ISyntaxToken token, Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?> withFunc)> GetAllTokens()
208 {
209 var tokens = new List<(ISyntaxToken token, Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?> withFunc)>();
210
211 foreach (var dotNetProject in DotNetProjectList)
212 {
213 tokens.Add((
214 dotNetProject.OpenAssociatedGroupToken,
215 new Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?>((inToken, textSpan) =>
216 {
217 var outOpenAssociatedGroupToken = dotNetProject.OpenAssociatedGroupToken with
218 {
219 TextSpan = textSpan
220 };
221
222 dotNetProject.OpenAssociatedGroupToken = outOpenAssociatedGroupToken;
223 return null;
224 })));
225
226 if (dotNetProject.CloseAssociatedGroupToken is not null)
227 {
228 tokens.Add((
229 dotNetProject.CloseAssociatedGroupToken,
230 new Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?>((inToken, textSpan) =>
231 {
232 dotNetProject.CloseAssociatedGroupToken = new CloseAssociatedGroupToken(textSpan);
233 return null;
234 })));
235 }
236 }
237
238 var global = DotNetSolutionGlobal;
239
240 if (global.OpenAssociatedGroupToken is not null)
241 {
242 tokens.Add((
243 global.OpenAssociatedGroupToken,
244 new Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?>((inToken, textSpan) =>
245 {
246 global.OpenAssociatedGroupToken = new OpenAssociatedGroupToken(textSpan);
247 return null;
248 })));
249 }
250
251 if (global.CloseAssociatedGroupToken is not null)
252 {
253 tokens.Add((
254 global.CloseAssociatedGroupToken,
255 new Func<ISyntaxToken, TextEditorTextSpan, ISyntaxToken?>((inToken, textSpan) =>
256 {
257 global.CloseAssociatedGroupToken = new CloseAssociatedGroupToken(textSpan);
258 return null;
259 })));
260 }
261
262 return tokens.ToImmutableArray();
263 }
