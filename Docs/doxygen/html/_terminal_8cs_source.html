110 private async Task HandleCommand(TerminalCommandRequest terminalCommandRequest)
111 {
112 TerminalOutput.ClearHistoryWhenExistingOutputTooLong();
113
114 var parsedCommand = await TerminalInteractive.TryHandleCommand(terminalCommandRequest);
115 ActiveTerminalCommandParsed = parsedCommand;
116
117 if (parsedCommand is null)
118 return;
119
120 var cliWrapCommand = Cli.Wrap(parsedCommand.TargetFileName);
121
122 cliWrapCommand = cliWrapCommand.WithWorkingDirectory(TerminalInteractive.WorkingDirectory);
123
124 if (!string.IsNullOrWhiteSpace(parsedCommand.Arguments))
125 cliWrapCommand = cliWrapCommand.WithArguments(parsedCommand.Arguments);
126
127 // TODO: Decide where to put invocation of 'parsedCommand.SourceTerminalCommandRequest.BeginWithFunc'...
128 // ...and invocation of 'cliWrapCommand'
129 // and invocation of 'parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc'
130 // |
131 // This comment is referring to the 'try/catch' block logic.
132 // If the 'BeginWithFunc' throws an exception should the 'cliWrapCommand' run?
133 // If the 'cliWrapCommand' throws an exception should the 'ContinueWithFunc' run?
134
135 try
136 {
137 SetHasExecutingProcess(true);
138
139 if (parsedCommand.SourceTerminalCommandRequest.BeginWithFunc is not null)
140 {
141 await parsedCommand.SourceTerminalCommandRequest.BeginWithFunc
142 .Invoke(parsedCommand)
143 .ConfigureAwait(false);
144 }
145
146 await cliWrapCommand
147 .Observe(_commandCancellationTokenSource.Token)
148 .ForEachAsync(HandleOutput)
149 .ConfigureAwait(false);
150 }
151 catch (Exception e)
152 {
153 // TODO: This will erroneously write 'StartedCommandEvent' out twice...
154 // ...unless a check is added to see WHEN the exception was thrown.
155 TerminalOutput.WriteOutput(
156 parsedCommand,
157 new StartedCommandEvent(-1));
158
159 TerminalOutput.WriteOutput(
160 parsedCommand,
161 new StandardErrorCommandEvent(
162 parsedCommand.SourceTerminalCommandRequest.CommandText +
163 " threw an exception" +
164 "\n"));
165
166 NotificationHelper.DispatchError("Terminal Exception", e.ToString(), _commonComponentRenderers, _dispatcher, TimeSpan.FromSeconds(14));
167 }
168 finally
169 {
170 if (parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc is not null)
171 {
172 try
173 {
174 // The code 'SetHasExecutingProcess(false);' needs to run
175 // So, in the case that their ContinueWithFunc throws an exception
176 // make sure its wrapped in a try catch block.
177 await parsedCommand.SourceTerminalCommandRequest.ContinueWithFunc
178 .Invoke(parsedCommand)
179 .ConfigureAwait(false);
180 }
181 catch (Exception e)
182 {
183 Console.WriteLine(e);
184 }
185 }
186
187 SetHasExecutingProcess(false);
188 }
189 }
